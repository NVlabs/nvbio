96 template <typename T>
97 struct VectorArrayView
98 {
101  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
102  VectorArrayView(
103  T* arena = NULL,
104  uint32* index = NULL,
105  uint32* sizes = NULL,
106  uint32* pool = NULL,
107  uint32 size = 0u)
108  : m_arena(arena), m_index(index), m_sizes(sizes), m_pool(pool), m_size(size) {}
109 
112  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
113  T* alloc(const uint32 index, const uint32 size)
114  {
115  const uint32 slot = atomic_add( m_pool, size );
116  if (slot + size >= m_size)
117  {
118  // mark an out-of-bounds allocation
119  m_index[index] = m_size;
120  m_sizes[index] = 0u;
121  return NULL;
122  }
123  m_index[index] = slot;
124  m_sizes[index] = size;
125  return m_arena + slot;
126  }
127 
130  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
131  T* operator[](const uint32 index) const
132  {
133  // for unsuccessful allocations m_index is set to m_size - in that case we return NULL
134  return (m_index[index] < m_size) ? m_arena + m_index[index] : NULL;
135  }
136 
139  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
140  uint32 slot(const uint32 index) const { return m_index[index]; }
141 
144  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
145  uint32 size(const uint32 index) const { return m_sizes[index]; }
146 
147 public:
148  T* m_arena;
149  uint32* m_index;
150  uint32* m_sizes;
151  uint32* m_pool;
152  uint32 m_size;
153 };
