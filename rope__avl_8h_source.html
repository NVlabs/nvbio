329  {

334  m_leaves[ leaf_index ].insert( i, s );

336  }
337  else

339  #if 0

427  bool spill(uint32 node_index, const Symbol s)
428  {
429  // go up until we are in a right branch
430  uint32 parent_index = m_nodes[ node_index ].parent();
431  while (1u)
432  {
433  // if we reached the root of the tree without ever being the left child,
434  // we were in the right-most leaf and there is no place to spill
435  if (parent_index == INVALID_NODE)
436  return false;
437 
438  // check if we are finally the left child
439  if (m_nodes[ parent_index ].left() == node_index)
440  break;
441 
442  // go up one level
443  node_index = parent_index;
444  parent_index = m_nodes[ node_index ].parent();
445  }
446 
447  // go down to the left-most leaf
448  while (m_nodes[ node_index ].is_leaf() == false)
449  node_index = m_nodes[ node_index ].left();
450 
451  // fetch the leaf
452  const uint32 leaf_index = m_nodes[ node_index ].left();
453 
454  leaf_type& leaf = m_leaves[ leaf_index ];
455 
456  // check if there is enough room to spill our symbol
457  if (leaf.size() < SYMBOLS_PER_LEAF)
458  {
459  // spill!
460  leaf.insert( 0u, s );
461 
462  // update the tree weights above
463  while (node_index != INVALID_NODE)
464  {
465  reset_weight( node_index );
466 
467  node_index = m_nodes[ node_index ].parent();
468  }
469  return true;
470  }
471  // no success
472  return false;
473  }

563  // adjust the AVL property for a given node
564  void adjust_AVL(uint32& node_index)
565  {
566  // we presume the AVL class has access to AVLnode's Height data member
567  uint32& l = m_nodes[ node_index ].left();
568  uint32& r = m_nodes[ node_index ].right();
569 
570  const int32 lht = l == INVALID_NODE ? -1 : (int32)m_nodes[l].height();
571  const int32 rht = r == INVALID_NODE ? -1 : (int32)m_nodes[r].height();
572 
573  const int32 diff = lht - rht;
574 
575  // check if there's nothing to do
576  if (diff > -2 && diff < 2)
577  return;
578 
579  if (lht > rht)
580  {
581  const uint32 ll = m_nodes[ l ].left();
582  const uint32 lr = m_nodes[ l ].right();
583 
584  const int32 lck = ll == INVALID_NODE ? -1 : (int32)m_nodes[ ll ].height();
585  const int32 rck = lr == INVALID_NODE ? -1 : (int32)m_nodes[ lr ].height();
586 
587  if (lck < rck)
588  {
589  rotate_left( l ); // make left the longer
590 
591  m_nodes[ l ].set_parent( node_index );
592  }
593 
594  rotate_right( node_index ); // adjust node itself
595  }
596  else // mirror image logic to that above: right is two longer than left
597  {
598  const uint32 rl = m_nodes[ r ].left();
599  const uint32 rr = m_nodes[ r ].right();
600 
601  const int32 lck = rl == INVALID_NODE ? -1 : (int32)m_nodes[ rl ].height();

603 
604  if (lck > rck)
605  {
606  rotate_right( r ); // make right the longer
607 
608  m_nodes[ r ].set_parent( node_index );
609  }
610 
611  rotate_left( node_index ); // adjust node itself
612  }
