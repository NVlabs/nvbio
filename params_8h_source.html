87 struct SimpleFunc

91  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
92  SimpleFunc(const Type _type = LinearFunc, const float _k = 0.0f, const float _m = 1.0f) : type(_type), k(_k), m(_m) {}
93 
94  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
95  int32 operator() (const int32 x) const
96  {
97  return int32( k + m * (type == LogFunc ? logf(float(x)) :
98  type == SqrtFunc ? sqrtf(float(x)) :
99  float(x)) );
100  }
101 
102  const char* type_string() const
103  {
104  return type == LogFunc ? "log" :
105  type == SqrtFunc ? "sqrt" :
106  "linear";
107  }
108  const char* type_symbol() const
109  {
110  return type == LogFunc ? "G" :
111  type == SqrtFunc ? "S" :

113  }
