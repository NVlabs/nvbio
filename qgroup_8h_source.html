111  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
112  uint2 range(const uint64 g) const
113  {
114  const uint32 i = uint32( g / WORD_SIZE );
115  const uint32 j = uint32( g % WORD_SIZE );
116 
117  // check whether the j-th bit of I[i] is set
118  if ((I[i] & (1u << j)) == 0u)
119  return make_uint2( 0u, 0u );
120 
121  // compute j' such that bit j is the j'-th set bit in I[i]
122  const uint32 j_prime = popc( I[i] & ((1u << j) - 1u) );
123 
124  return make_uint2(
125  SS[ S[i] + j_prime ],
126  SS[ S[i] + j_prime + 1u ] );
127  }
