118 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
119 uint32 bits_per_symbol(const Alphabet alphabet)
120 {
121  return alphabet == DNA ? 2 :
122  alphabet == DNA_N ? 4 :
123  alphabet == DNA_IUPAC ? 4 :
124  alphabet == PROTEIN ? 8 :
125  alphabet == RNA ? 2 :
126  alphabet == RNA_N ? 4 :
