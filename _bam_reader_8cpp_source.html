173 // calculate bins that overlap region ( left to reference end for now )
174 int BamReader::BamReaderPrivate::BinsFromRegion(int refID, int left, uint16_t list[MAX_BIN]) {
175 
176  // get region boundaries
177  uint32_t begin = (unsigned int)left;
178  uint32_t end = (unsigned int)References.at(refID).RefLength - 1;
179 
180  // initialize list, bin '0' always a valid bin
181  int i = 0;
182  list[i++] = 0;
183 
184  // get rest of bins that contain this region
185  unsigned int k;
186  for (k = 1 + (begin>>26); k <= 1 + (end>>26); ++k) { list[i++] = k; }
187  for (k = 9 + (begin>>23); k <= 9 + (end>>23); ++k) { list[i++] = k; }
188  for (k = 73 + (begin>>20); k <= 73 + (end>>20); ++k) { list[i++] = k; }
189  for (k = 585 + (begin>>17); k <= 585 + (end>>17); ++k) { list[i++] = k; }
190  for (k = 4681 + (begin>>14); k <= 4681 + (end>>14); ++k) { list[i++] = k; }
191 
192  // return number of bins stored
193  return i;
194 }

196 // populates BAM index data structure from BAM file data
197 bool BamReader::BamReaderPrivate::BuildIndex(void) {
198 
199  // check to be sure file is open
200  if (!mBGZF.IsOpen) { return false; }
201 
202  // move file pointer to beginning of alignments
203  Rewind();
204 
205  // get reference count, reserve index space
206  int numReferences = (int)References.size();
207  for ( int i = 0; i < numReferences; ++i ) {
208  Index.push_back(ReferenceIndex());
209  }
210 
211  // sets default constant for bin, ID, offset, coordinate variables
212  const uint32_t defaultValue = 0xffffffffu;
213 
214  // bin data
215  uint32_t saveBin(defaultValue);
216  uint32_t lastBin(defaultValue);
217 
218  // reference ID data
219  int32_t saveRefID(defaultValue);
220  int32_t lastRefID(defaultValue);
221 
222  // offset data
223  uint64_t saveOffset = mBGZF.Tell();
224  uint64_t lastOffset = saveOffset;
225 
226  // coordinate data
227  int32_t lastCoordinate = defaultValue;
228 
229  BamAlignment bAlignment;
230  while( GetNextAlignment(bAlignment) ) {
231 
232  // change of chromosome, save ID, reset bin
233  if ( lastRefID != bAlignment.RefID ) {
234  lastRefID = bAlignment.RefID;
235  lastBin = defaultValue;
236  }
237 
238  // if lastCoordinate greater than BAM position - file not sorted properly
239  else if ( lastCoordinate > bAlignment.Position ) {
240  printf("BAM file not properly sorted:\n");
241  printf("Alignment %s : %d > %d on reference (id = %d)", bAlignment.Name.c_str(), lastCoordinate, bAlignment.Position, bAlignment.RefID);
242  exit(1);
243  }
244 
245  // if valid reference && BAM bin spans some minimum cutoff (smaller bin ids span larger regions)
246  if ( (bAlignment.RefID >= 0) && (bAlignment.Bin < 4681) ) {
247 
248  // save linear offset entry (matched to BAM entry refID)
249  ReferenceIndex& refIndex = Index.at(bAlignment.RefID);
250  LinearOffsetVector& offsets = refIndex.Offsets;
251  InsertLinearOffset(offsets, bAlignment, lastOffset);
252  }
253 
254  // if current BamAlignment bin != lastBin, "then possibly write the binning index"
255  if ( bAlignment.Bin != lastBin ) {
256 
257  // if not first time through
258  if ( saveBin != defaultValue ) {
259 
260  // save Bam bin entry
261  ReferenceIndex& refIndex = Index.at(saveRefID);
262  BamBinMap& binMap = refIndex.Bins;
263  InsertBinEntry(binMap, saveBin, saveOffset, lastOffset);
264  }
265 
266  // update saveOffset
267  saveOffset = lastOffset;
268 
269  // update bin values
270  saveBin = bAlignment.Bin;
271  lastBin = bAlignment.Bin;
272 
273  // update saveRefID
274  saveRefID = bAlignment.RefID;
275 
276  // if invalid RefID, break out (why?)
277  if ( saveRefID < 0 ) { break; }
278  }
279 
280  // make sure that current file pointer is beyond lastOffset
281  if ( mBGZF.Tell() <= (int64_t)lastOffset ) {
282  printf("Error in BGZF offsets.\n");
283  exit(1);
284  }
285 
286  // update lastOffset
287  lastOffset = mBGZF.Tell();
288 
289  // update lastCoordinate
290  lastCoordinate = bAlignment.Position;
291  }
292 
293  // save any leftover BAM data (as long as refID is valid)
294  if ( saveRefID >= 0 ) {
295  // save Bam bin entry
296  ReferenceIndex& refIndex = Index.at(saveRefID);
297  BamBinMap& binMap = refIndex.Bins;
298  InsertBinEntry(binMap, saveBin, saveOffset, lastOffset);
299  }
300 
301  // simplify index by merging chunks
302  MergeChunks();
303 
304  // iterate over references
305  BamIndex::iterator indexIter = Index.begin();
306  BamIndex::iterator indexEnd = Index.end();
307  for ( int i = 0; indexIter != indexEnd; ++indexIter, ++i ) {
308 
309  // get reference index data
310  ReferenceIndex& refIndex = (*indexIter);
311  BamBinMap& binMap = refIndex.Bins;
312  LinearOffsetVector& offsets = refIndex.Offsets;
313 
314  // store whether reference has alignments or no
315  References[i].RefHasAlignments = ( binMap.size() > 0 );
316 
317  // sort linear offsets
318  sort(offsets.begin(), offsets.end());
319  }
320 
321 
322  // rewind file pointer to beginning of alignments, return success/fail
323  return Rewind();
324 }

411 // calculate closest indexed file offset for region specified
412 int64_t BamReader::BamReaderPrivate::GetOffset(int refID, int left) {
413 
414  // calculate which bins overlap this region
415  uint16_t* bins = (uint16_t*)calloc(MAX_BIN, 2);
416  int numBins = BinsFromRegion(refID, left, bins);
417 
418  // get bins for this reference
419  const ReferenceIndex& refIndex = Index.at(refID);
420  const BamBinMap& binMap = refIndex.Bins;
421 
422  // get minimum offset to consider
423  const LinearOffsetVector& offsets = refIndex.Offsets;
424  uint64_t minOffset = ( (unsigned int)(left>>BAM_LIDX_SHIFT) >= offsets.size() ) ? 0 : offsets.at(left>>BAM_LIDX_SHIFT);
425 
426  // store offsets to beginning of alignment 'chunks'
427  std::vector<int64_t> chunkStarts;
428 
429  // store all alignment 'chunk' starts for bins in this region
430  for (int i = 0; i < numBins; ++i ) {
431  uint16_t binKey = bins[i];
432 
433  map<uint32_t, ChunkVector>::const_iterator binIter = binMap.find(binKey);
434  if ( (binIter != binMap.end()) && ((*binIter).first == binKey) ) {
435 
436  const ChunkVector& chunks = (*binIter).second;
437  std::vector<Chunk>::const_iterator chunksIter = chunks.begin();
438  std::vector<Chunk>::const_iterator chunksEnd = chunks.end();
439  for ( ; chunksIter != chunksEnd; ++chunksIter) {
440  const Chunk& chunk = (*chunksIter);
441  if ( chunk.Stop > minOffset ) {
442  chunkStarts.push_back( chunk.Start );
443  }
444  }
445  }
446  }
447 
448  // clean up memory
449  free(bins);
450 
451  // if no alignments found, else return smallest offset for alignment starts
452  if ( chunkStarts.size() == 0 ) { return -1; }
453  else { return *min_element(chunkStarts.begin(), chunkStarts.end()); }
454 }

484  const BamAlignment& bAlignment,

573 // load existing index data from BAM index file (".bai"), return success/fail
574 bool BamReader::BamReaderPrivate::LoadIndex(void) {
575 
576  // clear out index data
577  ClearIndex();
578 
579  // skip if index file empty
580  if ( IndexFilename.empty() ) { return false; }
581 
582  // open index file, abort on error
583  FILE* indexStream = fopen(IndexFilename.c_str(), "rb");
584  if(!indexStream) {
585  printf("ERROR: Unable to open the BAM index file %s for reading.\n", IndexFilename.c_str() );
586  return false;
587  }
588 
589  // see if index is valid BAM index
590  char magic[4];
591  fread(magic, 1, 4, indexStream);
592  if (strncmp(magic, "BAI\1", 4)) {
593  printf("Problem with index file - invalid format.\n");
594  fclose(indexStream);
595  return false;
596  }
597 
598  // get number of reference sequences
599  uint32_t numRefSeqs;
600  fread(&numRefSeqs, 4, 1, indexStream);
601 
602  // intialize space for BamIndex data structure
603  Index.reserve(numRefSeqs);
604 
605  // iterate over reference sequences
606  for (unsigned int i = 0; i < numRefSeqs; ++i) {
607 
608  // get number of bins for this reference sequence
609  int32_t numBins;
610  fread(&numBins, 4, 1, indexStream);
611 
612  if (numBins > 0) {
613  RefData& refEntry = References[i];
614  refEntry.RefHasAlignments = true;
615  }
616 
617  // intialize BinVector
618  BamBinMap binMap;
619 
620  // iterate over bins for that reference sequence
621  for (int j = 0; j < numBins; ++j) {
622 
623  // get binID
624  uint32_t binID;
625  fread(&binID, 4, 1, indexStream);
626 
627  // get number of regionChunks in this bin
628  uint32_t numChunks;
629  fread(&numChunks, 4, 1, indexStream);
630 
631  // intialize ChunkVector
632  ChunkVector regionChunks;
633  regionChunks.reserve(numChunks);
634 
635  // iterate over regionChunks in this bin
636  for (unsigned int k = 0; k < numChunks; ++k) {
637 
638  // get chunk boundaries (left, right)
639  uint64_t left;
640  uint64_t right;
641  fread(&left, 8, 1, indexStream);
642  fread(&right, 8, 1, indexStream);
643 
644  // save ChunkPair
645  regionChunks.push_back( Chunk(left, right) );
646  }
647 
648  // sort chunks for this bin
649  sort( regionChunks.begin(), regionChunks.end(), ChunkLessThan );
650 
651  // save binID, chunkVector for this bin
652  binMap.insert( pair<uint32_t, ChunkVector>(binID, regionChunks) );
653  }
654 
655  // load linear index for this reference sequence
656 
657  // get number of linear offsets
658  int32_t numLinearOffsets;
659  fread(&numLinearOffsets, 4, 1, indexStream);
660 
661  // intialize LinearOffsetVector
662  LinearOffsetVector offsets;
663  offsets.reserve(numLinearOffsets);
664 
665  // iterate over linear offsets for this reference sequeence
666  uint64_t linearOffset;
667  for (int j = 0; j < numLinearOffsets; ++j) {
668  // read a linear offset & store
669  fread(&linearOffset, 8, 1, indexStream);
670  offsets.push_back(linearOffset);
671  }
672 
673  // sort linear offsets
674  sort( offsets.begin(), offsets.end() );
675 
676  // store index data for that reference sequence
677  Index.push_back( ReferenceIndex(binMap, offsets) );
678  }
679 
680  // close index file (.bai) and return
681  fclose(indexStream);
682  return true;
683 }

685 // populates BamAlignment with alignment data under file pointer, returns success/fail
686 bool BamReader::BamReaderPrivate::LoadNextAlignment(BamAlignment& bAlignment) {
687 
688  // read in the 'block length' value, make sure it's not zero
689  char buffer[4];
690  mBGZF.Read(buffer, 4);
691  const unsigned int blockLength = BgzfData::UnpackUnsignedInt(buffer);
692  if ( blockLength == 0 ) { return false; }
693 
694  // keep track of bytes read as method progresses
695  int bytesRead = 4;
696 
697  // read in core alignment data, make sure the right size of data was read
698  char x[BAM_CORE_SIZE];
699  if ( mBGZF.Read(x, BAM_CORE_SIZE) != BAM_CORE_SIZE ) { return false; }
700  bytesRead += BAM_CORE_SIZE;
701 
702  // set BamAlignment 'core' data and character data lengths

738  char* tagData = ((char*)allCharData) + tagDataOffset;

769  bAlignment.Qualities.append( 1, singleQuality );
770  }

773  int k = 0;
774  for (unsigned int i = 0; i < numCigarOperations; ++i) {
775 
776  // build CigarOp struct
777  CigarOp op;
778  op.Length = (cigarData[i] >> BAM_CIGAR_SHIFT);
779  op.Type = CIGAR_LOOKUP[ (cigarData[i] & BAM_CIGAR_MASK) ];
780 
781  // save CigarOp
782  bAlignment.CigarData.push_back(op);
783

788  case ('I') : bAlignment.AlignedBases.append( bAlignment.QueryBases.substr(k, op.Length) ); // for 'M', 'I' - write bases
789  case ('S') : k += op.Length; // for 'S' - skip over query bases
790  break;

793  break;
794 
795  case ('P') : bAlignment.AlignedBases.append( op.Length, '*' ); // for 'P' - write padding character;
796  break;
797 
798  case ('N') : bAlignment.AlignedBases.append( op.Length, 'N' ); // for 'N' - write N's, skip bases in query sequence
799  k += op.Length;
800  break;
801 
802  case ('H') : break; // for 'H' - do nothing, move to next op
803 
804  default : printf("ERROR: Invalid Cigar op type\n"); // shouldn't get here
805  exit(1);
806  }
807  }
808 
809  // read in the tag data
810  bAlignment.TagData.resize(tagDataLen);
811  memcpy((char*)bAlignment.TagData.data(), tagData, tagDataLen);
812  }
813 
814  free(allCharData);
815  return true;

854 
855  // iterate over reference enties
856  BamIndex::iterator indexIter = Index.begin();
857  BamIndex::iterator indexEnd = Index.end();
858  for ( ; indexIter != indexEnd; ++indexIter ) {
859

867  for ( ; binIter != binEnd; ++binIter ) {
868

870  ChunkVector& binChunks = (*binIter).second;
871  if ( binChunks.size() == 0 ) { continue; }
872 
873  ChunkVector mergedChunks;
874  mergedChunks.push_back( binChunks[0] );
875 
876  // iterate over chunks
877  int i = 0;
878  ChunkVector::iterator chunkIter = binChunks.begin();
879  ChunkVector::iterator chunkEnd = binChunks.end();
880  for ( ++chunkIter; chunkIter != chunkEnd; ++chunkIter) {
881 
882  // get 'currentChunk' based on numeric index
883  Chunk& currentChunk = mergedChunks[i];
884 
885  // get iteratorChunk based on vector iterator
886  Chunk& iteratorChunk = (*chunkIter);
887 
888  // if currentChunk.Stop(shifted) == iterator Chunk.Start(shifted)
889  if ( currentChunk.Stop>>16 == iteratorChunk.Start>>16 ) {
890 
891  // set currentChunk.Stop to iteratorChunk.Stop
892  currentChunk.Stop = iteratorChunk.Stop;
893  }
894 
895  // otherwise
896  else {
897  // set currentChunk + 1 to iteratorChunk
898  mergedChunks.push_back(iteratorChunk);
899  ++i;
900  }
901  }
902 
903  // saved merged chunk vector
904  (*binIter).second = mergedChunks;
905  }
906  }
