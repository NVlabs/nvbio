35 // utility function to copy a thrust device vector to a thrust host vector
36 // the sole reason for this is to eliminate warnings from thrust when using the assignment operator
37 template<typename TTargetVector, typename TSourceVector>
38 static NVBIO_FORCEINLINE void thrust_copy_vector(TTargetVector& target, TSourceVector& source)
39 {
40  if (target.size() != source.size())
41  {
42  target.clear();
43  target.resize(source.size());
44  }
45 
46  thrust::copy(source.begin(), source.end(), target.begin());
47 }
48 
49 template<typename TTargetVector, typename TSourceVector>
50 static NVBIO_FORCEINLINE void thrust_copy_vector(TTargetVector& target, TSourceVector& source, uint32 count)
51 {
52  if (target.size() != count)
53  {
54  target.clear();
55  target.resize(count);
56  }
57 
58  thrust::copy(source.begin(), source.begin() + count, target.begin());
59 }
