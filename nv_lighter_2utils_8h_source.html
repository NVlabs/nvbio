156  template <typename KernelFunction>
157  uint32 n_blocks(KernelFunction kernel, const uint32 cta_size) const
158  {
159  if (m_blocks_hi == 0)
160  return cuda::multiprocessor_count() * cuda::max_active_blocks_per_multiprocessor( kernel, cta_size, 0u );
161  else if (m_blocks_lo == 0)
162  return cuda::multiprocessor_count();
163  else
164  return cuda::multiprocessor_count() * (m_blocks_lo + m_blocks_hi) / 2;
165  }

169  void update(const uint32 n_blocks, const float speed)
170  {
171  // carry out a little binary search over the best number of blocks/SM
172  if (m_blocks_hi == 0)
173  {
174  m_blocks_hi = n_blocks / cuda::multiprocessor_count();
175  m_speed_hi = speed;
176  }
177  else if (m_blocks_lo == 0)
178  {
179  m_blocks_lo = n_blocks / cuda::multiprocessor_count();
180  m_speed_lo = speed;
181  }
182  else if (m_speed_lo > m_speed_hi)
183  {
184  m_blocks_hi = n_blocks / cuda::multiprocessor_count();
185  m_speed_hi = speed;
186  }
187  else
188  {
189  m_blocks_lo = n_blocks / cuda::multiprocessor_count();
190  m_speed_lo = speed;
191  }
192  // TODO: once the optimizer settles to a given value, it will never change:
193  // we should explore using occasional "mutations" to adapt to possibly
194  // changing conditions...
195  }
