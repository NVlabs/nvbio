175 template <uint32 SYMBOL_SIZE, bool BIG_ENDIAN, typename word_type>
176 struct HostBWTHandler< PackedStream<word_type*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> > : public SetBWTHandler
177 {
178  typedef PackedStream<word_type*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> OutputIterator;
179 
180  static const uint32 WORD_SIZE = uint32( 8u * sizeof(word_type) );
181  static const uint32 SYMBOLS_PER_WORD = WORD_SIZE / SYMBOL_SIZE;
182 
183 
186  HostBWTHandler(OutputIterator _output) : output(_output), offset(0) {}
187 
190  template <typename bwt_iterator>
191  void do_process(
192  const uint32 n_suffixes,
193  const bwt_iterator bwt)
194  {
195  const uint32 word_offset = offset & (SYMBOLS_PER_WORD-1);
196  uint32 word_rem = 0;
197  uint32 word_idx = offset / SYMBOLS_PER_WORD;
198 
199  word_type* words = output.stream();
200 
201  if (word_offset)
202  {
203  // compute how many symbols we still need to encode to fill the current word
204  word_rem = SYMBOLS_PER_WORD - word_offset;
205 
206  // fetch the word in question
207  word_type word = words[ word_idx ];
208 
209  for (uint32 i = 0; i < word_rem; ++i)
210  {
211  const uint32 bit_idx = (word_offset + i) * SYMBOL_SIZE;
212  const uint32 symbol_offset = BIG_ENDIAN ? (WORD_SIZE - SYMBOL_SIZE - bit_idx) : bit_idx;
213  const word_type symbol = word_type(bwt[i]) << symbol_offset;
214 
215  // set bits
216  word |= symbol;
217  }
218 
219  // write out the word
220  words[ word_idx ] = word;
221 
222  // advance word_idx
223  ++word_idx;
224  }
225 
226  for (uint32 i = word_rem; i < n_suffixes; i += SYMBOLS_PER_WORD)
227  {
228  // encode a word's worth of characters
229  word_type word = 0u;
230 
231  const uint32 n_symbols = nvbio::min( SYMBOLS_PER_WORD, n_suffixes - i );
232 
233  for (uint32 j = 0; j < n_symbols; ++j)
234  {
235  const uint32 bit_idx = j * SYMBOL_SIZE;
236  const uint32 symbol_offset = BIG_ENDIAN ? (WORD_SIZE - SYMBOL_SIZE - bit_idx) : bit_idx;
237  const word_type symbol = word_type(bwt[i + j]) << symbol_offset;
238 
239  // set bits
240  word |= symbol;
241  }
242 
243  // write out the word and advance word_idx
244  words[ ++word_idx ] = word;
245  }
246 
247  // advance the offset
248  offset += n_suffixes;
249  }

293 template <typename string_type, typename output_iterator>
294 struct StringBWTHandler
295 {
296  typedef typename string_type::index_type index_type;
297 
298  static const uint32 NULL_PRIMARY = uint32(-1); // TODO: switch to index_type
299 
300  // constructor
301  //
302  StringBWTHandler(
303  const index_type _string_len,
304  const string_type _string,
305  output_iterator _output) :
306  string_len ( _string_len ),
307  string ( _string ),
308  primary ( NULL_PRIMARY ),
309  n_output ( 0 ),
310  output ( _output )
311  {
312  // encode the first BWT symbol explicitly
313  priv::device_copy( 1u, string + string_len-1, output, index_type(0u) );
314  }
315 
316  // process the next batch of suffixes
317  //
318  void process_batch(
319  const uint32 n_suffixes,
320  const uint32* d_suffixes)
321  {
322  priv::alloc_storage( d_block_bwt, n_suffixes );
323 
324  // compute the bwt of the block
325  thrust::transform(
326  thrust::device_ptr<const uint32>( d_suffixes ),
327  thrust::device_ptr<const uint32>( d_suffixes ) + n_suffixes,
328  d_block_bwt.begin(),
329  priv::string_bwt_functor<string_type>( string_len, string ) );
330 
331  // check if there is a $ sign
332  const uint32 block_primary = uint32( thrust::find(
333  d_block_bwt.begin(),
334  d_block_bwt.begin() + n_suffixes,
335  255u ) - d_block_bwt.begin() );
336 
337  if (block_primary < n_suffixes)
338  {
339  // keep track of the global primary position
340  primary = n_output + block_primary + 1u; // +1u for the implicit empty suffix
341  }
342 
343  // and copy the transformed block to the output
344  priv::device_copy(
345  n_suffixes,
346  d_block_bwt.begin(),
347  output,
348  n_output + 1u ); // +1u for the implicit empty suffix
349 
350  // advance the output counter
351  n_output += n_suffixes;
352  }
353 
354  // process a sparse set of suffixes
355  //
356  void process_scattered(
357  const uint32 n_suffixes,
358  const uint32* d_suffixes,
359  const uint32* d_slots)
360  {
361  priv::alloc_storage( d_block_bwt, n_suffixes );
362 
363  // compute the bwt of the block
364  thrust::transform(
365  thrust::device_ptr<const uint32>( d_suffixes ),
366  thrust::device_ptr<const uint32>( d_suffixes ) + n_suffixes,
367  d_block_bwt.begin(),
368  priv::string_bwt_functor<string_type>( string_len, string ) );
369 
370  // check if there is a $ sign
371  const uint32 block_primary = uint32( thrust::find(
372  d_block_bwt.begin(),
373  d_block_bwt.begin() + n_suffixes,
374  255u ) - d_block_bwt.begin() );
375 
376  if (block_primary < n_suffixes)
377  {
378  // keep track of the global primary position
379  primary = thrust::device_ptr<const uint32>( d_slots )[ block_primary ] + 1u; // +1u for the implicit empty suffix
380  }
381 
382  // and scatter the resulting symbols in the proper place
383  priv::device_scatter(
384  n_suffixes,
385  d_block_bwt.begin(),
386  thrust::make_transform_iterator(
387  thrust::device_ptr<const uint32>( d_slots ),
388  priv::offset_functor(1u) ), // +1u for the implicit empty suffix
389  output );
390  }

392  // remove the dollar symbol
393  //
394  void remove_dollar()
395  {
396  // shift back all symbols following the primary
397  const uint32 max_block_size = 32*1024*1024;
398 
399  priv::alloc_storage( d_block_bwt, max_block_size );
400 
401  for (index_type block_begin = primary; block_begin < string_len; block_begin += max_block_size)
402  {
403  const index_type block_end = nvbio::min( block_begin + max_block_size, string_len );
404 
405  // copy all symbols to a temporary buffer
406  priv::device_copy(
407  block_end - block_begin,
408  output + block_begin + 1u,
409  d_block_bwt.begin(),
410  uint32(0) );
411 
412  // and copy the shifted block to the output
413  priv::device_copy(
414  block_end - block_begin,
415  d_block_bwt.begin(),
416  output,
417  block_begin );
418  }
419  }

422  const string_type string;
423  uint32 primary;
424  uint32 n_output;

432 struct StringSSAHandler
433 {
434  // constructor
435  //
436  StringSSAHandler(
437  const uint32 _string_len,
438  const uint32 _mod,
439  output_iterator _output) :
440  string_len ( _string_len ),
441  mod ( _mod ),
442  n_output ( 1 ),
443  output ( _output )
444  {
445  // encode the implicit empty suffix directly
446  output[0] = uint32(-1);
447  }
448 
449  // process the next batch of suffixes
450  //
451  void process_batch(
452  const uint32 n_suffixes,
453  const uint32* d_suffixes)
454  {
455  priv::alloc_storage( h_suffixes, n_suffixes );
456 
457  thrust::copy(
458  thrust::device_ptr<const uint32>(d_suffixes),
459  thrust::device_ptr<const uint32>(d_suffixes) + n_suffixes,
460  h_suffixes.begin() );
461 
462  // copy_if
463  #pragma omp parallel for
464  for (int i = 0; i < int( n_suffixes ); ++i)
465  {
466  const uint32 slot = i + n_output;
467 
468  if ((slot & (mod-1)) == 0)
469  output[slot / mod] = h_suffixes[i];
470  }
471 
472  // advance the output counter
473  n_output += n_suffixes;
474  }
475 
476  // process a sparse set of suffixes
477  //
478  void process_scattered(
479  const uint32 n_suffixes,
480  const uint32* d_suffixes,
481  const uint32* d_slots)
482  {
483  priv::alloc_storage( h_slots, n_suffixes );
484  priv::alloc_storage( h_suffixes, n_suffixes );
485 
486  thrust::copy(
487  thrust::device_ptr<const uint32>(d_slots),
488  thrust::device_ptr<const uint32>(d_slots) + n_suffixes,
489  h_slots.begin() );
490 
491  thrust::copy(
492  thrust::device_ptr<const uint32>(d_suffixes),
493  thrust::device_ptr<const uint32>(d_suffixes) + n_suffixes,
494  h_suffixes.begin() );
495 
496  // scatter_if
497  #pragma omp parallel for
498  for (int i = 0; i < int( n_suffixes ); ++i)
499  {
500  const uint32 slot = h_slots[i] + 1u; // +1 for the implicit empty suffix
501 
502  if ((slot & (mod-1)) == 0)
503  output[slot / mod] = h_suffixes[i];
504  }
505  }
506 
507  const uint32 string_len;
508  const uint32 mod;
509  uint32 n_output;
510  output_iterator output;
511  thrust::host_vector<uint32> h_slots;
512  thrust::host_vector<uint32> h_suffixes;
513 };
