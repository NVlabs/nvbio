47 // push an element
48 //
49 template <typename Key, typename Container, typename Compare>
50 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE void priority_queue<Key,Container,Compare>::push(const Key key)
51 {
52  // check whether the queue is full
53  m_size++;
54  m_queue.resize( m_size+1 ); // we need one more entry for things to work out
55 
56  uint32 r = m_size;
57  while (r > 1) // sift up new item
58  {
59  const uint32 p = r/2;
60  if (! m_cmp( m_queue[p], key )) // in proper order
61  break;
62 
63  m_queue[r] = m_queue[p]; // else swap with parent
64  r = p;
65  }
66  m_queue[r] = key; // insert new item at final location
67 }

69 // pop an element
70 //
71 template <typename Key, typename Container, typename Compare>
72 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE void priority_queue<Key,Container,Compare>::pop()
73 {
74  Key dn = m_queue[m_size--]; // last item in queue
75  m_queue.resize( m_size+1 ); // we need one more entry for things to work out
76 
77  uint32 p = 1; // p points to item out of position
78  uint32 r = p << 1; // left child of p
79 
80  while (r <= m_size) // while r is still within the heap
81  {
82  // set r to smaller child of p
83  if (r < m_size && m_cmp( m_queue[r], m_queue[r+1] )) r++;
84  if (! m_cmp( dn, m_queue[r] )) // in proper order
85  break;
86 
87  m_queue[p] = m_queue[r]; // else swap with child
88  p = r; // advance pointers
89  r = p<<1;
90  }
91  m_queue[p] = m_queue[m_size+1]; // insert last item in proper place
92 }
