46 template <typename T> __device__ __forceinline__ T scan_warp(T val, const int32 tidx, volatile T *red)
47 {
48  // pad initial segment with zeros
49  red[tidx] = 0;
50  red += 32;
51 
52  // Hillis-Steele scan
53  red[tidx] = val;
54  val += red[tidx-1]; red[tidx] = val;
55  val += red[tidx-2]; red[tidx] = val;
56  val += red[tidx-4]; red[tidx] = val;
57  val += red[tidx-8]; red[tidx] = val;
58  val += red[tidx-16]; red[tidx] = val;
59  return val;
60 }

235 __device__ __forceinline__
236 uint32 alloc(uint32 n, uint32* pool, const int32 warp_tid, volatile uint32* warp_red, volatile uint32* warp_broadcast)
237 {
238  uint32 warp_scan = scan_warp( n, warp_tid, warp_red ) - n;
239  uint32 warp_count = scan_warp_total( warp_red );
240  if (warp_tid == 0)
241  *warp_broadcast = atomicAdd( pool, warp_count );
242 
243  return *warp_broadcast + warp_scan;
244 }

251 template <uint32 N>
252 __device__ __forceinline__
253 uint32 alloc(bool pred, uint32* pool, const int32 warp_tid, volatile uint32* warp_broadcast)
254 {
255  const uint32 warp_mask = __ballot( pred );
256  const uint32 warp_count = __popc( warp_mask );
257  const uint32 warp_scan = __popc( warp_mask << (warpSize - warp_tid) );
258 
259  // acquire an offset for this warp
260  if (warp_scan == 0 && pred)
261  *warp_broadcast = atomicAdd( pool, warp_count * N );
262 
263  // find offset
264  return *warp_broadcast + warp_scan * N;
265 }
