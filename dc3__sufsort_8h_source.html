145 template <typename string_type, typename output_iterator>
146 void DC3SufSort::sort(
147  const typename stream_traits<string_type>::index_type string_len,
148  const string_type string,
149  output_iterator d_suffixes)
150 {
151  typedef typename stream_traits<string_type>::index_type index_type;
152  const uint32 SYMBOL_SIZE = stream_traits<string_type>::SYMBOL_SIZE;
153 
154  typedef uint32 word_type;
155  const uint32 WORD_BITS = uint32( 8u * sizeof(word_type) );
156 
157  const uint32 SYMBOLS_PER_WORD = 3u;
158 
159  const uint32 n_suffixes = string_len;
160 
161  // reserve temporary storage
162  reserve( n_suffixes );
163 
164  // initialize the segment flags
165  thrust::fill(
166  d_segment_flags.begin(),
167  d_segment_flags.begin() + n_suffixes,
168  uint8(0u) );
169 
170 
171  // initialize the device sorting indices
172  const uint32 n_1_suffixes = uint32( thrust::copy_if(
173  thrust::make_counting_iterator<uint32>(0u),
174  thrust::make_counting_iterator<uint32>(0u) + n_suffixes,
175  d_sort_indices.begin(),
176  equal_mod3_functor(1u) ) - d_sort_indices.begin() );
177 
178  const uint32 n_12_suffixes = uint32( thrust::copy_if(
179  thrust::make_counting_iterator<uint32>(0u),
180  thrust::make_counting_iterator<uint32>(0u) + n_suffixes,
181  d_sort_indices.begin() + n_1_suffixes,
182  equal_mod3_functor(2u) ) - d_sort_indices.begin() );
183 
184  // initialize the active slots
185  thrust::copy(
186  thrust::make_counting_iterator<uint32>(0u),
187  thrust::make_counting_iterator<uint32>(0u) + n_suffixes,
188  d_active_slots.begin() );
189 
190  // first pass: build the 1mod3 & 2mod3 suffixes
191  {
192  Timer timer;
193  timer.start();
194 
195  // extract the given radix word from each of the partially sorted suffixes and merge it with the existing keys
196  priv::string_suffix_word_functor<SYMBOL_SIZE,SYMBOLS_PER_WORD,WORD_BITS,DOLLAR_BITS,string_type,word_type> word_functor( string_len, string, 0u );
197 
198  thrust::copy(
199  thrust::make_transform_iterator( d_sort_indices.begin(), word_functor ),
200  thrust::make_transform_iterator( d_sort_indices.begin(), word_functor ) + n_12_suffixes,
201  d_sort_keys.begin() );
202 
203  CUFMI_CUDA_DEBUG_STATEMENT( cudaDeviceSynchronize() );
204  timer.stop();
205  extract_time += timer.seconds();
206 
207  timer.start();
208 
209  cuda::SortBuffers<uint32*,uint32*> sort_buffers;
210  cuda::SortEnactor sort_enactor;
211 
212  sort_buffers.selector = 0;
213  sort_buffers.keys[0] = cufmi::device_view( d_sort_keys );
214  sort_buffers.keys[1] = cufmi::device_view( d_segment_keys );
215  sort_buffers.values[0] = cufmi::device_view( d_active_slots );
216  sort_buffers.values[1] = cufmi::device_view( d_segment_heads );
217 
218  // sort the keys together with the indices
219  sort_enactor.sort( n_suffixes, sort_buffers );
220 
221  if (sort_buffers.selector)
222  {
223  // swap the buffers
224  d_sort_keys.swap( d_segment_keys );
225  d_active_slots.swap( d_segment_heads );
226  }
227 
228  CUFMI_CUDA_DEBUG_STATEMENT( cudaDeviceSynchronize() );
229  timer.stop();
230  radixsort_time += timer.seconds();
231  }
232 
233  CUFMI_CUDA_DEBUG_STATEMENT( fprintf(stderr," primary key compression\n") );
234  {
235  Timer timer;
236  timer.start();
237 
238  // find out consecutive items with equal keys
239  thrust::adjacent_difference(
240  d_sort_keys.begin(),
241  d_sort_keys.begin() + n_12_suffixes,
242  d_segment_flags.begin() );
243 
244  d_segment_flags[0] = 1u;
245 
246  // and rewrite the keys using the smallest set of contiguous integers
247  thrust::inclusive_scan(
248  d_segment_flags.begin(),
249  d_segment_flags.begin() + n_12_suffixes,
250  d_segment_keys.begin() );
251 
252  CUFMI_CUDA_DEBUG_STATEMENT( cudaDeviceSynchronize() );
253  timer.stop();
254  segment_time += timer.seconds();
255 
256  timer.start();
257 
258  // now scatter the new keys to obtain the inverse map
259  thrust::scatter(
260  d_segment_keys.begin(),
261  d_segment_keys.begin() + n_12_suffixes,
262  d_active_slots.begin(),
263  d_12_keys.begin() );
264 
265  CUFMI_CUDA_DEBUG_STATEMENT( cudaDeviceSynchronize() );
266  timer.stop();
267  inverse_time += timer.seconds();
268  }
