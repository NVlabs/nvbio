23 // find all MEMs covering a given base
24 //
25 // \return the right-most position covered by a MEM
26 //
27 template <typename pattern_type, typename fm_index_type, typename delegate_type>
28 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
29 uint32 find_mems(
30  const uint32 pattern_len,
31  const pattern_type pattern,
32  const uint32 x,
33  const fm_index_type f_index,
34  const fm_index_type r_index,
35  delegate_type& handler,
36  const uint32 min_intv,
37  const uint32 min_span)
38 {
39  typedef typename fm_index_type::index_type coord_type;
40  typedef typename fm_index_type::range_type range_type;
41 
42  // find how far can we extend right starting from x
43  //
44  uint32 n_ranges = 0;
45  {
46  const fm_index_type& index = r_index;
47 
48  // extend forward, using the reverse index
49  range_type range = make_vector( coord_type(0u), index.length() );
50 
51  for (uint32 i = x; i < pattern_len; ++i)
52  {
53  const uint8 c = pattern[i];
54  if (c > 3) // there is an N here. no match
55  break;
56 
57  // search c in the FM-index
58  const range_type c_rank = rank(
59  index,
60  make_vector( range.x-1, range.y ),
61  c );
62 
63  range.x = index.L2(c) + c_rank.x + 1;
64  range.y = index.L2(c) + c_rank.y;
65 
66  // check if the range is too small
67  if (1u + range.y - range.x < min_intv)
68  break;
69 
70  // store the range
71  //ranges[ n_ranges ] = range;
72  ++n_ranges;
73  }
74  }
75 
76  // no valid match covering x
77  if (n_ranges == 0u)
78  return x;
79 
80  // keep track of the left-most coordinate covered by a MEM
81  uint32 leftmost_coordinate = x+1;
82 
83  // now extend backwards, using the forward index
84  //
85  // we basically loop through all MEMs ending in [x,x+n_ranges) starting
86  // from the end of the range and walking backwards - and for each of them we:
87  //
88  // - find their starting point through backwards search
89  //
90  // - and add them to the output only if they extend further left than
91  // any of the previously found ones
92  //
93  for (int32 r = x + int32(n_ranges) - 1 ; r >= int32(x); --r)
94  {
95  const fm_index_type& index = f_index;
96 
97  // extend from r to the left as much possible
98  range_type range = make_vector( coord_type(0u), index.length() );
99 
100  int32 l;
101 
102  for (l = r; l >= 0; --l)
103  {
104  const uint8 c = pattern[l];
105  if (c > 3) // there is an N here. no match
106  break;
107 
108  // search c in the FM-index
109  const range_type c_rank = rank(
110  index,
111  make_vector( range.x-1, range.y ),
112  c );
113 
114  const range_type new_range = make_vector(
115  index.L2(c) + c_rank.x + 1,
116  index.L2(c) + c_rank.y );
117 
118  // stop if the range became too small
119  if (1u + new_range.y - new_range.x < min_intv)
120  break;
121 
122  // update the range
123  range = new_range;
124  }
125 
126  // only output the range if it's not contained in any other MEM
127  if (uint32(l+1) < leftmost_coordinate && uint32(l+1) < r)
128  {
129  // save the range, together with its span
130  const uint2 pattern_span = make_uint2( uint32(l+1), r );
131 
132  // keep the MEM only if it is above a certain length
133  if (pattern_span.y - pattern_span.x >= min_span)
134  {
135  // pass all results to the delegate
136  handler.output( range, pattern_span );
137  }
138 
139  // update the left-most covered coordinate
140  leftmost_coordinate = uint32(l+1);
141  }
142  }
143 
144  // return the right-most end of the MEMs covering x
145  return x + n_ranges;
146 }
