27 {

29 
30  void run()
31  {
32  const uint32 tid = ++m_data->BackgroundThreads;
33  log_verbose( stderr, "starting BAM output thread %u\n", tid );
34 
35  while (m_data->BackgroundThreads != 0)
36  {
37  if (m_data->WorkCounter > 0u)
38  {
39  const int32 id = --m_data->WorkCounter;

84  while (BackgroundThreads < ThreadCount) {}

121 // compresses the current block
122 int BgzfData::DeflateBlocks(void)
123 {
124  // count the number of blocks to deflate
125  int32_t blockSize = UncompressedBlockSize;
126  while (1)
127  {
128  const int n_blocks = (BlockOffset + blockSize-1) / blockSize;
129 
130  #if 0
131  // single threaded implementation
132  for (int i = 0; i < n_blocks; ++i)
133  BlockLengths[i] = DeflateBlock( i, blockSize );
134  #else
135  CurrentBlockSize = blockSize; // set current block size
136  WorkCounter = n_blocks; // set number of blocks to process
137  ActiveThreads = n_blocks; // activate background threads
138 
139  // poll until the blocks are being processed
140  while (ActiveThreads > 0) {}
141  #endif
142 
143  // check whether compression failed for any of the blocks
144  bool ok = true;
145  for (int i = 0; i < n_blocks; ++i)
146  {
147  if (BlockLengths[i] < 0)
148  ok = false;
149  }
150 
151  if (ok)
152  return n_blocks;
153 
154  // compression failed: reduce block size
155  blockSize -= 1024;
156  if (blockSize <= 0)
157  {
158  printf("ERROR: input reduction failed.\n");
159  exit(1);
160  }
161  }
162  return 0;
163 }

165 // compresses the current block
166 int BgzfData::DeflateBlock(int32_t id, const unsigned int uncompressedBlockSize)
167 {
168  // check if the block is within the proper range
169  if (uncompressedBlockSize * id >= BlockOffset)
170  return 0;
171 
172  // initialize the gzip header
173  char* buffer = CompressedBlock + id * CompressedBlockSize;
174  unsigned int bufferSize = CompressedBlockSize;
175 
176  memset(buffer, 0, 18);
177  buffer[0] = GZIP_ID1;
178  buffer[1] = (char)GZIP_ID2;
179  buffer[2] = CM_DEFLATE;
180  buffer[3] = FLG_FEXTRA;
181  buffer[9] = (char)OS_UNKNOWN;
182  buffer[10] = BGZF_XLEN;
183  buffer[12] = BGZF_ID1;
184  buffer[13] = BGZF_ID2;
185  buffer[14] = BGZF_LEN;
186 
187  // loop to retry for blocks that do not compress enough
188  int inputLength = std::min( uncompressedBlockSize, BlockOffset - uncompressedBlockSize * id );
189  int compressedLength = 0;
190 
191  z_stream zs;
192  zs.zalloc = NULL;
193  zs.zfree = NULL;
194  zs.next_in = (Bytef*)UncompressedBlock + id * uncompressedBlockSize;
195  zs.avail_in = inputLength;
196  zs.next_out = (Bytef*)&buffer[BLOCK_HEADER_LENGTH];
197  zs.avail_out = bufferSize - BLOCK_HEADER_LENGTH - BLOCK_FOOTER_LENGTH;
198 
199  // initialize the zlib compression algorithm
200  if(deflateInit2(&zs, Z_DEFAULT_COMPRESSION, Z_DEFLATED, GZIP_WINDOW_BITS, Z_DEFAULT_MEM_LEVEL, Z_DEFAULT_STRATEGY) != Z_OK) {
201  printf("ERROR: zlib deflate initialization failed.\n");
202  exit(1);
203  }
204 
205  // compress the data
206  int status = deflate(&zs, Z_FINISH);
207  if(status != Z_STREAM_END) {
208 
209  deflateEnd(&zs);
210 
211  // reduce the input length and try again
212  if(status == Z_OK)
213  return -1;
214  }
215 
216  // finalize the compression routine
217  if(deflateEnd(&zs) != Z_OK) {
218  printf("ERROR: deflate end failed.\n");
219  exit(1);
220  }
221 
222  compressedLength = zs.total_out;
223  compressedLength += BLOCK_HEADER_LENGTH + BLOCK_FOOTER_LENGTH;
224 
225  if(compressedLength > MAX_BLOCK_SIZE) {
226  printf("ERROR: deflate overflow.\n");
227  exit(1);
228  }
229 
230  // store the compressed length
231  BgzfData::PackUnsignedShort(&buffer[16], (unsigned short)(compressedLength - 1));
232 
233  // store the CRC32 checksum
234  unsigned int crc = crc32(0, NULL, 0);
235  crc = crc32(crc, (Bytef*)UncompressedBlock + id * uncompressedBlockSize, inputLength);
236  BgzfData::PackUnsignedInt(&buffer[compressedLength - 8], crc);
237  BgzfData::PackUnsignedInt(&buffer[compressedLength - 4], inputLength);
238 
239  return compressedLength;
240 }

243 void BgzfData::FlushBlocks(void) {
244 
245  // compress the data block
246  int n_blocks = DeflateBlocks();
247 
248  // flush the data to our output stream
249  for (int i = 0; i < n_blocks; ++i)
250  {
251  int blockLength = BlockLengths[i];
252 
253  int numBytesWritten = (int)fwrite(CompressedBlock + i * CompressedBlockSize, 1, blockLength, Stream);
254 
255  if(numBytesWritten != blockLength) {
256  printf("ERROR: Expected to write %u bytes during flushing, but wrote %u bytes.\n", blockLength, numBytesWritten);
257  exit(1);
258  }
259 
260  BlockAddress += blockLength;
261  }
262  BlockOffset = 0;
263 }

265 // de-compresses the current block
266 int BgzfData::InflateBlock(const int& blockLength) {
267 
268  // Inflate the block in m_BGZF.CompressedBlock into m_BGZF.UncompressedBlock
269  z_stream zs;
270  zs.zalloc = NULL;
271  zs.zfree = NULL;
272  zs.next_in = (Bytef*)CompressedBlock + 18;
273  zs.avail_in = blockLength - 16;
274  zs.next_out = (Bytef*)UncompressedBlock;
275  zs.avail_out = UncompressedBlockSize;
276 
277  int status = inflateInit2(&zs, GZIP_WINDOW_BITS);
278  if (status != Z_OK) {
279  printf("inflateInit failed\n");
280  exit(1);
281  }
282 
283  status = inflate(&zs, Z_FINISH);
284  if (status != Z_STREAM_END) {
285  inflateEnd(&zs);
286  printf("inflate failed\n");
287  exit(1);
288  }
289 
290  status = inflateEnd(&zs);
291  if (status != Z_OK) {
292  printf("inflateEnd failed\n");
293  exit(1);
294  }
295 
296  return zs.total_out;
297 }

319 
320  if (dataLength == 0) { return 0; }
321 
322  char* output = data;
323  unsigned int numBytesRead = 0;
324  while (numBytesRead < dataLength) {
325 
326  int bytesAvailable = BlockLength - BlockOffset;
327  if (bytesAvailable <= 0) {
328  if ( ReadBlock() != 0 ) { return -1; }
329  bytesAvailable = BlockLength - BlockOffset;
330  if ( bytesAvailable <= 0 ) { break; }
331  }
332 
333  char* buffer = UncompressedBlock;
334  int copyLength = min( (int)(dataLength-numBytesRead), bytesAvailable );
335  memcpy(output, buffer + BlockOffset, copyLength);
336 
337  BlockOffset += copyLength;
338  output += copyLength;
339  numBytesRead += copyLength;
340  }
341 
342  if ( BlockOffset == BlockLength ) {
343  BlockAddress = ftell(Stream);
344  BlockOffset = 0;
345  BlockLength = 0;
346  }
347 
348  return numBytesRead;

351 int BgzfData::ReadBlock(void) {
352 
353  char header[BLOCK_HEADER_LENGTH];
354  int64_t blockAddress = ftell(Stream);
355 
356  int count = (int)fread(header, 1, sizeof(header), Stream);
357  if (count == 0) {
358  BlockLength = 0;
359  return 0;
360  }
361 
362  if (count != sizeof(header)) {
363  printf("read block failed - count != sizeof(header)\n");
364  return -1;
365  }
366 
367  if (!BgzfData::CheckBlockHeader(header)) {
368  printf("read block failed - CheckBlockHeader() returned false\n");
369  return -1;
370  }
371 
372  int blockLength = BgzfData::UnpackUnsignedShort(&header[16]) + 1;
373  char* compressedBlock = CompressedBlock;
374  memcpy(compressedBlock, header, BLOCK_HEADER_LENGTH);
375  int remaining = blockLength - BLOCK_HEADER_LENGTH;
376 
377  count = (int)fread(&compressedBlock[BLOCK_HEADER_LENGTH], 1, remaining, Stream);
378  if (count != remaining) {
379  printf("read block failed - count != remaining\n");
380  return -1;
381  }
382 
383  count = InflateBlock(blockLength);
384  if (count < 0) { return -1; }
385 
386  if ( BlockLength != 0 ) {
387  BlockOffset = 0;
388  }
389 
390  BlockAddress = blockAddress;
391  BlockLength = count;
392  return 0;
393 }

416 unsigned int BgzfData::Write(const char* data, const unsigned int dataLen) {
417 
418  // initialize
419  unsigned int numBytesWritten = 0;
420  const char* input = data;
421  unsigned int blockLength = UncompressedBlockSize * ThreadCount;
422 
423  // copy the data to the buffer
424  while(numBytesWritten < dataLen) {
425  unsigned int copyLength = min(blockLength - BlockOffset, dataLen - numBytesWritten);
426  char* buffer = UncompressedBlock;
427  memcpy(buffer + BlockOffset, input, copyLength);
428 
429  BlockOffset += copyLength;
430  input += copyLength;

436  }
