126 inline uint8 ReadDataFile_TXT::get(void)
127 {
128  if (m_buffer_pos >= m_buffer_size)
129  {
130  // grab more data from the underlying file
131  m_file_state = fillBuffer();
132  m_buffer_pos = 0;
133 
134  // if we failed to read more data, return \0
135  if (m_file_state != FILE_OK)
136  return 0;
137  }
138 
139  return m_buffer[m_buffer_pos++];
140 }
