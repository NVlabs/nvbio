416 template <typename string_set_type>
417 int32 compare_suffixes(
418  const string_set_type string_set,
419  const uint2 suffix1,
420  const uint2 suffix2)
421 {
422  typedef typename string_set_type::string_type string_type;
423  typedef Suffix<string_type,uint32> suffix_type;
424 
425  // check whether both inputs represent the same suffix
426  if (suffix1.y == suffix2.y &&
427  suffix1.x == suffix2.x)
428  return 0;
429 
430  const suffix_type string1 = make_suffix( string_set[suffix1.y], suffix1.x );
431  const suffix_type string2 = make_suffix( string_set[suffix2.y], suffix2.x );
432 
433  const uint32 len1 = nvbio::length( string1 );
434  const uint32 len2 = nvbio::length( string2 );
435 
436  const uint32 min_len = nvbio::min( len1, len2 );
437 
438  // compare character by character
439  int32 cmp = 0;
440  for (uint32 j = 0; j < min_len; ++j)
441  {
442  const uint8 c_i = string1[j];
443  const uint8 c_n = string2[j];
444  if (c_i < c_n)
445  {
446  cmp = -1;
447  break;
448  }
449  if (c_i > c_n)
450  {
451  cmp = 1;
452  break;
453  }
454  }
455 
456  // break ties...
457  if (cmp == 0)
458  {
459  if (len1 < len2) // $ is smaller than any other character => s1 < s2
460  cmp = -1;
461  else if (len2 < len1) // $ is smaller than any other character => s2 < s1
462  cmp = 1;
463  else // $_1 < $_2 ? -1 : 1
464  cmp = suffix1.y < suffix2.y ? -1 : 1;
465  }
466  return cmp;
467 }
