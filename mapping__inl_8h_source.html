51 // Checks if the seed's Ns are compatible with the search, i.e.
52 // if there are 0 Ns in the exactly-matched region, and at most 1 N
53 // in the inexact matching region.
54 template <typename ReadStream>
55 NVBIO_DEVICE NVBIO_FORCEINLINE
56 bool check_N(const ReadStream& seed, const uint32 exact_len, const uint32 seed_len)
57 {
58  for (uint32 i = 0; i < exact_len; ++i)
59  if (seed[i] >= 4)
60  return false;
61 
62  uint32 N_count = 0;
63  for (uint32 i = exact_len; i < seed_len; ++i)
64  {
65  if (seed[i] >= 4 && ++N_count > 1)
66  return false;
67  }
68  return true;
69 }

71 // This function is a lot like match_reverse,
72 // except that it accepts lower and upper bounds on the sequence stream
73 template< typename FMType, typename StreamType > NVBIO_DEVICE NVBIO_FORCEINLINE
74 uint2 match_range(uint2 range, const FMType index, StreamType query,
75  uint32 begin, uint32 end)
76 {
77  for(uint32 i=begin; i<end && range.x<=range.y; ++i) {
78  uint8 c = query[i];
79  if(c>3) return make_uint2(1, 0);
80 
81  uint2 bwt_cnts = rank(index, make_uint2(range.x-1, range.y), c);
82  range.x = index.L2(c) + bwt_cnts.x + 1;
83  range.y = index.L2(c) + bwt_cnts.y;
84  }
85  return range;
86 }

114 template< bool find_exact,
115  typename Stream,
116  typename FMType,
117  typename HitType> NVBIO_DEVICE NVBIO_FORCEINLINE
118 void map(const Stream query,
119  const uint32 len1,
120  const uint32 len2,
121  const FMType index,
122  const SeedHit::Flags hit_flags,
123  HitType& hitheap, uint32 max_hits,
124  uint32& range_sum,
125  uint32& range_count)
126 {
127  // TODO:
128  // if we know there is one (and only one) N in the second part of the seed,
129  // we could avoid doing the mismatch search at all other positions.
130  // However, due to the control divergence it would generate, it might be
131  // better to filter these seeds away and leave them for a second kernel
132  // call.
133  //
134 
135  // Exact Match first half of read
136  uint2 base_range = make_uint2(0, index.length());
137  base_range = match_range(base_range, index, query, 0, len1);
138 
139  // Allow an error in the next batch
140  for(uint32 i = len1; i < len2 && base_range.x <= base_range.y; ++i)
141  {
142  const uint8 c = query[i];
143 
144  // Get ranges for all chars at pos i
145  uint4 occ_lo, occ_hi;
146  rank4(index, make_uint2(base_range.x-1, base_range.y), &occ_lo, &occ_hi);
147 
148  // Try each character
149  for(uint8 sub=0; sub<4; ++sub) {
150  if( sub != c && comp(occ_hi, sub) > comp(occ_lo, sub) ) {
151  // Manually insert sub
152  uint2 range = make_uint2(index.L2(sub) + comp(occ_lo, sub) + 1,
153  index.L2(sub) + comp(occ_hi, sub) );
154  // Extend match to end of read
155  range = match_range(range, index, query, i+1, len2);
156 
157  // Store Result
158  if (range.x <= range.y) {
159  if (hitheap.size() == max_hits)
160  hitheap.pop_bottom();
161  hitheap.push(SeedHit(hit_flags, inclusive_to_exclusive(range)));
162 
163  range_sum += range.y - range.x + 1u;
164  range_count++;
165  }
166  }
167  }
168  // Extend base_range
169  if( c < 4 ) {
170  base_range.x = index.L2(c) + comp(occ_lo, c) + 1;
171  base_range.y = index.L2(c) + comp(occ_hi, c);
172  } else { base_range = make_uint2(1, 0); break; }
173  }
174 
175  if (find_exact && base_range.x<=base_range.y)
176  {
177  if (hitheap.size() == max_hits)
178  hitheap.pop_bottom();
179  hitheap.push(SeedHit(hit_flags, inclusive_to_exclusive(base_range)));
180 
181  range_sum += base_range.y - base_range.x + 1u;
182  range_count++;
183  }
184 }

205  uint32& range_sum,

216

218  const uint32 SYMBOLS_PER_WORD = BatchType::SEQUENCE_SYMBOLS_PER_WORD;
219  const uint32 seed_offs = pos & (SYMBOLS_PER_WORD-1); // pos % 8, there are 8 bases per uint32

228  typedef index_transform_iterator< Reader, ReverseXform<typename Reader::index_type> > rSeedReader;
229 
230  SeedHit::Flags flags;
231  uint2 range;
232

243  hitheap.pop_bottom();

245 
246  range_sum += range.y - range.x + 1u;
247  range_count++;
248  }
249 
250  #if USE_REVERSE_INDEX

257  {
258  flags = SeedHit::build_flags(COMPLEMENT, REVERSE, pos-read_range.x+seed_len-1);
259  if (hitheap.size() == params.max_hits)
260  hitheap.pop_bottom();
261  hitheap.push(SeedHit(flags, inclusive_to_exclusive(range)));
262  }
263  #else
264  // Complement seed=1, reverse scan, forward index=0
265  const rSeedReader r_reader(reader, reverse_offset);
266  const transform_iterator<rSeedReader, complement_functor<4> > cr_reader(r_reader, complement_functor<4>());
267 
268  range = make_uint2(0, fmi.length());
269  range = match_range(range, fmi, cr_reader, 0, seed_len);
270  if (range.x <= range.y)
271  {
272  flags = SeedHit::build_flags(COMPLEMENT, FORWARD, pos-read_range.x);
273  if (hitheap.size() == params.max_hits)
274  hitheap.pop_bottom();

485 template<MappingAlgorithm ALGO, uint32 MAX_SEED, typename BatchType, typename FMType, typename rFMType> __global__
486 void map_kernel(
487  const BatchType read_batch, const FMType fmi, const rFMType rfmi,
488  const uint32 retry,
489  const nvbio::cuda::PingPongQueuesView<uint32> queues,
490  SeedHitDequeArrayDeviceView hits,
491  const ParamsPOD params)
492 {
493  __shared__ volatile uint32 sm_broadcast[BLOCKDIM >> 5];
494  volatile uint32& warp_broadcast = sm_broadcast[ warp_id() ];
495 
496  // Pad shared by 1 uint32 because seed may not start at beginning of uint.
497  // For MAX_SEED=32, SHARED_DIM=5, so gcd(SHARED_DIM, SMEM_BANKS)=1, and
498  // we don't have to worry about bank conflicts.
499  enum { SHARED_DIM = MAX_SEED/8+1 };
500  __shared__ uint32 S[BLOCKDIM][SHARED_DIM];
501 
502  const uint32 thread_id = threadIdx.x + BLOCKDIM*blockIdx.x;
503  if (thread_id >= queues.in_size) return;
504  const uint32 read_id = queues.in_queue[ thread_id ];
505  NVBIO_CUDA_ASSERT( read_id < read_batch.size() );
506  const uint2 read_range = read_batch.get_range( read_id );
507  const uint32 read_len = read_range.y - read_range.x;
508 
509  // filter away short strings
510  if (read_len < params.min_read_len)
511  {
512  hits.resize_deque( read_id, 0 );
513  return;
514  }
515 
516  const uint32 seed_len = nvbio::min( params.seed_len, read_len );
517  const uint32 seed_freq = (uint32)params.seed_freq( read_len );
518  const uint32 retry_stride = seed_freq/(params.max_reseed+1);
519 
520  SeedHit local_hits[512];
521  typedef SeedHitDequeArrayDeviceView::hit_vector_type hit_storage_type;
522  typedef SeedHitDequeArrayDeviceView::hit_deque_type hit_deque_type;
523  hit_deque_type hitheap( hit_storage_type( 0, local_hits ), true );
524 
525  uint32 range_sum = 0;
526  uint32 range_count = 0;
527 
528  // loop over seeds
529  for (uint32 pos = read_range.x + retry * retry_stride; pos + seed_len <= read_range.y; pos += seed_freq)
530  {
531  seed_mapper<ALGO>::enact(
532  read_batch, fmi, rfmi,
533  read_range,
534  pos,
535  seed_len,
536  &S[threadIdx.x][0],
537  hitheap,
538  range_sum,
539  range_count,
540  params );
541  }
542 
543  // save the hits
544  store_deque( hits, read_id, hitheap.size(), local_hits );
545 
546  // enqueue for re-seeding if there are too many hits
547  if (queues.out_queue)
548  {
549  if (range_count == 0 || range_sum >= params.rep_seeds * range_count)
550  {
551  const uint32 slot = alloc( queues.out_size, &warp_broadcast );
552  NVBIO_CUDA_ASSERT( slot < queues.in_size );
553  queues.out_queue[ slot ] = read_id;
554  }
555  }
556 }

579  const uint32 thread_id = threadIdx.x + BLOCKDIM*blockIdx.x;
580  if (thread_id >= read_batch.size()) return;
581  const uint32 read_id = thread_id;
582  const uint2 read_range = read_batch.get_range( read_id );
583  const uint32 read_len = read_range.y - read_range.x;
584 
585  // filter away short strings
586  if (read_len < params.min_read_len)
587  {
588  hits.resize_deque( read_id, 0 );
589  return;
590  }
591 
592  const uint32 seed_len = nvbio::min( params.seed_len, read_len );
593  const uint32 seed_freq = (uint32)params.seed_freq( read_len );
594  const uint32 retry_stride = seed_freq/(params.max_reseed+1);
595 
596  SeedHit local_hits[512];
597  typedef SeedHitDequeArrayDeviceView::hit_vector_type hit_storage_type;
598  typedef SeedHitDequeArrayDeviceView::hit_deque_type hit_deque_type;
599 
600  for (uint32 retry = 0; retry <= params.max_reseed; ++retry)
601  {
602  // restore the result heap
603  hit_deque_type hitheap( hit_storage_type( 0, local_hits ), true );
604 
605  uint32 range_sum = 0;
606  uint32 range_count = 0;
607 
608  for (uint32 i = seed_range.x; i < seed_range.y; ++i)
609  {
610  const uint32 pos = read_range.x + retry * retry_stride + i * seed_freq;
611  if (pos + seed_len > read_range.y)
612  break;
613 
614  seed_mapper<ALGO>::enact(
615  read_batch, fmi, rfmi,
616  read_range,
617  pos,
618  seed_len,
619  &S[threadIdx.x][0],
620  hitheap,
621  range_sum,
622  range_count,
623  params );
624  }
625 
626  // check whether we could stop here or if we need re-seeding
627  if (retry == params.max_reseed || range_count == 0 && range_sum < params.rep_seeds * range_count)
628  {
629  // save the hits
630  store_deque( hits, read_id, hitheap.size(), local_hits );
631  break;
632  }
633  }
634 }

836  const BatchType& read_batch, const FMType fmi, const rFMType rfmi,

838  const nvbio::cuda::PingPongQueuesView<uint32> queues,
