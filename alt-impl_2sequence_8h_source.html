303  // reserve enough storage for a given number of reads and bps
304  //
305  void reserve(const uint32 n_seqs, const uint32 n_bps)
306  {
307  // a default read id length used to reserve enough space upfront and avoid frequent allocations
308  const uint32 AVG_NAME_LENGTH = 250;

314  m_qual_vec.reserve( n_bps );

317  }
