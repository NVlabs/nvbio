185  {
186  coords.x = range.x;
187  coords.y = range.y;
188  coords.z = string_id;
189  coords.w = span_begin | (span_end << 16);

191  if (flag)
192  set_group_flag();

196  void set_group_flag() { coords.z |= GROUP_FLAG; }
