203 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
204 uint32 bits_per_symbol(const Alphabet alphabet)
205 {
206  return alphabet == DNA ? 2 :
207  alphabet == DNA_N ? 4 :
208  alphabet == DNA_IUPAC ? 4 :
209  alphabet == PROTEIN ? 8 :
210  alphabet == RNA ? 2 :
211  alphabet == RNA_N ? 4 :
212  8u;
213 }
