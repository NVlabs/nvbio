73 template <typename ScoringScheme, typename PipelineType, typename ReduceContext> __global__
74 void score_reduce_kernel(
75  const ReduceContext context,
76  PipelineType pipeline,
77  const ParamsPOD params)
78 {
79  typedef ScoringQueuesDeviceView scoring_queues_type;
80  typedef ReadHitsReference<scoring_queues_type> read_hits_type;
81  typedef typename read_hits_type::reference hit_type;
82 
83  scoring_queues_type& scoring_queues = pipeline.scoring_queues;
84 
85  const uint32 thread_id = threadIdx.x + BLOCKDIM*blockIdx.x;
86  if (thread_id >= scoring_queues.active_read_count()) return;
87 
88  // fetch the active read corresponding to this thread
89  read_hits_type read_hits( scoring_queues, thread_id );
90 
91  const uint32 read_id = read_hits.read_info().read_id;
92 
93  // fetch the best alignments
94  io::BestAlignments best(
95  pipeline.best_alignments[ read_id ],
96  pipeline.best_alignments[ read_id + pipeline.best_stride ] );
97 
98  // setup the read stream
99  const uint2 read_range = pipeline.reads.get_range(read_id);
100  const uint32 read_len = read_range.y - read_range.x;
101 
102  const uint32 count = read_hits.size();
103 
104  for (uint32 i = 0; i < count; ++i)
105  {
106  hit_type hit = read_hits[i];
107  NVBIO_CUDA_DEBUG_ASSERT( hit.read_id == read_id, "reduce hit[%u][%u]: expected id %u, got: %u (slot: %u)\n", thread_id, i, read_id, hit.read_id, scoring_queues.hits_index( thread_id, i ));
108 
109  const packed_seed seed_info = hit.seed;
110  const uint32 read_rc = seed_info.rc;
111  const uint32 top_flag = seed_info.top_flag;
112 
113  const int32 score = hit.score;
114  const uint32 g_pos = hit.loc;
115 
116  // skip locations that we have already visited without paying the extension attempt
117  if ((read_rc == best.m_a1.m_rc && g_pos == best.m_a1.m_align) ||
118  (read_rc == best.m_a2.m_rc && g_pos == best.m_a2.m_align))
119  continue;
120 
121  if (score > best.m_a1.score())
122  {
123  context.best_score( read_id, params );
124 
125  // set the first best score
126  best.m_a2 = best.m_a1;
127  best.m_a1 = io::Alignment( g_pos, 0u, score, read_rc );
128  NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "update best: (parent[%u:%u])\n score[%d], rc[%u], pos[%u]\n", thread_id, i, score, read_rc, g_pos );
129  }
130  else if ((score > best.m_a2.score()) && io::distinct_alignments( best.m_a1.m_align, best.m_a1.m_rc, g_pos, read_rc, read_len/2 ))
131  {
132  context.second_score( read_id, params );
133 
134  // set the second best score
135  best.m_a2 = io::Alignment( g_pos, 0u, score, read_rc );
136  NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "update second: (parent[%u:%u])\n score[%d], rc[%u], pos[%u]\n", thread_id, i, score, read_rc, g_pos );
137  }
138  else if (context.failure( i, read_id, top_flag, params ))
139  {
140  // stop traversal
141  pipeline.hits.erase( read_id );
142 
143  // NOTE: we keep using the score entries rather than breaking here,
144  // as we've already gone through the effort of computing them, and
145  // this reduction is only a minor amount of work.
146  }
147  }
148 
149  // report best alignments
150  pipeline.best_alignments[ read_id ] = best.m_a1;
151  pipeline.best_alignments[ read_id + pipeline.best_stride ] = best.m_a2;
152 }

168 template <typename ScoringScheme, typename PipelineType, typename ReduceContext> __global__
169 void score_reduce_paired_kernel(
170  const ReduceContext context,
171  PipelineType pipeline,
172  const ParamsPOD params)
173 {
174  typedef ScoringQueuesDeviceView scoring_queues_type;
175  typedef ReadHitsReference<scoring_queues_type> read_hits_type;
176  typedef typename read_hits_type::reference hit_type;
177 
178  scoring_queues_type& scoring_queues = pipeline.scoring_queues;
179 
180  const uint32 thread_id = threadIdx.x + BLOCKDIM*blockIdx.x;
181  if (thread_id >= scoring_queues.active_read_count()) return;
182 
183  const uint32 anchor = pipeline.anchor;
184 
185  // fetch the active read corresponding to this thread
186  read_hits_type read_hits( scoring_queues, thread_id );
187 
188  const uint32 read_id = read_hits.read_info().read_id;
189 
190  // fetch current best alignments
191  io::BestPairedAlignments best_pairs = io::BestPairedAlignments(
192  io::BestAlignments( pipeline.best_alignments[ read_id ],
193  pipeline.best_alignments[ read_id + pipeline.best_stride ] ),
194  io::BestAlignments( pipeline.best_alignments_o[ read_id ],
195  pipeline.best_alignments_o[ read_id + pipeline.best_stride ] ) );
196 
197  // setup the read stream
198  const uint2 read_range = pipeline.reads.get_range(read_id);
199  const uint32 read_len = read_range.y - read_range.x;
200 
201  const uint32 count = read_hits.size();
202 
203  // sort the entries in the score queue so that we process them in the order they were selected
204  for (uint32 i = 0; i < count; ++i)
205  {
206  hit_type hit = read_hits[i];
207  NVBIO_CUDA_DEBUG_ASSERT( hit.read_id == read_id, "reduce hit[%u][%u]: expected id %u, got: %u (slot: %u)\n", thread_id, i, read_id, hit.read_id, read_hits.slot(i) );
208 
209  const packed_seed seed_info = hit.seed;
210  const uint32 read_rc = seed_info.rc;
211  const uint32 top_flag = seed_info.top_flag;
212 
213  const uint2 g_pos = make_uint2( hit.loc, hit.sink );
214  const int32 score1 = hit.score;
215  const int32 score2 = hit.opposite_score;
216  const int32 score = score1 + score2;
217 
218  // compute opposite mate placement & orientation
219  bool o_left;
220  bool o_fw;
221 
222  detail::frame_opposite_mate(
223  params.pe_policy,
224  anchor,
225  !read_rc,
226  o_left,
227  o_fw );
228 
229  const uint2 o_g_pos = make_uint2( hit.opposite_loc, hit.opposite_sink );
230  const uint32 o_read_rc = !o_fw;
231 
232  const io::PairedAlignments pair(
233  io::Alignment( g_pos.x, g_pos.y - g_pos.x, score1, read_rc, anchor, true ),
234  io::Alignment( o_g_pos.x, o_g_pos.y - o_g_pos.x, score2, o_read_rc, !anchor, true ) );
235 
236  // skip locations that we have already visited without paying the extension attempt
237  if (distinct_alignments( best_pairs.pair<0>(), pair ) == false ||
238  distinct_alignments( best_pairs.pair<1>(), pair ) == false)
239  continue;
240 
241  // check if this is a paired alignment
242  if (score2 > pipeline.score_limit)
243  {
244  // check if the best alignment was unpaired or had a worse score
245  if (best_pairs.is_paired() == false || score > best_pairs.best_score())
246  {
247  context.best_score( read_id, params );
248 
249  // set the first best score
250  best_pairs.m_a2 = best_pairs.m_a1;
251  best_pairs.m_o2 = best_pairs.m_o1;
252  best_pairs.m_a1 = pair.m_a;
253  best_pairs.m_o1 = pair.m_o;
254  NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "update best (anchor[%u]): (parent[%u:%u])\n 1. score[%d], rc[%u], pos[%u]\n 2. score[%d], rc[%u], pos[%u,%u]\n", anchor, thread_id, i, score1, read_rc, g_pos.y, score2, o_read_rc, o_g_pos.x, o_g_pos.y );
255  } // check if the second-best alignment was unpaired or had a worse score
256  else if (best_pairs.has_second_paired() == false || (score > best_pairs.second_score() && distinct_alignments( best_pairs.pair<0>(), pair, read_len/2 )))
257  {
258  context.second_score( read_id, params );
259 
260  // set the second best score
261  best_pairs.m_a2 = pair.m_a;
262  best_pairs.m_o2 = pair.m_o;
263  NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "update second (anchor[%u]): (parent[%u:%u])\n 1. score[%d], rc[%u], pos[%u]\n 2. score[%d], rc[%u], pos[%u,%u]\n", anchor, thread_id, i, score1, read_rc, g_pos.y, score2, o_read_rc, o_g_pos.x, o_g_pos.y );
264  }
265  else if (context.failure( i, read_id, top_flag, params ))
266  {
267  //NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "discard: (parent[%u:%u])\n score[%d + %d], rc[%u], pos[%u]\n", anchor, thread_id, i, score1, score2, read_rc, g_pos );
268  // stop traversal
269  pipeline.hits.erase( read_id );
270  }
271  } // the alignment is unpaired, check if the best alignment so far was also unpaired
272  else if ((params.pe_unpaired == true) && (best_pairs.is_paired() == false))
273  {
274  //
275  // We didn't find a paired alignment yet - hence we proceed keeping track of the best unpaired alignments
276  // for the first and second mate separately.
277  // We store best two alignments for the first mate in a_best_data, and the ones of the second mate in o_best_data.
278  //
279 
280  io::BestAlignments best = best_pairs.mate( anchor );
281 
282  //
283  // update the first mate alignments
284  //
285  if (score1 > best.m_a1.score())
286  {
287  best.m_a2 = best.m_a1;
288  best.m_a1 = io::Alignment( g_pos.x, g_pos.y - g_pos.x, score1, read_rc, anchor, false );
289  if (anchor == 0)
290  {
291  best_pairs.m_a1 = best.m_a1;
292  best_pairs.m_a2 = best.m_a2;
293  }
294  else
295  {
296  best_pairs.m_o1 = best.m_a1;
297  best_pairs.m_o2 = best.m_a2;
298  }
299 
300  NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "update best unpaired[%u]: (parent[%u:%u])\n 1. score[%d], rc[%u], pos[%u]\n", anchor, thread_id, i, score1, read_rc, g_pos );
301  }
302  else if ((score1 > best.m_a2.score()) && io::distinct_alignments( best.m_a1.alignment(), best.m_a1.is_rc(), g_pos.x, read_rc, read_len/2 ))
303  {
304  best.m_a2 = io::Alignment( g_pos.x, g_pos.y - g_pos.x, score1, read_rc, anchor, false );
305  if (anchor == 0) best_pairs.m_a2 = best.m_a2;
306  else best_pairs.m_o2 = best.m_a2;
307 
308  NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "update second unpaired[%u]: (parent[%u:%u])\n score[%d], rc[%u], pos[%u]\n", anchor, thread_id, i, score1, read_rc, g_pos );
309  }
310  else if (context.failure( i, read_id, top_flag, params ))
311  {
312  // stop traversal
313  pipeline.hits.erase( read_id );
314  }
315  }
316  else if (context.failure( i, read_id, top_flag, params ))
317  {
318  //NVBIO_CUDA_DEBUG_PRINT_IF( params.debug.show_reduce( read_id ), "discard: (parent[%u:%u])\n score[%d + %d], rc[%u], pos[%u]\n", anchor, thread_id, i, score1, score2, read_rc, g_pos );
319  // stop traversal
320  pipeline.hits.erase( read_id );
321  }
322  }
323 
324  // report best alignments
325  pipeline.best_alignments[ read_id ] = best_pairs.m_a1;
326  pipeline.best_alignments[ read_id + pipeline.best_stride ] = best_pairs.m_a2;
327  pipeline.best_alignments_o[ read_id ] = best_pairs.m_o1;
328  pipeline.best_alignments_o[ read_id + pipeline.best_stride ] = best_pairs.m_o2;
329 }

337 //
338 // Reduce the scores associated to each read in the input queue to find
339 // the best 2 alignments.
340 //
341 template <typename ScoringScheme, typename ReduceContext>
342 void score_reduce_t(
343  const ReduceContext context,
344  const BestApproxScoringPipelineState<ScoringScheme>& pipeline,
345  const ParamsPOD params)
346 {
347  typedef BestApproxScoringPipelineState<ScoringScheme> pipeline_type;
348 
349  const int blocks = (pipeline.scoring_queues.active_reads.in_size + BLOCKDIM-1) / BLOCKDIM;
350 
351  detail::score_reduce_kernel<typename pipeline_type::scheme_type> <<<blocks, BLOCKDIM>>>(
352  context,
353  pipeline,
354  params );
355 }
