38 // copy scoring data to host, converting to io::AlignmentResult
39 void GPUOutputBatch::readback_scores(thrust::host_vector<io::AlignmentResult>& output,
40  const AlignmentMate mate,
41  const AlignmentScore score) const
42 {
43  // copy alignment data into a staging buffer
44  thrust::host_vector<io::BestAlignments> best_data_staging;
45  nvbio::cuda::thrust_copy_vector(best_data_staging, best_data_dvec, count);
46 
47  // convert the contents of the staging buffer into io::AlignmentResult
48  output.resize(count);
49  for(uint32 c = 0; c < count; c++)
50  {
51  io::BestAlignments& old_best_aln = best_data_staging[c];
52  io::Alignment& old_aln = (score == BEST_SCORE ? old_best_aln.m_a1 : old_best_aln.m_a2);
53  io::AlignmentResult& new_aln = output[c];
54 
55  if (score == BEST_SCORE)
56  {
57  new_aln.best[mate] = old_aln;
58  } else {
59  new_aln.second_best[mate] = old_aln;
60  }
61 
62  if (mate == MATE_2)
63  {
64  new_aln.is_paired_end = true;
65  }
66  }
67 }
