55 // extract alignment data for a given mate
56 // note that the mates can be different for the cigar, since mate 1 is always the anchor mate for cigars
57 AlignmentData get_mate(HostOutputBatchPE& batch, const uint32 aln_id, const AlignmentMate mate)
58 {
59  const uint32 read_id = batch.read_ids.size() ?
60  batch.read_ids[ aln_id ] : aln_id;
61 
62  if (batch.alignments[0][aln_id].mate() == mate)
63  {
64  return AlignmentData(&batch.alignments[0][aln_id],
65  batch.mapq[0][aln_id],
66  aln_id,
67  read_id,
68  batch.read_data[ mate ],
69  &batch.cigar[0],
70  &batch.mds[0]);
71  }
72  else

74  return AlignmentData(&batch.alignments[1][aln_id],
75  batch.mapq[1][aln_id],
76  aln_id,
77  read_id,

81  }
82 }
