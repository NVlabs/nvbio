72 // generate table for counting 11,10,01,00(pattern) for 8 bits number
73 // table [no# ] = representation ( # of count-pattern, . , . , . )
74 // ---------------------------------------------------------------------------
75 // e.g cnt_table[11111111] = 0x04000000 ( 4-11, 0-10, 0-01, 0-00 )
76 // cnt_table[00100001] = 0x00010102 ( 0-11, 1-10, 1-01, 2-00 )
77 // cnt_table[00000001] = 0x00000103 ( 0-11, 0-10, 1-01, 3-00 )
78 inline void gen_bwt_count_table(uint32* count_table)
79 {
80  for (int i = 0; i != 256; ++i)
81  {
82  uint32 x = 0;
83  for (int j = 0; j != 4; ++j)
84  x |= (((i&3) == j) + ((i>>2&3) == j) + ((i>>4&3) == j) + (i>>6 == j)) << (j<<3);
85 
86  count_table[i] = x;
87  }
88 }
