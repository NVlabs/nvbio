547  // reserve enough storage for a given number of reads and bps
548  //
549  void reserve(const uint32 n_seqs, const uint32 n_bps)
550  {
551  // a default read id length used to reserve enough space upfront and avoid frequent allocations
552  const uint32 AVG_NAME_LENGTH = 250;
553 
554  const uint32 bps_per_word = 32u / bits_per_symbol( SequenceDataInfo::m_alphabet );
555 
556  m_sequence_index_vec.reserve( n_seqs+1 );
557  m_sequence_vec.reserve( n_bps / bps_per_word );
558  m_qual_vec.reserve( n_bps );
559  m_name_index_vec.reserve( AVG_NAME_LENGTH * n_seqs );
560  m_name_index_vec.reserve( n_seqs+1 );
561  }
