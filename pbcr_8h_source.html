36 template <uint32 BLOCKDIM, uint32 OCC_INTV, uint32 SYMBOL_SIZE, bool BIG_ENDIAN>
37 __global__
38 void rank_reduction_kernel(
39  const uint32 size,
40  const PackedStream<uint32*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> bwt,
41  const uint32 stride,
42  uint32* red)
43 {
44  const uint32 SYMBOLS_PER_WORD = uint32( 8u * sizeof(uint32) );
45  const uint32 SYMBOL_COUNT = 1u << SYMBOL_SIZE;
46 
47  // compute a global thread id
48  const uint32 global_tid = threadIdx.x + blockIdx.x * BLOCKDIM;
49 
50  // count the occurrences of each symbol in each word
51  uint32 occ[SYMBOL_COUNT];
52 
53  // compute how many words worth of symbols we need to read to get OCC_INTV symbols per warp
54  const uint32 WORDS_PER_THREAD = OCC_INTV / (cuda::Arch::WARP_SIZE * SYMBOLS_PER_WORD);
55 
56  // local storage for the words we need to load
57  uint32 words[ WORDS_PER_THREAD ];
58 
59  // load all the words we need
60  cub::LoadDirectBlockedVectorized( threadIdx.x, bwt.stream() + blockIdx.x * BLOCKDIM, words );
61 
62  // loop through all the words we have loaded and count the occurrences of each symbol
63  for (uint32 wid = 0; wid < WORDS_PER_THREAD; ++wid)
64  {
65  const uint32 global_wid = global_tid * WORDS_PER_THREAD + wid;
66 
67  // build a packed stream out of the current word
68  const PackedStream<uint32*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> bwt_w( &words[ wid ] );
69 
70  // loop through all the symbols in this word
71  for (uint32 i = 0; i < SYMBOLS_PER_WORD; ++i)
72  {
73  // unpack the i-th symbol
74  const uint32 c = bwt_w[i];
75 
76  // and add it to the corresponding counter
77  if (global_wid * SYMBOLS_PER_WORD + i < size)
78  {
79  #pragma unroll
80  for (uint32 s = 0; s < SYMBOL_COUNT; ++s)
81  {
82  if (c == s)
83  ++occ[c];
84  }
85  }
86  }
87  }
88 
89  // specialize WarpReduce for type uint32
90  typedef cub::WarpReduce<uint32> WarpReduce;
91 
92  // declare its shared temp storage
93  __shared__ typename cub::WarpReduce::TempStorage temp_storage[ BLOCKDIM / cuda::Arch::WARP_SIZE ];
94 
95  // compute the output index
96  const uint32 output_id = warp_id() + blockIdx.x * BLOCKDIM / cuda::Arch::WARP_SIZE;
97 
98  // for each symbol, reduce the occurrence counters across the warp
99  #pragma unroll
100  for (uint32 s = 0; s < SYMBOL_COUNT; ++s)
101  {
102  const uint32 sum = WarpReduce( temp_storage[ warp_id() ] ).Sum( occ[s] );
103 
104  if (warp_tid() == 0)
105  red[ output_id + s * stride ] = sum;
106  }
107 }

111 template <uint32 OCC_INTV, uint32 SYMBOL_SIZE, bool BIG_ENDIAN>
112 void build_rank_dictionary(
113  const uint32 size,
114  const PackedStream<uint32*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> bwt,
115  nvbio::vector<device_tag,uint32>& rank_dict,
116  uint64* rank_sums)
117 {
118  const uint32 SYMBOL_COUNT = 1u << SYMBOL_SIZE;
119 
120  const uint32 N_entries = util::divide_ri( size, OCC_INTV ) + 1u;
121 
122  // make sure we have enough storage
123  priv::alloc_storage( rank_dict, N_entries * SYMBOL_COUNT );
124 
125  nvbio::vector<device_tag,uint32> red( N_entries * SYMBOL_COUNT );
126  nvbio::vector<device_tag,uint32> temp_storage;
127 
128  // compute the occurrence counts every OCC_INTV entries
129  {
130  const uint32 blockdim = 128;
131  const uint32 n_blocks = (n + blockdim-1) / blockdim;
132  rank_reduction_kernel<BLOCKDIM,OCC_INTV> <<<n_blocks,blockdim>>>( size, bwt, N_entries, red );
133  }
134 
135  for (uint32 c = 0; c < SYMBOL_COUNT; ++c)
136  {
137  // set the last entry to zero
138  red[ c * N_entries + N_entries - 1 ] = 0u;
139 
140  exclusive_scan(
141  N_entries,
142  red.begin() + c * N_entries,
143  make_strided_iterator( rank_dict.begin(), SYMBOL_COUNT ),
144  thrust::plus<uint32>(),
145  0u,
146  temp_storage );
147 
148  // save the total sum
149  rank_sums[c] = rank_dict[ N_entries - 1 ];
150  }
151 }

153 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
154 uint32 rank_2bits(const uint32 intv, const uint32* rank_dict, const uint32* bwt, const uint8 c, const uint32 i)
155 {
156  const uint32 LOG_SYMS_PER_WORD = 4u;
157  const uint32 SYMS_PER_WORD = 16u;
158 
159  const uint32 k = uint32( i / intv ); // i / intv
160  const uint32 m = (i - k*intv) >> LOG_SYMS_PER_WORD; // (i % intv) / SYMS_PER_WORD
161  const word_type i_mod = ~i & (SYMS_PER_WORD-1u); // ~i % SYMS_PER_WORD
162 
163  // fetch base occurrence counter
164  const uint32 base = rank_dict[ k*4 + c ];
165 
166  const uint32 off = k*(intv >> LOG_SYMS_PER_WORD);
167 
168  // sum up all the pop-counts of the relevant masks
169  return base + occ::popc_2bit( bwt, c, off, off + m, i_mod );
170 }

280 template <uint32 SYMBOL_SIZE, bool BIG_ENDIAN, typename storage_type, typename output_handler>
281 void pbcr(
282  const StridedStringSet<
283  PackedStream<storage_type,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64>,
284  uint64*> string_set,
285  output_handler& output,
286  BWTParams* params = NULL)
287 {
288  typedef PackedStream<storage_type,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> input_packed_stream_type;
289  typedef PackedStream<nvbio::vector<device_tag,uint32>::const_iterator,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> row_packed_stream_type;
290  typedef PackedStream<uint32*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> bwt_packed_stream_type;
291 
292  typedef StridedStringSet<input_packed_stream_type,uint64*> input_string_set_type;
293 
294 
295  const uint32 SYMBOLS_PER_WORD = (8u*sizeof(uint32)) / SYMBOL_SIZE;
296  const uint32 SYMBOL_COUNT = 1u << SYMBOL_SIZE;
297 
298  // fetch the number of strings
299  const uint32 N = string_set.size();
300 
301  // compute the number of words per input row
302  const uint32 N_words = util::divide_ri( N, SYMBOLS_PER_WORD );
303 
304  //
305  // sort the strings by the first character
306  //
307 
308  nvbio::vector<device_tag,uint32> d_row_storage( N_words );
309  nvbio::vector<device_tag,uint8> d_row( N*2 );
310  nvbio::vector<device_tag,uint32> d_index( N*2 );
311  nvbio::vector<device_tag,uint64> d_frequencies( SYMBOL_COUNT + 1u );
312  nvbio::vector<device_tag,uint64> d_rank( N );
313  nvbio::vector<device_tag,uint32> d_B_storage( N_words );
314  nvbio::vector<device_tag,uint32> d_rank_dictionary;
315 
316  const input_packed_stream_type input_packed_stream = string_set.base_string();
317 
318  const uint32 row_stride = string_set.stride();
319 
320  // find the longest string in the set
321  uint32 max_length = 0u;
322  for (uint32 i = 0; i < N; ++i)
323  max_length = nvbio::max( max_length, nvbio::length( string_set[i] ) );
324 
325  // copy the first row on the device
326  thrust::copy(
327  input_packed_stream.stream(),
328  input_packed_stream.stream() + N_words,
329  d_row_storage.begin() );
330 
331  const row_packed_stream_type d_packed_row( d_row_storage.begin() );
332 
333  // and unpack it
334  thrust::copy( d_packed_row, d_packed_row + N, d_row.begin() );
335 
336  // initialize the string indices
337  thrust::copy( thrust::make_counting_iterator<uint32>(0), thrust::make_counting_iterator<uint32>(N), d_index.begin() );
338 
339  cuda::SortBuffers<uint8*,uint32*> sort_buffers;
340  sort_buffers.keys[0] = nvbio::raw_pointer( d_row );
341  sort_buffers.keys[1] = nvbio::raw_pointer( d_row ) + N;
342  sort_buffers.values[0] = nvbio::raw_pointer( d_index );
343  sort_buffers.values[1] = nvbio::raw_pointer( d_index ) + N;
344 
345  cuda::SortEnactor sort_enactor;
346  sort_enactor.sort( N, sort_buffers );
347 
348  //
349  // count the occurrences of each character in the alphabet - this is equivalent
350  // to the number of 1-suffixes starting with each character
351  //
352 
353  // find the end of each bin of values
354  thrust::upper_bound(
355  d_row.begin() + N * sort_buffers.selector,
356  d_row.begin() + N * sort_buffers.selector + N,
357  thrust::make_counting_iterator<uint8>(0u),
358  thrust::make_counting_iterator<uint8>(0u) + SYMBOL_COUNT + 1u,
359  d_frequencies.begin() );
360 
361  // compute the histogram by taking differences of the cumulative histogram
362  thrust::adjacent_difference( d_frequencies.begin(), d_frequencies.end(), d_frequencies.begin() );
363 
364  // initialize the rank array, giving, for each suffix, the number of suffixes preceding it in sorted order
365  thrust::scatter(
366  thrust::make_counting_iterator<uint32>(0u),
367  thrust::make_counting_iterator<uint32>(0u) + N,
368  d_index.begin() + N * sort_buffers.selector,
369  d_rank.begin() );
370 
371  // allocate the ping-pong buffers for storing the BWT
372  nvbio::vector<host_tag,uint32> B_storage( N * max_length * 2u );
373 
374  uint32* B_storage0 = nvbio::raw_pointer( B_storage );
375  uint32* B_storage1 = nvbio::raw_pointer( B_storage ) + N * max_length;
376 
377  bwt_packed_stream_type B( B_storage0 );
378  bwt_packed_stream_type B_out( B_storage1 );
379 
380  // keep track of the size of each bucket
381  uint64 B_offsets[SYMBOL_COUNT+1];
382  uint64 B_size[SYMBOL_COUNT+1];
383  {
384  B_offsets[0] = 0u;
385 
386  nvbio::vector<device_tag,uint32> d_segments( SYMBOL_COUNT );
387  thrust::upper_bound(
388  d_row.begin() + N * sort_buffers.selector,
389  d_row.begin() + N * sort_buffers.selector + N,
390  thrust::make_counting_iterator<uint8>(0u),
391  thrust::make_counting_iterator<uint8>(0u) + SYMBOL_COUNT,
392  d_segments.begin() );
393 
394  thrust::copy( d_segments.begin(), d_segments.end(), &B_offsets[1] );
395  }
396  for (uint32 c = 0; c < SYMBOL_COUNT; ++c)
397  B_size[c] = B_offsets[c+1] - B_offsets[c];
398 
399  // build the BWT block by block, aligning each block to the next word boundary
400  {
401  uint64 begin = 0u;
402  uint64 shift = 0u;
403 
404  // copy the unpermuted BWT symbols in a vector
405  nvbio::vector<device_tag,uint8> h_row( N );
406  nvbio::transform<host_tag>(
407  N,
408  thrust::make_counting_iterator<uint32>(0u),
409  h_row.begin(),
410  priv::string_set_slice_functor<input_string_set_type>( string_set, 1u ) );
411 
412  nvbio::vector<device_tag,uint32> h_index( N );
413  thrust::copy(
414  d_index.begin() + N * sort_buffers.selector,
415  d_index.begin() + N * sort_buffers.selector + N,
416  h_index.begin() );
417 
418  // and copy them block by block
419  for (uint32 c = 0; c < SYMBOL_COUNT; ++c)
420  {
421  const uint64 end = B_offsets[c+1];
422 
423  // align the beginning of the output block to the next word boundary
424  B_offsets[c] = util::round_i( begin + shift, SYMBOLS_PER_WORD );
425 
426  // compute the cumulative shift we have applied so far
427  shift = B_offsets[c] - begin;
428 
429  // and assign the permuted symbols to the current block
430  nvbio::assign(
431  end - begin,
432  thrust::make_permutation_iterator( h_row.begin(), h_index.begin() ),
433  B + B_offsets[c] );
434 
435  // keep track of the beginning of the next block
436  begin = B_offsets[c+1];
437  }
438  }

440  //
441  // TODO:
442  // keep track separately of the dollars in the BWT, so as to avoid wrongly counting
443  // them as 0's while ranking...
444  // two options here:
445  // 1. use a separate bit-vector (very large, O(M) in the number of symbols - O(1) access time)
446  // 2. use a an array of indices (very compact, O(N) in the number of strings - O(log(N)) access time)
447  //
448 
449  const uint32 OCC_INTV = 512u;
450 
451  // loop through the suffixes by increasing length
452  for (uint32 j = 2; j <= max_length; ++j)
453  {
454  //
455  // At this point, we can think of the partial BWT B(j-1) of the suffixes of length up to j-1
456  // as being divided in buckets B(j-1)(0),...,B(j-1)(M), where B(j-1)(c) contains all
457  // suffixes of length up to (j-1) starting with the letter c.
458  // For each j-suffix S[L-j:L], we now have to compute its rank in the corresponding bucket
459  // B(j-1)(S[L-j]), so as to insert it in the correct position to obtain the new bucket
460  // B(j)(S[L-j]).
461  //
462  // Let z = S[L-j] and v = S[L-j+1].
463  // The rank of S[L-j:L] is then given by the sum of:
464  // - the rank of z in { B(j-1)(0), ..., B(j-1)(v-1) } and
465  // - the rank of z in B(j-1)(v)[0,r-1],
466  //
467  // where r is equal to the rank of the (j-1)-suffix S[L-j+1:L] among those (j-1)-suffixes
468  // that start with v.
469  // This is because { B(j-1)(0), ..., B(j-1)(v-1) } contain all (j-1)-suffixes starting with a
470  // letter less than v, and among those we are counting the ones whose preceding symbol is z
471  // (remember that the BWT stores the symbol preceeding each suffix).
472  // B(j-1)(v)[0,r-1] instead contain all those (j-1)-suffixes which start precisely with v,
473  // that are smaller than S[L-j+1:L] - and again among them we count only those whose preceding
474  // symbol is z.
475  //
476  // We now have to consider how to compute all those terms rank( z, B(j-1)(v)[0,r-1] ) in parallel,
477  // for all suffixes.
478  // An idea is that at each pass we can build an occurrence table for each bucket B(j-1).
479  // This involves doing a prefix-sum across all of B(j-1) once per symbol c, each time counting
480  // how many symbols are effectively equal to c.
481  // As storing the entire occurrence table would require too much memory, we will only store
482  // a sparsely sampled version, with one entry every X.
483  //
484  // To give an idea, with X = 1024 and storing 8-bytes per counter for 4 symbols, the occurrence table
485  // for 100B symbols would require 3.2 GB.
486  //
487  // The problem at this point is that we might not be able to store all of B(j-1) in device memory at once.
488  // However, if again we bucket the new j-suffixes by their second letter v, at any given point in time
489  // we will only need the bucket B(j-1)(v).
490  // In the case above, assuming uniform distribution of the symbols, B(j-1)(v) will roughly occupy ~6.25GB.
491  // To further improve memory consumption, we could further sort the j-suffixes by r - thus allowing to
492  // break each B(j-1)(v) in smaller sub-buckets.
493  //
494 
495  //
496  // TODO: filter away suffixes that are too long wrt their string.
497  // Note that we'll need to filter all the following arrays:
498  // - d_index
499  // - d_ranks
500  // - d_row
501  //
502 
503  // segment the current row by symbol, so as to bucket the j-suffixes by their second character v
504  // (remember the current row contains the previous character)
505  uint32 segments[SYMBOL_COUNT+1u];
506  {
507  segments[0] = 0u;
508 
509  nvbio::vector<device_tag,uint32> d_segments( SYMBOL_COUNT );
510  thrust::upper_bound(
511  d_row.begin() + N * sort_buffers.selector,
512  d_row.begin() + N * sort_buffers.selector + N,
513  thrust::make_counting_iterator<uint8>(0u),
514  thrust::make_counting_iterator<uint8>(0u) + SYMBOL_COUNT,
515  d_segments.begin() );
516 
517  thrust::copy( d_segments.begin(), d_segments.end(), &segments[1] );
518  }
519 
520  // copy the j-th row on the device (i.e. the first character of the j-suffixes)
521  thrust::copy(
522  input_packed_stream.stream() + row_stride * (j-1),
523  input_packed_stream.stream() + row_stride * (j-1) + N_words,
524  d_row_storage.begin() );
525 
526  // and gather it in the current (j-1)-suffix order
527  thrust::gather(
528  d_index.begin() + N * sort_buffers.selector,
529  d_index.begin() + N * sort_buffers.selector + N,
530  d_packed_row,
531  d_row.begin() + N * sort_buffers.selector );
532 
533  //
534  // process the new suffixes in batches organized by their second character
535  //
536 
537  // keep track of the character frequencies up to the current bucket
538  uint64 rank_offsets[SYMBOL_COUNT+1][SYMBOL_COUNT];
539  for (uint32 c1 = 0; c1 <= SYMBOL_COUNT; ++c1)
540  for (uint32 c2 = 0; c2 < SYMBOL_COUNT; ++c2)
541  rank_offsets[c1][c2] = 0u;
542 
543  // keep track of the global offset for the current BWT block
544  uint32 block_offset = 0u;
545 
546  // process bucket by bucket
547  for (uint32 v = 0; v < SYMBOL_COUNT; ++v)
548  {
549  // copy B(j-1)(v) to the device
550  {
551  // find the range of words overlapped by B(j-1)(v) on the host
552  const uint64 begin_word = util::divide_rz( B_offsets[v], SYMBOLS_PER_WORD );
553  const uint64 end_word = util::divide_ri( B_offsets[v+1], SYMBOLS_PER_WORD );
554 
555  // make sure we have enough storage
556  alloc_storage( d_B_storage, end_word - begin_word );
557 
558  thrust::copy( B.stream() + begin_word, B.stream() + end_word, d_B_storage.begin() );
559  }
560  const bwt_packed_stream_type d_B( nvbio::raw_pointer( d_B_storage ) );
561 
562  // build the rank dictionary for B(j-1)(v)
563  priv::build_rank_dictionary<OCC_INTV>( B_size[v], d_B, d_rank_dictionary, &rank_offsets[v+1][0] );
564  for (uint8 c = 0; c < SYMBOL_COUNT; ++c)
565  rank_offsets[v+1][c] += rank_offsets[v][c];
566 
567  // build a rank dictionary functor
568  priv::rank_functor<SYMBOL_SIZE> rank_functor(
569  B_size[v],
570  OCC_INTV,
571  rank_offsets[v],
572  nvbio::raw_pointer( d_rank_dictionary ),
573  nvbio::raw_pointer( d_B_storage ) );
574 
575  // rank the new j-suffixes in the segment (segments[v],segments[v+1]),
576  // remembering d_row now contains their new character z
577  nvbio::for_each<device_tag>(
578  segments[v+1] - segments[v]
579  thrust::make_counting_iterator<uint32>( segments[v] ),
