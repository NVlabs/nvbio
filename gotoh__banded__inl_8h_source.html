46 template <uint32 BAND_LEN, AlignmentType TYPE, typename H_band_type, typename F_band_type, typename scoring_type, typename score_type>
47 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
48 void init_row_zero(
49  H_band_type& H_band,
50  F_band_type& F_band,
51  const scoring_type& scoring,
52  const score_type infimum)
53 {
54  H_band[0] = 0;
55  #pragma unroll
56  for (uint32 j = 1; j < BAND_LEN; ++j)
57  H_band[j] = TYPE == GLOBAL ? scoring.text_gap_open() + (j-1)*scoring.text_gap_extension() : 0;
58 
59  #if defined(SW_F_INITIALIZED_TO_INF)
60  // F[0,*] = -inf
61  #pragma unroll
62  for (uint32 j = 0; j < BAND_LEN; ++j)
63  F_band[j] = infimum;
64  #else
65  // F[0,*] = 0
66  #pragma unroll
67  for (uint32 j = 0; j < BAND_LEN; ++j)
68  F_band[j] = 0;
69  #endif
70 
71  // The following is the initialization code for the "post" F-loop, i.e. for the case in which
72  // F is updated _after_ updating H
73  // F[0,*] = -inf, so F[1,*] is the gap open cost, except F[1,BAND_LEN-1] remains -inf
74  //#pragma unroll
75  //for (uint32 j = 0; j < BAND_LEN-1; ++j)
76  // F_band[j] = scoring.pattern_gap_open();
77 }

350 struct gotoh_alignment_score_dispatch

406  template <
407  typename pattern_type,
408  typename qual_type,
409  typename text_type,
410  typename scoring_type,
411  typename context_type,
412  typename sink_type>
413  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
414  static
415  bool run(
416  const scoring_type& scoring,
417  pattern_type pattern,
418  qual_type quals,
419  text_type text,
420  const uint32 window_begin,
421  const uint32 window_end,
422  const uint32 pos,
423  const int32 min_score,
424  context_type& context,
425  sink_type& sink)
426  {
427  const uint32 pattern_len = pattern.length();
428  const uint32 text_len = text.length();
429  const uint32 start = pos;
430 
431  if (text_len < pattern_len)
432  return false;
433 
434  typedef int32 score_type;
435 
436  typedef typename Reference_cache<BAND_LEN>::type text_cache_type;
437  uint32 text_cache_storage[Reference_cache<BAND_LEN>::BAND_WORDS];
438  text_cache_type text_cache( text_cache_storage );
439 
440  // load first band of text
441  for (uint32 j = 0; j < BAND_LEN-1; ++j)
442  text_cache[j] = text[start+window_begin+j];
443 
444  const score_type G_o = scoring.pattern_gap_open();
445  const score_type G_e = scoring.pattern_gap_extension();
446  const score_type infimum = Field_traits<short>::min() -
447  nvbio::max( nvbio::max( G_o, G_e ),
448  nvbio::max( scoring.text_gap_open(), scoring.text_gap_extension() ) );
449 
450  // initialize the first band (corresponding to the 0-th row of the DP matrix)
451  score_type H_band[BAND_LEN];
452  score_type F_band[BAND_LEN];
453 
454  // initialize bands
455  context.init(
456  window_begin,
457  H_band,
458  F_band,
459  scoring,
460  infimum );
461 
462  // loop across the short edge of the DP matrix: each band is a segment of the long columns
463  for (uint32 i = window_begin; i < window_end; ++i)
464  {
465  // pass the previous row to the context
466  context.previous_row( i, H_band, F_band );
467 
468  // load the new pattern character
469  const uint8 q = pattern[i];
470  const uint8 qq = quals[i];
471 
472  const score_type V = scoring.match(qq);
473 
474  //
475  // arrows for backtracking - in the E,F case, SUBSTITUTION means the score came from H,
476  // INSERTION means from E or F respectively (to fit in 1 bit we use INSERTION = 1 for both)
477  // note for hdir we have separate values for INSERTION (from F), DELETION (from E)
478  //
479  DirectionVector hdir = SUBSTITUTION;
480  DirectionVector edir = SUBSTITUTION;
481 
482  // j == 0 case
483  {
484  // update F
485  const score_type ftop = F_band[1] + G_e;
486  const score_type htop = H_band[1] + G_o;
487  F_band[0] = nvbio::max( ftop, htop );
488  const DirectionVector fdir = ftop > htop ? DELETION_EXT : SUBSTITUTION;
489 
490  const uint8 g = text_cache[0];
491  const score_type S_ij = (g == q) ? V : scoring.mismatch( g, q, qq );
492  const score_type diagonal = H_band[0] + S_ij;
493  const score_type top = F_band[0];
494  score_type hi = nvbio::max( top, diagonal );
495 
496  if (TYPE == LOCAL)
497  {
498  hi = nvbio::max( hi, score_type(0) ); // clamp to zero
499  hdir = ( hi == score_type(0) ) ? SINK : hdir; // a score of 0 would terminate backtracking
500  sink.report( hi, make_uint2( i+1, i+1 ) );
501  }
502  H_band[0] = hi;
503 
504  // update the maximum score
505  //max_score = hi;
506 
507  // pass the new cell to the context (we save the fdir vector at the end of the previous iteration)
508  context.new_cell(
509  i, 0u,
510  hi,
511  (hdir == SINK) ? SINK : (top > diagonal ? INSERTION : SUBSTITUTION),
512  SUBSTITUTION,
513  fdir );
514  }
515  // compute E_1
516  score_type E_j = H_band[0] + G_o;
517 
518  #pragma unroll
519  for (uint32 j = 1; j < BAND_LEN-1; ++j)
520  {
521  // update F
522  const score_type ftop = F_band[j+1] + G_e;
523  const score_type htop = H_band[j+1] + G_o;
524  F_band[j] = nvbio::max( ftop, htop );
525  const DirectionVector fdir = ftop > htop ? DELETION_EXT : SUBSTITUTION;
526  /*DirectionVector fdir;
527  if (j < BAND_LEN-2)
528  {
529  const score_type ftop = F_band[j+1] + G_e;
530  const score_type htop = H_band[j+1] + G_o;
531  F_band[j] = nvbio::max( ftop, htop );
532  fdir = ftop > htop ? DELETION_EXT : SUBSTITUTION;
533  }
534  else
535  {
536  // except for the special case i = 0, F[BAND_LEN-1] = -inf, so F[BAND_LEN-2] = H[BAND_LEN-1] + G_o
537  F_band[ BAND_LEN-2 ] = H_band[BAND_LEN-1] + G_o;
538  fdir = SUBSTITUTION;
539  }*/
540 
541  const uint32 g = text_cache[j]; text_cache[j-1] = g;
542  const score_type S_ij = (g == q) ? V : scoring.mismatch( g, q, qq );
543  const score_type diagonal = H_band[j] + S_ij;
544  const score_type top = F_band[j];
545  const score_type left = E_j;
546  score_type hi = nvbio::max3( top, left, diagonal );
547  hdir = SUBSTITUTION;
548  if (TYPE == LOCAL)
549  {
550  hi = nvbio::max( hi, score_type(0) ); // clamp to zero
551  hdir = ( hi == score_type(0) ) ? SINK : hdir; //at a score of 0 we would terminate backtracking in LOCAL alignment
552  sink.report( hi, make_uint2( i + j + 1, i+1 ) );
553  }
554  H_band[j] = hi;
555 
556  // update the maximum score
557  //max_score = nvbio::max( max_score, hi );
558 
559  // pass the new cell to the context, again fdir is saved in the previous iteration
560  context.new_cell(
561  i, j,
562  hi,
563  (hdir == SINK) ? SINK :
564  (top > left ?
565  (top > diagonal ? INSERTION : SUBSTITUTION) :
566  (left > diagonal ? DELETION : SUBSTITUTION)),
567  edir,
568  fdir );
569 
570  // update E for the next round, i.e. j+1
571  const score_type eleft = E_j + G_e;
572  const score_type ediagonal = hi + G_o;
573  edir = (eleft > ediagonal) ? INSERTION_EXT : SUBSTITUTION;
574  E_j = nvbio::max( ediagonal, eleft );
575  }
576 
577  // load the new text character and fill last entry of the band cache
578  const uint8 g = start+i+BAND_LEN-1 < text_len ? text[start+i+BAND_LEN-1] : 255u;
579  text_cache[ BAND_LEN-2 ] = g;
580

583  // F[BAND_LEN] && H[BAND_LEN] = -inf so the max here is always -inf
584  F_band[ BAND_LEN-1 ] = infimum;
585  const DirectionVector fdir = SUBSTITUTION;
586

589  const score_type diagonal = H_band[BAND_LEN-1] + S_ij;
590  const score_type left = E_j;
591  score_type hi = nvbio::max( left, diagonal );
592  hdir = SUBSTITUTION;
593  if (TYPE == LOCAL)
594  {
595  hi = nvbio::max( hi, score_type(0) ); // clamp to zero
596  hdir = ( hi == score_type(0) ) ? SINK : hdir; // a score of 0 terminates backtracking in LOCAL alignment
597  sink.report( hi, make_uint2( i + BAND_LEN, i+1 ) );
598  }
599  H_band[ BAND_LEN-1 ] = hi;
600 
601  // update the maximum score
602  //max_score = nvbio::max( max_score, hi );
603 
604  // pass the new cell to the context
605  context.new_cell(
606  i, BAND_LEN-1,
607  hi,
608  (hdir == SINK) ? SINK : (left > diagonal ? DELETION : SUBSTITUTION),
609  edir,
610  fdir );
611  }
612 
613  // check whether min_score is within reach
614  //const score_type threshold_score = score_type( min_score ) + score_type(pattern_len - i - 1)*scoring.match(0);
615  //if (max_score < threshold_score)
616  // return false;
617  }
618 
619  if (window_end < pattern_len)
620  {
621  // compute the maximum score we got
622  score_type max_score = H_band[0];
623  #pragma unroll
624  for (uint32 j = 1; j < BAND_LEN; ++j)
625  max_score = nvbio::max( max_score, H_band[j] );
626 
627  // and check whether min_score is within reach
628  const score_type threshold_score = score_type( min_score ) + score_type(pattern_len - window_end)*scoring.match(0);
629  if (max_score < threshold_score)
630  return false;
631  }
632

634  context.last_row( window_end, H_band, F_band );
635

637  {
638  if (TYPE == GLOBAL)
639  sink.report( H_band[ BAND_LEN-1 ], make_uint2( pattern_len + BAND_LEN-1, pattern_len ) );

643

653  }

883 template <
884  uint32 BAND_LEN,
885  uint32 CHECKPOINTS,
886  AlignmentType TYPE,
887  typename scoring_type,
888  typename checkpoint_type,
889  typename submatrix_type,
890  typename backtracer_type>
891 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
892 bool banded_alignment_traceback(
893  const GotohAligner<TYPE,scoring_type>& aligner,
894  checkpoint_type checkpoints,
895  const uint32 checkpoint_id,
896  submatrix_type submatrix,
897  const uint32 submatrix_height,
898  uint8& state,
899  uint2& sink,
900  backtracer_type& backtracer)
901 {
902  // Backtrack from the second checkpoint to the first looking up the flow submatrices H,E,F
903  // and keeping track of which we are in via state
904 
905  int32 current_entry = sink.x - sink.y;
906  int32 current_row = sink.y - checkpoint_id*CHECKPOINTS - 1u;
907 
908  NVBIO_CUDA_DEBUG_ASSERT( current_entry >= 0 &&
909  current_entry < BAND_LEN, "sw::banded_alignment_backtrack(): sink (%u,%u) -> local x coordinate %d not in [0,%d[\n", sink.x, sink.y, current_entry, BAND_LEN );
910  NVBIO_CUDA_DEBUG_ASSERT( current_row >= 0, "sw::banded_alignment_backtrack(): sink (%u,%u) -> local y coordinate %d not in [0,%u[\n", sink.x, sink.y, current_row, submatrix_height );
911  NVBIO_CUDA_DEBUG_ASSERT( current_row < submatrix_height, "sw::banded_alignment_backtrack(): sink (%u,%u) -> local y coordinate %d not in [0,%u[\n", sink.x, sink.y, current_row, submatrix_height );
912 
913  while (current_row >= 0)
914  {
915  const uint32 submatrix_cell = current_row * BAND_LEN + current_entry;
916  const uint8 op = submatrix[ submatrix_cell ];
917  const uint8 h_op = op & HMASK;
918 
919  if (TYPE == LOCAL)
920  {
921  if (state == HSTATE && h_op == SINK)
922  {
923  sink.y = current_row + checkpoint_id*CHECKPOINTS + 1u;
924  sink.x = current_entry + sink.y;
925  return true;
926  }
927  }
928 
929  if (state == ESTATE)
930  {
931  if ((op & INSERTION_EXT) == 0u) state = HSTATE;
932  --current_entry;
933  backtracer.push( DELETION );
934  }
935  else if (state == FSTATE)
936  {
937  if ((op & DELETION_EXT) == 0u) state = HSTATE;
938  ++current_entry; --current_row;
939  backtracer.push( INSERTION );
940  }
941  else
942  {
943  if (h_op == DELETION)
944  state = ESTATE;
945  else if (h_op == INSERTION)
946  state = FSTATE;
947  else
948  {
949  --current_row;
950  backtracer.push( SUBSTITUTION );
951  }
952 
953  }
954  NVBIO_CUDA_ASSERT( current_entry >= 0 && current_entry < BAND_LEN );
955  }
956  sink.y = checkpoint_id*CHECKPOINTS;
957  sink.x = current_entry + sink.y;
958  return false;
959 }
