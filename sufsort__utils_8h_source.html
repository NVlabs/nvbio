111 template <uint32 SYMBOL_SIZE, bool BIG_ENDIAN, typename word_type>
112 struct HostBWTHandler< PackedStream<word_type*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> > : public BaseBWTHandler
113 {
114  typedef PackedStream<word_type*,uint8,SYMBOL_SIZE,BIG_ENDIAN,uint64> OutputIterator;
115 
116  static const uint32 WORD_SIZE = uint32( 8u * sizeof(word_type) );
117  static const uint32 SYMBOLS_PER_WORD = WORD_SIZE / SYMBOL_SIZE;
118 
119 
122  HostBWTHandler(OutputIterator _output) : output(_output), offset(0) {}
123 
126  void process(
127  const uint32 n_suffixes,
128  const uint8* h_bwt,
129  const uint8* d_bwt,
130  const uint2* h_suffixes,
131  const uint2* d_suffixes,
132  const uint32* d_indices)
133  {
134  const uint32 word_offset = offset & (SYMBOLS_PER_WORD-1);
135  uint32 word_rem = 0;
136  uint32 word_idx = offset / SYMBOLS_PER_WORD;
137 
138  word_type* words = output.stream();
139 
140  if (word_offset)
141  {
142  // compute how many symbols we still need to encode to fill the current word
143  word_rem = SYMBOLS_PER_WORD - word_offset;
144 
145  // fetch the word in question
146  word_type word = words[ word_idx ];
147 
148  for (uint32 i = 0; i < word_rem; ++i)
149  {
150  const uint32 bit_idx = (word_offset + i) * SYMBOL_SIZE;
151  const uint32 symbol_offset = BIG_ENDIAN ? (WORD_SIZE - SYMBOL_SIZE - bit_idx) : bit_idx;
152  const word_type symbol = word_type(h_bwt[i]) << symbol_offset;
153 
154  // set bits
155  word |= symbol;
156  }
157 
158  // write out the word
159  words[ word_idx ] = word;
160 
161  // advance word_idx
162  ++word_idx;
163  }
164 
165  for (uint32 i = word_rem; i < n_suffixes; i += SYMBOLS_PER_WORD)
166  {
167  // encode a word's worth of characters
168  word_type word = 0u;
169 
170  const uint32 n_symbols = nvbio::min( SYMBOLS_PER_WORD, n_suffixes - i );
171 
172  for (uint32 j = 0; j < n_symbols; ++j)
173  {
174  const uint32 bit_idx = j * SYMBOL_SIZE;
175  const uint32 symbol_offset = BIG_ENDIAN ? (WORD_SIZE - SYMBOL_SIZE - bit_idx) : bit_idx;
176  const word_type symbol = word_type(h_bwt[i + j]) << symbol_offset;
177 
178  // set bits
179  word |= symbol;
180  }
181 
182  // write out the word and advance word_idx
183  words[ ++word_idx ] = word;
184  }
185 
186  // advance the offset
187  offset += n_suffixes;
188  }
189 
190  OutputIterator output;
191  uint64 offset;
192 };

218  // constructor
219  //
220  StringBWTHandler(
221  const index_type _string_len,
222  const string_type _string,
223  output_iterator _output) :
224  string_len ( _string_len ),
225  string ( _string ),
226  primary ( NULL_PRIMARY ),
227  n_output ( 0 ),
228  output ( _output )
229  {
230  // encode the first BWT symbol explicitly
231  priv::device_copy( 1u, string + string_len-1, output, index_type(0u) );
232  }

234  // process the next batch of suffixes
235  //
236  void process_batch(
237  const uint32 n_suffixes,
238  const uint32* d_suffixes)
239  {
240  priv::alloc_storage( d_block_bwt, n_suffixes );
241 
242  // compute the bwt of the block
243  thrust::transform(
244  thrust::device_ptr<const uint32>( d_suffixes ),
245  thrust::device_ptr<const uint32>( d_suffixes ) + n_suffixes,
246  d_block_bwt.begin(),
247  priv::string_bwt_functor<string_type>( string_len, string ) );
248 
249  // check if there is a $ sign
250  const uint32 block_primary = uint32( thrust::find(
251  d_block_bwt.begin(),
252  d_block_bwt.begin() + n_suffixes,
253  255u ) - d_block_bwt.begin() );
254 
255  if (block_primary < n_suffixes)
256  {
257  // keep track of the global primary position
258  primary = n_output + block_primary + 1u; // +1u for the implicit empty suffix
259  }
260 
261  // and copy the transformed block to the output
262  priv::device_copy(
263  n_suffixes,
264  d_block_bwt.begin(),
265  output,
266  n_output + 1u ); // +1u for the implicit empty suffix
267 
268  // advance the output counter
269  n_output += n_suffixes;
270  }

273  //
274  void process_scattered(
275  const uint32 n_suffixes,
276  const uint32* d_suffixes,
277  const uint32* d_slots)
278  {
279  priv::alloc_storage( d_block_bwt, n_suffixes );
280 
281  // compute the bwt of the block
282  thrust::transform(

284  thrust::device_ptr<const uint32>( d_suffixes ) + n_suffixes,
285  d_block_bwt.begin(),

291  d_block_bwt.begin() + n_suffixes,

293 
294  if (block_primary < n_suffixes)

310  // remove the dollar symbol
311  //
312  void remove_dollar()
313  {
314  // shift back all symbols following the primary
315  const uint32 max_block_size = 32*1024*1024;
316 
317  priv::alloc_storage( d_block_bwt, max_block_size );
318 
319  for (index_type block_begin = primary; block_begin < string_len; block_begin += max_block_size)
320  {
321  const index_type block_end = nvbio::min( block_begin + max_block_size, string_len );
322 
323  // copy all symbols to a temporary buffer
324  priv::device_copy(
325  block_end - block_begin,
326  output + block_begin + 1u,
327  d_block_bwt.begin(),
328  uint32(0) );
329 
330  // and copy the shifted block to the output
331  priv::device_copy(
332  block_end - block_begin,
333  d_block_bwt.begin(),
334  output,
335  block_begin );
336  }
337  }

374 
375  thrust::copy(
376  thrust::device_ptr<const uint32>(d_suffixes),

391  n_output += n_suffixes;
392  }

417  {
418  const uint32 slot = h_slots[i] + 1u; // +1 for the implicit empty suffix
