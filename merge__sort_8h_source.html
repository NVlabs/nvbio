72 template <typename Iterator, typename Compare>
73 NVBIO_HOST_DEVICE
74 bool merge_sort(uint32 n, Iterator A, Iterator B, const Compare cmp)
75 {
76  if (n == 1)
77  return false;
78 
79  // each 1-element run in A is already "sorted":
80  // make successively longer sorted runs of length 2, 4, 8, 16...
81  // until whole array is sorted
82 
83  // sort pairs in place to avoid unnecessary memory traffic
84  const uint32 nn = (n & 1) ? n-1 : n;
85  for (uint32 i = 0; i < nn; i += 2u)
86  {
87  if (cmp( A[i+1], A[i] ))
88  {
89  typename std::iterator_traits<Iterator>::value_type tmp = A[i];
90  A[i] = A[i+1];
91  A[i+1] = tmp;
92  }
93  }
94 
95  // merge longer and longer runs in a loop
96  bool swap = 0;
97  for (uint32 width = 2u; width < n; width *= 2u, swap = !swap)
98  {
99  // array A is full of runs of length width
100  for (uint32 i = 0; i < n; i += 2u * width)
101  {
102  // merge two runs: A[i:i+width-1] and A[i+width:i+2*width-1] to B[]
103  // or copy A[i:n-1] to B[] ( if(i+width >= n) )
104  mergesort::merge( A, i, nvbio::min(i+width, n), nvbio::min(i+2*width, n), B, cmp );
105  }
106 
107  // now work array B is full of runs of length 2*width, swap A and B.
108  Iterator tmp = A;
109  A = B;
110  B = tmp;
111  }
112  return swap;
113 }
