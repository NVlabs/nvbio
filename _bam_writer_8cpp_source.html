84 // creates a cigar string from the supplied alignment
85 void BamWriter::BamWriterPrivate::CreatePackedCigar(const vector<CigarOp>& cigarOperations, string& packedCigar) {
86 
87  // initialize
88  const unsigned int numCigarOperations = (unsigned int)cigarOperations.size();
89  packedCigar.resize(numCigarOperations * BT_SIZEOF_INT);
90 
91  // pack the cigar data into the string
92  unsigned int* pPackedCigar = (unsigned int*)packedCigar.data();
93 
94  unsigned int cigarOp;
95  vector<CigarOp>::const_iterator coIter;
96  for(coIter = cigarOperations.begin(); coIter != cigarOperations.end(); coIter++) {
97 
98  switch(coIter->Type) {
99  case 'M':
100  cigarOp = BAM_CMATCH;
101  break;
102  case 'I':
103  cigarOp = BAM_CINS;
104  break;
105  case 'D':
106  cigarOp = BAM_CDEL;
107  break;
108  case 'N':
109  cigarOp = BAM_CREF_SKIP;
110  break;
111  case 'S':
112  cigarOp = BAM_CSOFT_CLIP;
113  break;
114  case 'H':
115  cigarOp = BAM_CHARD_CLIP;
116  break;
117  case 'P':
118  cigarOp = BAM_CPAD;
119  break;
120  default:
121  printf("ERROR: Unknown cigar operation found: %c\n", coIter->Type);
122  exit(1);
123  }
124 
125  *pPackedCigar = coIter->Length << BAM_CIGAR_SHIFT | cigarOp;
126  pPackedCigar++;
127  }
128 }

130 // encodes the supplied query sequence into 4-bit notation
131 void BamWriter::BamWriterPrivate::EncodeQuerySequence(const string& query, string& encodedQuery) {
132 
133  // prepare the encoded query string
134  const unsigned int queryLen = (unsigned int)query.size();
135  const unsigned int encodedQueryLen = (unsigned int)((queryLen / 2.0) + 0.5);
136  encodedQuery.resize(encodedQueryLen);
137  char* pEncodedQuery = (char*)encodedQuery.data();
138  const char* pQuery = (const char*)query.data();
139 
140  unsigned char nucleotideCode;
141  bool useHighWord = true;
142 
143  while(*pQuery) {
144 
145  switch(*pQuery) {
146  case '=':
147  nucleotideCode = 0;
148  break;
149  case 'A':
150  nucleotideCode = 1;
151  break;
152  case 'C':
153  nucleotideCode = 2;
154  break;
155  case 'G':
156  nucleotideCode = 4;
157  break;
158  case 'T':
159  nucleotideCode = 8;
160  break;
161  case 'N':
162  nucleotideCode = 15;
163  break;
164  default:
165  printf("ERROR: Only the following bases are supported in the BAM format: {=, A, C, G, T, N}. Found [%c]\n", *pQuery);
166  exit(1);
167  }
168 
169  // pack the nucleotide code
170  if(useHighWord) {
171  *pEncodedQuery = nucleotideCode << 4;
172  useHighWord = false;
173  } else {
174  *pEncodedQuery |= nucleotideCode;
175  pEncodedQuery++;
176  useHighWord = true;
177  }
178 
179  // increment the query position
180  pQuery++;
181  }
182 }

186 
187  // open the BGZF file for writing
188  mBGZF.Open(filename, "wb");
189 
190  // ================
191  // write the header
192  // ================
193 
194  // write the BAM signature
195  const unsigned char SIGNATURE_LENGTH = 4;
196  const char* BAM_SIGNATURE = "BAM\1";
197  mBGZF.Write(BAM_SIGNATURE, SIGNATURE_LENGTH);
198 
199  // write the SAM header text length
200  const unsigned int samHeaderLen = (unsigned int)samHeader.size();
201  mBGZF.Write((char*)&samHeaderLen, BT_SIZEOF_INT);
202 
203  // write the SAM header text
204  if(samHeaderLen > 0) {
205  mBGZF.Write(samHeader.data(), samHeaderLen);
206  }
207 
208  // write the number of reference sequences
209  const unsigned int numReferenceSequences = (unsigned int)referenceSequences.size();
210  mBGZF.Write((char*)&numReferenceSequences, BT_SIZEOF_INT);
211 
212  // =============================
213  // write the sequence dictionary
214  // =============================
215 
216  RefVector::const_iterator rsIter;
217  for(rsIter = referenceSequences.begin(); rsIter != referenceSequences.end(); rsIter++) {
218 
219  // write the reference sequence name length
220  const unsigned int referenceSequenceNameLen = (unsigned int)rsIter->RefName.size() + 1;
221  mBGZF.Write((char*)&referenceSequenceNameLen, BT_SIZEOF_INT);
222 
223  // write the reference sequence name
224  mBGZF.Write(rsIter->RefName.c_str(), referenceSequenceNameLen);
225 
226  // write the reference sequence length
227  mBGZF.Write((char*)&rsIter->RefLength, BT_SIZEOF_INT);
228  }

231 // saves the alignment to the alignment archive
232 void BamWriter::BamWriterPrivate::SaveAlignment(const BamAlignment& al) {
233 
234  // initialize
235  const unsigned int nameLen = (unsigned int)al.Name.size() + 1;
236  const unsigned int queryLen = (unsigned int)al.QueryBases.size();
237  const unsigned int numCigarOperations = (unsigned int)al.CigarData.size();
238 
239  // create our packed cigar string
240  string& packedCigar = mpackedCigar;
241  packedCigar.erase( packedCigar.begin(), packedCigar.end() );
242  CreatePackedCigar(al.CigarData, packedCigar);
243  const unsigned int packedCigarLen = (unsigned int)packedCigar.size();
244 
245  // encode the query
246  string& encodedQuery = mencodedQuery;
247  encodedQuery.erase( encodedQuery.begin(), encodedQuery.end() );
248  EncodeQuerySequence(al.QueryBases, encodedQuery);
249  const unsigned int encodedQueryLen = (unsigned int)encodedQuery.size();
250 
251  // store the tag data length
252  const unsigned int tagDataLength = (unsigned int)al.TagData.size() + 1;
253 
254  // assign the BAM core data
255  unsigned int buffer[8];
256  buffer[0] = al.RefID;
257  buffer[1] = al.Position;
258  buffer[2] = (al.Bin << 16) | (al.MapQuality << 8) | nameLen;
259  buffer[3] = (al.AlignmentFlag << 16) | numCigarOperations;
260  buffer[4] = queryLen;
261  buffer[5] = al.MateRefID;
262  buffer[6] = al.MatePosition;
263  buffer[7] = al.InsertSize;
264 
265  // write the block size
266  const unsigned int dataBlockSize = nameLen + packedCigarLen + encodedQueryLen + queryLen + tagDataLength;
267  const unsigned int blockSize = BAM_CORE_SIZE + dataBlockSize;
268  mBGZF.Write((char*)&blockSize, BT_SIZEOF_INT);
269 
270  // write the BAM core
271  mBGZF.Write((char*)&buffer, BAM_CORE_SIZE);
272 
273  // write the query name
274  mBGZF.Write(al.Name.c_str(), nameLen);
275 
276  // write the packed cigar
277  mBGZF.Write(packedCigar.data(), packedCigarLen);
278 
279  // write the encoded query sequence
280  mBGZF.Write(encodedQuery.data(), encodedQueryLen);
281 
282  // write the base qualities
283  mbaseQualities = al.Qualities;
284  char* pBaseQualities = (char*)mbaseQualities.data();
285  for(unsigned int i = 0; i < queryLen; i++) { pBaseQualities[i] -= 33; }
286  mBGZF.Write(pBaseQualities, queryLen);
287 
288  // write the read group tag
289  mBGZF.Write(al.TagData.data(), tagDataLength);
290 }
