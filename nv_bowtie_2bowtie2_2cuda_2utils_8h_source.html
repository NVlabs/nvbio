57 // alloc some slots from a counter using a warp-synchronous reduction
58 NVBIO_FORCEINLINE NVBIO_DEVICE uint32 alloc(uint32* counter, volatile uint32 *warp_broadcast)
59 {
60 #if USE_WARP_SYNCHRONOUS_QUEUES
61  const uint32 mask = __ballot( true );
62  const uint32 pop_scan = __popc( mask << (32u - warp_tid()) );
63  const uint32 pop_count = __popc( mask );
64  if (pop_scan == 0)
65  *warp_broadcast = atomicAdd( counter, pop_count );
66 
67  return *warp_broadcast + pop_scan;
68 #else
69  return atomicAdd( counter, 1u );
70 #endif
71 }
