33 void SEAnalyzer::push(
34  const Alignment& alnL,
35  const Alignment& alnR)
36 {
37  if (alnL.read_id != alnR.read_id ||
38  alnL.read_len != alnR.read_len)
39  {
40  n_mismatched++;
41  return;
42  }
43 
44  mapped.push( alnL.is_mapped(), alnR.is_mapped() );
45  unique.push( alnL.is_unique(), alnR.is_unique() );
46  ambiguous.push( alnL.is_ambiguous(), alnR.is_ambiguous() );
47 
48  if ((alnL.is_mapped() == true) && (alnR.is_mapped() == false)) mapped_L_not_R_by_mapQ.push( log_bin( alnL.mapQ ) );
49  if ((alnR.is_mapped() == true) && (alnL.is_mapped() == false)) mapped_R_not_L_by_mapQ.push( log_bin( alnR.mapQ ) );
50  if ((alnL.is_unique() == true) && (alnR.is_unique() == false)) unique_L_not_R_by_mapQ.push( log_bin( alnL.mapQ ) );
51  if ((alnR.is_unique() == true) && (alnL.is_unique() == false)) unique_R_not_L_by_mapQ.push( log_bin( alnR.mapQ ) );
52  if ((alnL.is_ambiguous() == true) && (alnR.is_ambiguous() == false)) ambiguous_L_not_R_by_mapQ.push( log_bin( alnL.mapQ ) );
53  if ((alnR.is_ambiguous() == true) && (alnL.is_ambiguous() == false)) ambiguous_R_not_L_by_mapQ.push( log_bin( alnR.mapQ ) );
54 
55  if (alnL.is_mapped() && alnR.is_mapped())
56  {
57  const uint32 mapQ_bin = log_bin( alnR.mapQ );
58 
59  uint32 read_flags = 0u;
60 
61  if (alnL.ref_id != alnR.ref_id)
62  {
63  n_different_ref.push( mapQ_bin );
64  n_distant.push( mapQ_bin );
65  read_flags |= Filter::DISTANT;
66  read_flags |= Filter::DIFFERENT_REF;
67  }
68  else if (alndiff::distant( alnL, alnR ))
69  {
70  n_distant.push( mapQ_bin );
71  read_flags |= Filter::DISTANT;
72  }
73 
74  if (alnL.is_rc() != alnR.is_rc())
75  {
76  n_discordant.push( mapQ_bin );
77  read_flags |= Filter::DISCORDANT;
78  }
79 
80  const uint32 length_bin = read_length_bin( alnL.read_len );
81 
82  // generic stats
83  {
84  // ed
85  m_filter( al_stats.lower_ed.push( alnL.ed, alnR.ed, length_bin, alnL.mapQ, alnR.mapQ ), read_flags, Filter::ED, alnL.read_id );
86 
87  // mapQ
88  m_filter( al_stats.higher_mapQ.push( alnL.mapQ, alnR.mapQ, length_bin, alnL.mapQ, alnR.mapQ ), read_flags, Filter::MAPQ, alnL.read_id );
89 
90  // longer mapping
91  al_stats.longer_mapping.push( alnL.mapped_read_bases(), alnR.mapped_read_bases(), length_bin, alnL.mapQ, alnR.mapQ );
92 
93  al_stats.lower_subs.push( alnL.subs, alnR.subs, length_bin, alnL.mapQ, alnR.mapQ );
94  m_filter( al_stats.lower_mms.push( alnL.n_mm, alnR.n_mm, length_bin, alnL.mapQ, alnR.mapQ ), read_flags, Filter::MMS, alnL.read_id );
95  m_filter( al_stats.lower_ins.push( alnL.ins, alnR.ins, length_bin, alnL.mapQ, alnR.mapQ ), read_flags, Filter::INS, alnL.read_id );
96  m_filter( al_stats.lower_dels.push( alnL.dels, alnR.dels, length_bin, alnL.mapQ, alnR.mapQ ), read_flags, Filter::DELS, alnL.read_id );
97 
98  al_stats.higher_pos.push( alnL.pos, alnR.pos, length_bin, alnL.mapQ, alnR.mapQ );
99  }
100  if (read_flags & Filter::DISTANT)
101  {
102  // ed
103  distant_stats.lower_ed.push( alnL.ed, alnR.ed, length_bin, alnL.mapQ, alnR.mapQ );
104 
105  // mapQ
106  distant_stats.higher_mapQ.push( alnL.mapQ, alnR.mapQ, length_bin, alnL.mapQ, alnR.mapQ );
107 
108  // longer mapping
109  distant_stats.longer_mapping.push( alnL.mapped_read_bases(), alnR.mapped_read_bases(), length_bin, alnL.mapQ, alnR.mapQ );
110 
111  distant_stats.lower_subs.push( alnL.subs, alnR.subs, length_bin, alnL.mapQ, alnR.mapQ );
112  distant_stats.lower_mms.push( alnL.n_mm, alnR.n_mm, length_bin, alnL.mapQ, alnR.mapQ );
113  distant_stats.lower_ins.push( alnL.ins, alnR.ins, length_bin, alnL.mapQ, alnR.mapQ );
114  distant_stats.lower_dels.push( alnL.dels, alnR.dels, length_bin, alnL.mapQ, alnR.mapQ );
115  distant_stats.higher_pos.push( alnL.pos, alnR.pos, length_bin, alnL.mapQ, alnR.mapQ );
116  }
117  if (read_flags & Filter::DISCORDANT)
118  {
119  // ed
120  discordant_stats.lower_ed.push( alnL.ed, alnR.ed, length_bin, alnL.mapQ, alnR.mapQ );
121 
122  // mapQ
123  discordant_stats.higher_mapQ.push( alnL.mapQ, alnR.mapQ, length_bin, alnL.mapQ, alnR.mapQ );
124 
125  // longer mapping
126  discordant_stats.longer_mapping.push( alnL.mapped_read_bases(), alnR.mapped_read_bases(), length_bin, alnL.mapQ, alnR.mapQ );
127 
128  discordant_stats.lower_subs.push( alnL.subs, alnR.subs, length_bin, alnL.mapQ, alnR.mapQ );
129  discordant_stats.lower_mms.push( alnL.n_mm, alnR.n_mm, length_bin, alnL.mapQ, alnR.mapQ );
130  discordant_stats.lower_ins.push( alnL.ins, alnR.ins, length_bin, alnL.mapQ, alnR.mapQ );
131  discordant_stats.lower_dels.push( alnL.dels, alnR.dels, length_bin, alnL.mapQ, alnR.mapQ );
132  discordant_stats.higher_pos.push( alnL.pos, alnR.pos, length_bin, alnL.mapQ, alnR.mapQ );
133  }
134  }
135 
136  ++n;
137 }
