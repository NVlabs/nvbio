38 int SequenceDataFile_TXT::nextChunk(SequenceDataEncoder* output, uint32 max_reads, uint32 max_bps)
39 {
40  const char* name = "";
41 
42  uint32 n_reads = 0;
43  uint32 n_bps = 0;
44 
45  const uint32 read_mult =
46  ((m_flags & FORWARD) ? 1u : 0u) +
47  ((m_flags & REVERSE) ? 1u : 0u) +
48  ((m_flags & FORWARD_COMPLEMENT) ? 1u : 0u) +
49  ((m_flags & REVERSE_COMPLEMENT) ? 1u : 0u);
50 
51  while (n_reads + read_mult <= max_reads &&
52  n_bps + read_mult*SequenceDataFile::LONG_READ <= max_bps)
53  {
54  // reset the read
55  m_read_bp.erase( m_read_bp.begin(), m_read_bp.end() );
56 
57  // read an entire line
58  for (uint8 c = get(); c != '\n' && c != 0; c = get())
59  {
60  // if (isgraph(c))
61  if (c >= 0x21 && c <= 0x7E)
62  m_read_bp.push_back( c );
63  }
64 
65  ++m_line;
66 
67  if (m_read_q.size() < m_read_bp.size())
68  {
69  // extend the quality score vector if needed
70  const size_t old_size = m_read_q.size();
71  m_read_q.resize( m_read_bp.size() );
72  for (size_t i = old_size; i < m_read_bp.size(); ++i)
73  m_read_q[i] = char(255);
74  }
75 
76  if (m_read_bp.size())
77  {
78  if (m_flags & FORWARD)
79  {
80  output->push_back(uint32( m_read_bp.size() ),
81  name,
82  &m_read_bp[0],
83  &m_read_q[0],
84  m_quality_encoding,
85  m_truncate_read_len,
86  SequenceDataEncoder::NO_OP );
87  }
88  if (m_flags & REVERSE)
89  {
90  output->push_back(uint32( m_read_bp.size() ),
91  name,
92  &m_read_bp[0],
93  &m_read_q[0],
94  m_quality_encoding,
95  m_truncate_read_len,
96  SequenceDataEncoder::REVERSE_OP );
97  }
98  if (m_flags & FORWARD_COMPLEMENT)
99  {
100  output->push_back(uint32( m_read_bp.size() ),
101  name,
102  &m_read_bp[0],
103  &m_read_q[0],
104  m_quality_encoding,
105  m_truncate_read_len,
106  SequenceDataEncoder::COMPLEMENT_OP );
107  }
108  if (m_flags & REVERSE_COMPLEMENT)
109  {
110  output->push_back(uint32( m_read_bp.size() ),
111  name,
112  &m_read_bp[0],
113  &m_read_q[0],
114  m_quality_encoding,
115  m_truncate_read_len,
116  SequenceDataEncoder::REVERSE_COMPLEMENT_OP );
117  }
118 
119  n_bps += read_mult * (uint32)m_read_bp.size();
120  n_reads += read_mult;
121  }
122 
123  // check for end-of-file
124  if (m_file_state != FILE_OK)
125  break;
126  }
127  return n_reads;
128 }
