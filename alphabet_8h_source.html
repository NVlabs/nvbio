194 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
195 uint32 bits_per_symbol(const Alphabet alphabet)
196 {
197  return alphabet == DNA ? 2 :
198  alphabet == DNA_N ? 4 :
199  alphabet == DNA_IUPAC ? 4 :
200  alphabet == PROTEIN ? 8 :
201  alphabet == RNA ? 2 :
202  alphabet == RNA_N ? 4 :
203  8u;
204 }
