449 //
450 // A template struct used to possibly specialize the implementation of the Gotoh-based alignment based on
451 // the template parameters.
452 //
453 template <uint32 BAND_LEN, AlignmentType TYPE, typename symbol_type>
454 struct gotoh_alignment_score_dispatch<BAND_LEN,TYPE,PatternBlockingTag,symbol_type>
455 {
456  // update a DP row
457  //
458  template <
459  bool CHECK_M,
460  typename context_type,
461  typename query_cache,
462  typename score_type,
463  typename temp_iterator,
464  typename sink_type,
465  typename scoring_type>
466  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
467  static void update_row(
468  context_type& context,
469  const uint32 block,
470  const uint32 M,
471  const uint32 i,
472  const uint32 N,
473  const symbol_type r_i,
474  const query_cache q_cache,
475  temp_iterator temp,
476  score_type& temp_i,
477  score_type* H_band,
478  score_type* F_band,
479  sink_type& sink,
480  const score_type min_score,
481  score_type& max_score,
482  const score_type G_o,
483  const score_type G_e,
484  const score_type zero,
485  const scoring_type scoring)
486  {
487  typedef typename std::iterator_traits<temp_iterator>::value_type temp_cell_type;
488  typedef typename vector_traits<temp_cell_type>::value_type temp_scalar_type;
489 
490  //
491  // NOTE:
492  // It might look as if we were going to make lots of out-of-bounds accesses here,
493  // as we loop across BAND_LEN cells irrespectively of whether the band straddles
494  // the end of the pattern.
495  // However, these accesses don't cause any page faults as they refer to properly
496  // sized temporary arrays (that in practice are placed in registers), and don't
497  // affect the results as they contribute to unused portions of the DP matrices.
498  // The reporting of the scores is properly guarded.
499  //
500 
501  // set the 0-th coefficient in the band to be equal to the (i-1)-th row of the left column (diagonal term)
502  score_type H_diag = temp_i;
503 
504  H_band[0] = temp_i = temp[i].x;
505  score_type E = temp[i].y;
506 
507  #pragma unroll
508  for (uint32 j = 1; j <= BAND_LEN; ++j)
509  {
510  // update F
511  const score_type ftop = F_band[j] + G_e;
512  const score_type htop = H_band[j] + G_o;
513  F_band[j] = nvbio::max( ftop, htop );
514  const DirectionVector fdir = ftop > htop ? DELETION_EXT : SUBSTITUTION;
515 
516  // update E
517  const score_type eleft = E + G_e;
518  const score_type hleft = H_band[j-1] + G_o;
519  E = nvbio::max( eleft, hleft );
520  const DirectionVector edir = eleft > hleft ? INSERTION_EXT : SUBSTITUTION;
521 
522  const int4 info_j = q_cache[ j-1 ];
523  const symbol_type q_j = symbol_type(info_j.x);
524  const score_type m_j = score_type(info_j.y);
525  const score_type qq_j = score_type(info_j.z);
526  const score_type S_ij = (r_i == q_j) ? m_j : scoring.mismatch( r_i, q_j, qq_j );
527  const score_type diagonal = H_diag + S_ij;
528  const score_type top = F_band[j];
529  const score_type left = E;
530  score_type hi = nvbio::max3( left, top, diagonal );
531  hi = (TYPE == LOCAL)? nvbio::max( hi, zero ) : hi; // clamp to zero
532  H_diag = H_band[j];
533  H_band[j] = hi;
534 
535  if ((CHECK_M == false) || (block + j <= M))
536  {
537  context.new_cell(
538  i, N,
539  block + j - 1, M,
540  hi,
541  top > left ?
542  (top > diagonal ? DELETION : SUBSTITUTION) :
543  (left > diagonal ? INSERTION : SUBSTITUTION),
544  edir,
545  fdir );
546  }
547  }
548 
549  // save the last entry of the band
550  temp[i] = make_vector<temp_scalar_type>( H_band[ BAND_LEN ], E );
551 
552  NVBIO_CUDA_ASSERT( H_band[ BAND_LEN ] >= Field_traits<temp_scalar_type>::min() );
553  NVBIO_CUDA_ASSERT( H_band[ BAND_LEN ] <= Field_traits<temp_scalar_type>::max() );
554  NVBIO_CUDA_ASSERT( E >= Field_traits<temp_scalar_type>::min() );
555  NVBIO_CUDA_ASSERT( E <= Field_traits<temp_scalar_type>::max() );
556 
557  max_score = nvbio::max( max_score, H_band[ BAND_LEN ] );
558 
559  if (TYPE == LOCAL)
560  {
561  if (CHECK_M)
562  {
563  // during local alignment we save the best score across all bands
564  for (uint32 j = 1; j <= BAND_LEN; ++j)
565  {
566  if (block + j <= M)
567  sink.report( H_band[j], make_uint2( i+1, block+j ) );
568  }
569  }
570  else
571  {
572  // during local alignment we save the best score across all bands
573  for (uint32 j = 1; j <= BAND_LEN; ++j)
574  sink.report( H_band[j], make_uint2( i+1, block+j ) );
575  }
576  }
577  else if (CHECK_M)
578  {
579  if (TYPE == SEMI_GLOBAL)
580  {
581  // during semi-global alignment we save the best score across the last column H[*][M], at each row
582  save_boundary<BAND_LEN>( block, M, H_band, i, sink );
583  }
584  }
585  }

587  //
588  // Calculate the alignment score between a string and a reference, using the Smith-Waterman algorithm,
589  // using a templated column storage.
590  //
591  // This function is templated over:
592  // - a context that is passed the computed DP matrix values, and can be
593  // used to specialize its behavior.
594  // - a sink that is used to report successful alignments
595  //
596  // Furthermore, the function can be called on a window of the pattern, assuming that the context
597  // will provide the proper initialization for the first column of the corresponding DP matrix window.
598  //
599  // \param context template context class, used to specialize the behavior of the aligner
600  // \param query input pattern (horizontal string)
601  // \param quals input pattern qualities (horizontal string)
602  // \param ref input text (vertical string)
603  // \param scoring scoring scheme
604  // \param min_score minimum output score
605  // \param sink alignment sink
606  // \param window_begin beginning of pattern window
607  // \param window_end end of pattern window
608  //
609  // \return false if early-exited, true otherwise
610  //
611  template <
612  typename context_type,
613  typename string_type,
614  typename qual_type,
615  typename ref_type,
616  typename scoring_type,
617  typename sink_type,
618  typename column_type>
619  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
620  static
621  bool run(
622  const scoring_type& scoring,
623  context_type& context,
624  string_type query,
625  qual_type quals,
626  ref_type ref,
627  const int32 min_score,
628  sink_type& sink,
629  const uint32 window_begin,
630  const uint32 window_end,
631  column_type temp)
632  {
633  //
634  // This function breaks the DP matrix in vertical stripes of BAND_LEN cells,
635  // so as to keep the current reduced row (a band of coefficients) in registers
636  // throughout the entire computation.
637  // Within each stripe, the matrix is updated top-to-bottom, and the right-most
638  // border of the stripe is saved to a local memory array so as to allow resuming
639  // its values when processing the next stripe.
640  //
641  const uint32 M = query.length();
642  const uint32 N = ref.length();
643 
644  typedef int32 score_type;
645  score_type H_band[BAND_LEN+1];
646  score_type F_band[BAND_LEN+1];
647 
648  typedef typename std::iterator_traits<column_type>::value_type temp_cell_type;
649  typedef typename vector_traits<temp_cell_type>::value_type temp_scalar_type;
650 
651  const score_type G_o = scoring.pattern_gap_open();
652  const score_type G_e = scoring.pattern_gap_extension();
653  const score_type zero = score_type(0);
654  const score_type infimum = Field_traits<temp_scalar_type>::min() - nvbio::min( G_o, G_e );
655 
656  int4 q_cache[ BAND_LEN ];
657 
658  // initialize the first column
659  context.init( window_begin, N, temp, scoring, zero, infimum );
660 
661  const uint32 end_block = (window_end == M) ?
662  nvbio::max( BAND_LEN, BAND_LEN * ((M + BAND_LEN-1) / BAND_LEN) ) :
663  window_end + BAND_LEN;
664 
665  // loop across the short edge of the DP matrix (i.e. the columns)
666  for (uint32 block = window_begin; block + BAND_LEN < end_block; block += BAND_LEN)
667  {
668  // save the previous column
669  context.previous_column( block, N, temp );
670 
671  // load a block of entries from each query
672  #pragma unroll
673  for (uint32 t = 0; t < BAND_LEN; ++t)
674  {
675  const symbol_type q = query[ block + t ];
676  const uint8 qq = quals[ block + t ];
677 
678  q_cache[ t ] = make_int4( q, scoring.match(qq), qq, 0 );
679  }
680 
681  // initialize the first band
682  #pragma unroll
683  for (uint32 j = 0; j <= BAND_LEN; ++j)
684  {

856 // A template struct used to possibly specialize the implementation of the Gotoh-based alignment based on

860 struct gotoh_alignment_score_dispatch<BAND_LEN,TYPE,TextBlockingTag,symbol_type>
861 {
862  // update a DP row
863  //
864  template <
865  bool CHECK_N,
866  typename context_type,
867  typename ref_cache,
868  typename score_type,
869  typename temp_iterator,
870  typename sink_type,

873  static void update_row(
874  context_type& context,
875  const uint32 block,
876  const uint32 N,
877  const uint32 i,
878  const uint32 M,
879  const symbol_type q_i,
880  const uint8 qq_i,
881  const score_type m_i,
882  const ref_cache r_cache,
883  temp_iterator temp,
884  score_type& temp_i,
885  score_type* H_band,
886  score_type* F_band,
887  sink_type& sink,
888  const score_type min_score,
889  score_type& max_score,
890  const score_type G_o,
891  const score_type G_e,
892  const score_type zero,
893  const scoring_type scoring)
894  {
895  typedef typename std::iterator_traits<temp_iterator>::value_type temp_cell_type;
896  typedef typename vector_traits<temp_cell_type>::value_type temp_scalar_type;
897 
898  //
899  // NOTE:
900  // It might look as if we were going to make lots of out-of-bounds accesses here,
901  // as we loop across BAND_LEN cells irrespectively of whether the band straddles
902  // the end of the pattern.
903  // However, these accesses don't cause any page faults as they refer to properly
904  // sized temporary arrays (that in practice are placed in registers), and don't
905  // affect the results as they contribute to unused portions of the DP matrices.
906  // The reporting of the scores is properly guarded.
907  //
908 
909  // set the 0-th coefficient in the band to be equal to the (i-1)-th row of the left column (diagonal term)
910  score_type H_diag = temp_i;
911 
912  H_band[0] = temp_i = temp[i].x;
913  score_type E = temp[i].y;
914 
915  #pragma unroll
916  for (uint32 j = 1; j <= BAND_LEN; ++j)
917  {
918  // update F
919  const score_type ftop = F_band[j] + G_e;
920  const score_type htop = H_band[j] + G_o;
921  F_band[j] = nvbio::max( ftop, htop );
922  const DirectionVector fdir = ftop > htop ? INSERTION_EXT : SUBSTITUTION;
923 
924  // update E
925  const score_type eleft = E + G_e;
926  const score_type hleft = H_band[j-1] + G_o;
927  E = nvbio::max( eleft, hleft );
928  const DirectionVector edir = eleft > hleft ? DELETION_EXT : SUBSTITUTION;
929 
930  const symbol_type r_j = r_cache[ j-1 ];
931  const score_type S_ij = (r_j == q_i) ? m_i : scoring.mismatch( r_j, q_i, qq_i );
932  const score_type diagonal = H_diag + S_ij;
933  const score_type top = F_band[j];
934  const score_type left = E;
935  score_type hi = nvbio::max3( left, top, diagonal );
936  hi = (TYPE == LOCAL)? nvbio::max( hi, zero ) : hi; // clamp to zero
937  H_diag = H_band[j];
938  H_band[j] = hi;
939 
940  if ((CHECK_N == false) || (block + j <= N))
941  {
942  context.new_cell(
943  i, M,
944  block + j - 1, N,
945  hi,
946  top > left ?
947  (top > diagonal ? INSERTION : SUBSTITUTION) :
948  (left > diagonal ? DELETION : SUBSTITUTION),
949  edir,
950  fdir );
951  }
952  }
953 
954  // save the last entry of the band
955  temp[i] = make_vector<temp_scalar_type>( H_band[ BAND_LEN ], E );
956 
957  NVBIO_CUDA_ASSERT( H_band[ BAND_LEN ] >= Field_traits<temp_scalar_type>::min() );
958  NVBIO_CUDA_ASSERT( H_band[ BAND_LEN ] <= Field_traits<temp_scalar_type>::max() );
959  NVBIO_CUDA_ASSERT( E >= Field_traits<temp_scalar_type>::min() );
960  NVBIO_CUDA_ASSERT( E <= Field_traits<temp_scalar_type>::max() );
961 
962  max_score = nvbio::max( max_score, H_band[ BAND_LEN ] );
963 
964  if (TYPE == LOCAL)
965  {
966  if (CHECK_N)
967  {
968  // during local alignment we save the best score across all bands
969  for (uint32 j = 1; j <= BAND_LEN; ++j)
970  {
971  if (block + j <= N)
972  sink.report( H_band[j], make_uint2( block+j, i+1 ) );
973  }
974  }
975  else
976  {
977  // during local alignment we save the best score across all bands
978  for (uint32 j = 1; j <= BAND_LEN; ++j)
979  sink.report( H_band[j], make_uint2( block+j, i+1 ) );
980  }
981  }
982  }

1549 // The function returns the resulting source cell.
1550 //
1551 // \tparam BAND_LEN size of the DP band
1552 //
1553 // \tparam CHECKPOINTS number of DP rows between each checkpoint
1554 //
1555 // \tparam checkpoint_type a class to represent the collection of checkpoints,

1561 // as a linear array of size (BAND_LEN*CHECKPOINTS).

1566 // \tparam backtracer_type a class to store the resulting list of backtracking operations.
1567 // A model of \ref Backtracer.
1568 //
1569 // \param checkpoints precalculated checkpoints

1571 // storing all bands between checkpoint_id and checkpoint_id+1.
1572 // \param submatrix precalculated flow submatrix
1573 // \param submatrix_height submatrix width
1574 // \param submatrix_height submatrix height
1575 // \param sink in/out sink of the DP solution
1576 // \param output backtracking output handler
1577 //
1578 // \return true if the alignment source has been found, false otherwise
1579 //
1580 template <
1581  uint32 CHECKPOINTS,
1582  AlignmentType TYPE,
1583  typename scoring_type,
1584  typename checkpoint_type,
1585  typename submatrix_type,
1586  typename backtracer_type>
1587 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
1588 bool alignment_traceback(
1589  const GotohAligner<TYPE,scoring_type> aligner,
1590  checkpoint_type checkpoints,
1591  const uint32 checkpoint_id,
1592  submatrix_type submatrix,
1593  const uint32 submatrix_width,
1594  const uint32 submatrix_height,
1595  uint8& state,
1596  uint2& sink,
1597  backtracer_type& backtracer)
1598 {
1599  //
1600  // Backtrace from the second checkpoint to the first looking up the flow submatrix.
1601  //
1602  int32 current_row = sink.x;
1603  int32 current_col = sink.y - checkpoint_id*CHECKPOINTS - 1u;
1604 
1605  NVBIO_CUDA_DEBUG_ASSERT( current_row > 0 &&
1606  current_row <= submatrix_height, "sw::alignment_backtrack(): sink (%u,%u) -> local x coordinate %d not in [0,%d[\n", sink.x, sink.y, current_row, submatrix_height );
1607  NVBIO_CUDA_DEBUG_ASSERT( current_col >= 0, "sw::alignment_backtrack(): sink (%u,%u) -> local y coordinate %d not in [0,%u[ (checkpt %u)\n", sink.x, sink.y, current_col, submatrix_width, checkpoint_id );
1608  NVBIO_CUDA_DEBUG_ASSERT( current_col < submatrix_width, "sw::alignment_backtrack(): sink (%u,%u) -> local y coordinate %d not in [0,%u[ (checkpt %u)\n", sink.x, sink.y, current_col, submatrix_width, checkpoint_id );
1609 
1610  while (current_row > 0 &&
1611  current_col >= 0)
1612  {
1613  const uint32 submatrix_cell = (current_row-1u) * CHECKPOINTS + current_col;
1614  const uint8 op = submatrix[ submatrix_cell ];
1615  const uint8 h_op = op & HMASK;
1616 
1617  if (TYPE == LOCAL)
1618  {
1619  if (state == HSTATE && h_op == SINK)
1620  {
1621  sink.x = current_row;
1622  sink.y = current_col + checkpoint_id*CHECKPOINTS + 1u;
1623  return true;
1624  }
1625  }
1626 
1627  if (state == ESTATE)
1628  {
1629  if ((op & INSERTION_EXT) == 0u) state = HSTATE;
1630  --current_col; backtracer.push( INSERTION );
1631  }
1632  else if (state == FSTATE)
1633  {
1634  if ((op & DELETION_EXT) == 0u) state = HSTATE;
1635  --current_row; backtracer.push( DELETION );
1636  }
1637  else
1638  {
1639  if (h_op == INSERTION)
1640  state = ESTATE;
1641  else if (h_op == DELETION)
1642  state = FSTATE; // NOTE: commenting this line reverts backtracking to plain SW behavior
1643  else
1644  {
1645  --current_col;
1646  --current_row;
1647  backtracer.push( SUBSTITUTION );
1648  }
1649  }
1650  }
1651  sink.x = current_row;
1652  sink.y = current_col + checkpoint_id*CHECKPOINTS + 1u;
1653  return current_row ? false : true; // if current_row == 0 we reached the end of the alignment along the text
1654 }
