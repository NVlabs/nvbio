79 void OutputFile::readback(struct CPUOutputBatch& cpu_batch,
80  const struct GPUOutputBatch& gpu_batch,
81  const AlignmentMate mate,
82  const AlignmentScore score)
83 {
84  Timer timer;
85  timer.start();
86 
87  // read back the scores
88  // readback_scores will do the "right thing" based on the pass and score that we're getting
89  // (i.e., it'll update either the first or second score for either the anchor or opposite mate)
90  gpu_batch.readback_scores(cpu_batch.best_alignments, mate, score);
91 
92  if (score == BEST_SCORE)
93  {
94  // if this is the best alignment, read back CIGARs and MD strings as well
95  gpu_batch.readback_cigars(cpu_batch.cigar[mate]);
96  gpu_batch.readback_mds(cpu_batch.mds[mate]);
97 
98  if (mate == MATE_1)
99  {
100  // mate 1 best score comes first; stash the count
101  cpu_batch.count = gpu_batch.count;
102 
103  // set up the read data pointers
104  // this is not strictly related to which mate or scoring pass we're processing,
105  // but must be done once per batch, so we do it here
106  cpu_batch.read_data[MATE_1] = read_data_1;
107  cpu_batch.read_data[MATE_2] = read_data_2;
108  }
109  }
110 
111  // sanity check to make sure the number of reads matches what we got previously
112  // (for mate 1 best score this will always pass due to the assignment above)
113  NVBIO_CUDA_ASSERT(cpu_batch.count == gpu_batch.count);
114 
115  timer.stop();
116  iostats.output_process_timings.add(gpu_batch.count, timer.seconds());
117 
118  iostats.alignments_DtoH_time += timer.seconds();
119  iostats.alignments_DtoH_count += gpu_batch.count;
120 }

122 OutputFile *OutputFile::open(const char *file_name, AlignmentType aln_type, BNT bnt)
123 {
124  // parse out file extension; look for .sam, .bam suffixes
125  uint32 len = uint32(strlen(file_name));
126 
127  if (strcmp(file_name, "/dev/null") == 0)
128  {
129  return new OutputFile(file_name, aln_type, bnt);
130  }
131 
132  if (len >= strlen(".sam"))
133  {
134  if (strcmp(&file_name[len - strlen(".sam")], ".sam") == 0)
135  {
136  return new SamOutput(file_name, aln_type, bnt);
137  }
138  }
139 
140  if (len >= strlen(".bam"))
141  {
142  if (strcmp(&file_name[len - strlen(".bam")], ".bam") == 0)
143  {
144  return new BamOutput(file_name, aln_type, bnt);
145  }
146  }
147 
148  if (len >= strlen(".dbg"))
149  {
150  if (strcmp(&file_name[len - strlen(".dbg")], ".dbg") == 0)
151  {
152  return new DebugOutput(file_name, aln_type, bnt);
153  }
154  }
155 
156  log_warning(stderr, "could not determine file type for %s; guessing SAM\n", file_name);
157  return new SamOutput(file_name, aln_type, bnt);
158 }
