362 // enact the batch execution
363 //
364 template <typename stream_type>
365 void BatchedAlignmentScore<stream_type,DeviceThreadScheduler>::enact(stream_type stream, uint64 temp_size, uint8* temp)
366 {
367  const uint32 column_size = equal<typename aligner_type::algorithm_tag,PatternBlockingTag>() ?
368  uint32( stream.max_pattern_length() ) :
369  uint32( stream.max_text_length() );
370 
371  // if the column is small, let's just use statically allocated local memory
372  if (column_size <= 1024)
373  {
374  const uint32 n_blocks = (stream.size() + BLOCKDIM-1) / BLOCKDIM;
375 
376  lmem_batched_alignment_score_kernel<BLOCKDIM,1024> <<<n_blocks, BLOCKDIM>>>(
377  stream,
378  (cell_type*)temp,
379  stream.size() );
380  }
381  else
382  {
383  // the column is large to be allocated in local memory, let's use global memory
384  const uint64 min_temp_size = min_temp_storage(
385  stream.max_pattern_length(),
386  stream.max_text_length(),
387  stream.size() );
388 
389  nvbio::vector<device_tag,uint8> temp_vec;
390  if (temp == NULL)
391  {
392  temp_size = nvbio::max( min_temp_size, temp_size );
393  temp_vec.resize( temp_size );
394  temp = nvbio::plain_view( temp_vec );
395  }
396 
397  // set the queue capacity based on available memory
398  const uint32 queue_capacity = uint32( temp_size / column_storage( stream.max_pattern_length(), stream.max_text_length() ) );
399 
400  if (queue_capacity >= stream.size())
401  {
402  const uint32 n_blocks = (stream.size() + BLOCKDIM-1) / BLOCKDIM;
403 
404  batched_alignment_score_kernel<BLOCKDIM> <<<n_blocks, BLOCKDIM>>>(
405  stream,
406  (cell_type*)temp,
407  stream.size() );
408  }
409  else
410  {
411  // compute the number of blocks we are going to launch
412  const uint32 n_blocks = nvbio::max( nvbio::min(
413  (uint32)cuda::max_active_blocks( persistent_batched_alignment_score_kernel<BLOCKDIM,stream_type,cell_type>, BLOCKDIM, 0u ),
414  queue_capacity / BLOCKDIM ), 1u );
415 
416  persistent_batched_alignment_score_kernel<BLOCKDIM> <<<n_blocks, BLOCKDIM>>>(
417  stream,
418  (cell_type*)temp,
419  queue_capacity );
420  }
421  }
422 }

470 {

472  stream.max_pattern_length(),
473  stream.max_text_length(),

476  nvbio::vector<device_tag,uint8> temp_vec;

674  {
675  const uint32 work_id = thread_id + stream_begin;
676 
677  if (work_id < stream_end)
678  batched_alignment_traceback<CHECKPOINTS>( stream, checkpoints, submatrices, columns, stride, work_id, thread_id );
679  }
680 }

689 template <uint32 CHECKPOINTS, typename stream_type>
690 struct BatchedAlignmentTraceback<CHECKPOINTS, stream_type,DeviceThreadScheduler>
691 {
692  static const uint32 BLOCKDIM = 128;
693 
694  typedef typename stream_type::aligner_type aligner_type;
695  typedef typename column_storage_type<aligner_type>::type cell_type;
696 
699  static uint32 column_storage(const uint32 max_pattern_len, const uint32 max_text_len)
700  {
701  const uint32 column_size = equal<typename aligner_type::algorithm_tag,PatternBlockingTag>() ?
702  uint32( max_text_len * sizeof(cell_type) ) :
703  uint32( max_pattern_len * sizeof(cell_type) );
704 
705  return align<4>( column_size );
706  }
707 
710  static uint32 checkpoint_storage(const uint32 max_pattern_len, const uint32 max_text_len)
711  {
712  if (equal<typename aligner_type::algorithm_tag,PatternBlockingTag>())
713  return align<4>( uint32( max_text_len * ((max_pattern_len + CHECKPOINTS-1) / CHECKPOINTS) * sizeof(cell_type) ) );
714  else
715  return align<4>( uint32( max_pattern_len * ((max_text_len + CHECKPOINTS-1) / CHECKPOINTS) * sizeof(cell_type) ) );
716  }
717 
720  static uint32 submatrix_storage(const uint32 max_pattern_len, const uint32 max_text_len)
721  {
722  if (equal<typename aligner_type::algorithm_tag,PatternBlockingTag>())
723  {
724  typedef typename stream_type::aligner_type aligner_type;
725  const uint32 BITS = direction_vector_traits<aligner_type>::BITS;
726  const uint32 ELEMENTS_PER_WORD = 32 / BITS;
727  return ((max_text_len * CHECKPOINTS + ELEMENTS_PER_WORD-1) / ELEMENTS_PER_WORD) * sizeof(uint32);
728  }
729  else
730  {
731  typedef typename stream_type::aligner_type aligner_type;
732  const uint32 BITS = direction_vector_traits<aligner_type>::BITS;
733  const uint32 ELEMENTS_PER_WORD = 32 / BITS;
734  return ((max_pattern_len * CHECKPOINTS + ELEMENTS_PER_WORD-1) / ELEMENTS_PER_WORD) * sizeof(uint32);
735  }
736  }
737 
740  static uint32 element_storage(const uint32 max_pattern_len, const uint32 max_text_len)
741  {
742  return column_storage( max_pattern_len, max_text_len ) +
743  checkpoint_storage( max_pattern_len, max_text_len ) +
744  submatrix_storage( max_pattern_len, max_text_len );
745  }
746 
749  static uint64 min_temp_storage(const uint32 max_pattern_len, const uint32 max_text_len, const uint32 stream_size);
750 
753  static uint64 max_temp_storage(const uint32 max_pattern_len, const uint32 max_text_len, const uint32 stream_size);
754 
757  void enact(stream_type stream, uint64 temp_size = 0u, uint8* temp = NULL);
758 };

776 // enact the batch execution
777 //
778 template <uint32 CHECKPOINTS, typename stream_type>
779 void BatchedAlignmentTraceback<CHECKPOINTS,stream_type,DeviceThreadScheduler>::enact(stream_type stream, uint64 temp_size, uint8* temp)
780 {
781  const uint64 min_temp_size = min_temp_storage(
782  stream.max_pattern_length(),
783  stream.max_text_length(),
784  stream.size() );
785 
786  nvbio::vector<device_tag,uint8> temp_vec;
787  if (temp == NULL)
788  {
789  temp_size = nvbio::max( min_temp_size, temp_size );
790  temp_vec.resize( temp_size );
791  temp = nvbio::plain_view( temp_vec );
792  }
793 
794  // set the queue capacity based on available memory
795  const uint32 max_pattern_len = stream.max_pattern_length();
796  const uint32 max_text_len = stream.max_text_length();
797  const uint32 queue_capacity = uint32( temp_size / element_storage( max_pattern_len, max_text_len ) );
798 
799  const uint64 column_size = column_storage( max_pattern_len, max_text_len );
800  const uint64 checkpoints_size = checkpoint_storage( max_pattern_len, max_text_len );
801 
802  if (queue_capacity >= stream.size())
803  {
804  const uint32 n_blocks = (stream.size() + BLOCKDIM-1) / BLOCKDIM;
805 
806  cell_type* checkpoints = (cell_type*)(temp);
807  cell_type* columns = (cell_type*)(temp + (checkpoints_size) * stream.size());
808  uint32* submatrices = (uint32*) (temp + (checkpoints_size + column_size) * stream.size());
809 
810  batched_alignment_traceback_kernel<BLOCKDIM,CHECKPOINTS> <<<n_blocks, BLOCKDIM>>>(
811  stream,
812  checkpoints,
813  submatrices,
814  columns,
815  stream.size() );
816  }
817  else
818  {
819  // compute the number of blocks we are going to launch
820  const uint32 n_blocks = nvbio::max( nvbio::min(
821  (uint32)cuda::max_active_blocks( persistent_batched_alignment_traceback_kernel<BLOCKDIM,CHECKPOINTS,stream_type,cell_type>, BLOCKDIM, 0u ),
822  queue_capacity / BLOCKDIM ), 1u );
823 
824  cell_type* checkpoints = (cell_type*)(temp);
825  cell_type* columns = (cell_type*)(temp + (checkpoints_size) * queue_capacity);
826  uint32* submatrices = (uint32*) (temp + (checkpoints_size + column_size) * queue_capacity);
827 
828  persistent_batched_alignment_traceback_kernel<BLOCKDIM,CHECKPOINTS> <<<n_blocks, BLOCKDIM>>>(
829  stream,
830  checkpoints,
831  submatrices,
832  columns,
833  queue_capacity );
834  }
835 }
