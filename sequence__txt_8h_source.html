120 inline uint8 SequenceDataFile_TXT::get(void)
121 {
122  if (m_buffer_pos >= m_buffer_size)
123  {
124  // grab more data from the underlying file
125  m_file_state = fillBuffer();
126  m_buffer_pos = 0;
127 
128  // if we failed to read more data, return \0
129  if (m_file_state != FILE_OK)
130  return 0;
131  }
132 
133  return m_buffer[m_buffer_pos++];
134 }
