47 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE uint8 dna_to_iupac16(const uint8 c)
48 {
49  // DNA: A, C, G, T -> { 0, 1, 2, 3 }
50  // IUPAC16: A, C, G, T -> { 1, 2, 4, 8 }
