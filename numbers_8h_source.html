530 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
531 uint64 hash3(uint64 key)
532 {
533  uint32 hash = 0u;
534 
535  #if defined(__CUDA_ARCH__)
536  #pragma unroll
537  #endif
538  for (uint32 i = 0; i < 8; ++i)
539  {
540  hash = (hash << 4) + ((key >> (i*8)) & 255u); // shift/mix
541 
542  // get high nybble
543  const uint32 hi_bits = hash & 0xF0000000;
544  if (hi_bits != 0u)
545  hash ^= hi_bits >> 24; // xor high nybble with second nybble
