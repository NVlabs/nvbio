107 // generate table for counting 11,10,01,00(pattern) for 8 bits number
108 // table [no# ] = representation ( # of count-pattern, . , . , . )
109 // ---------------------------------------------------------------------------
110 // e.g cnt_table[11111111] = 0x04000000 ( 4-11, 0-10, 0-01, 0-00 )
111 // cnt_table[00100001] = 0x00010102 ( 0-11, 1-10, 1-01, 2-00 )
112 // cnt_table[00000001] = 0x00000103 ( 0-11, 0-10, 1-01, 3-00 )
113 inline void gen_2bit_count_table(uint32* count_table)
114 {
115  for (int i = 0; i != 256; ++i)
116  {
117  uint32 x = 0;
118  for (int j = 0; j != 4; ++j)
119  x |= (((i&3) == j) + ((i>>2&3) == j) + ((i>>4&3) == j) + (i>>6 == j)) << (j<<3);
120 
121  count_table[i] = x;
122  }
123 }
