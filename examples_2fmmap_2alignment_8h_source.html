116  // initialize the i-th context
117  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
118  bool init_context(
119  const uint32 i,
120  context_type* context) const
121  {
122  // fetch the diagonal for the i-th alignment task
123  const uint2 diagonal = m_diagonals[i];
124 
125  // retrieve the read id and the text position
126  const uint32 read_id = diagonal.y;
127  const uint32 text_pos = diagonal.x;
128 
129  // fetch the read range
130  context->read_range = m_reads.get_range( read_id );
131  const uint32 read_len = context->read_range.y - context->read_range.x;
132 
133  // compute the segment of text to align to
134  context->genome_begin = text_pos > BAND_LEN/2 ? text_pos - BAND_LEN/2 : 0u;
135  context->genome_end = nvbio::min( context->genome_begin + read_len + BAND_LEN, m_genome_len );
136 
137  // initialize the sink
138  context->sink = aln::BestSink<int16>();
139 
140  // setup the minimum score
141  context->min_score = Field_traits<int16>::min();
142  return true;
143  }
