225 inline uint32 balance_batch_size(uint32 batch_size, uint32 total_count, uint32 thread_count)
226 {
227  // How many batches we'd get with the proposed batch_size
228  const uint32 batch_count = util::divide_ri(total_count, batch_size);
229  // How many rounds we'd need for those batches
230  const uint32 rounds = util::divide_ri(batch_count, thread_count);
231  // Might as well assume all threads should work, and see how many batches
232  // they would consume
233  const uint32 bal_batches = rounds * thread_count;
234  // So that the batch size that will attain it, is computed as follows
235  return util::divide_ri(total_count, bal_batches);
236 }
