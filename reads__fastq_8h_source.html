119 inline uint8 ReadDataFile_FASTQ_parser::get(void)
120 {
121  if (m_buffer_pos >= m_buffer_size)
122  {
123  // grab more data from the underlying file
124  m_file_state = fillBuffer();
125  m_buffer_pos = 0;
126 
127  // if we failed to read more data, return \0
128  if (m_file_state != FILE_OK)
129  return 0;
130  }
131 
132  return m_buffer[m_buffer_pos++];
133 }
