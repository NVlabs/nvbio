120  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
121  uint2 range(const uint64 g) const
122  {
123  const uint32 i = uint32( g / WORD_SIZE );
124  const uint32 j = uint32( g % WORD_SIZE );
125 
126  // check whether the j-th bit of I[i] is set
127  if ((I[i] & (1u << j)) == 0u)
128  return make_uint2( 0u, 0u );
129 
130  // compute j' such that bit j is the j'-th set bit in I[i]
131  const uint32 j_prime = popc( I[i] & ((1u << j) - 1u) );
132 
133  return make_uint2(
134  SS[ S[i] + j_prime ],
135  SS[ S[i] + j_prime + 1u ] );
136  }
137 
140  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
141  uint2 operator() (const uint64 g) const { return range( g ); }
