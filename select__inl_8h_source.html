732  const uint32 n_multi,

737  // single-hit or batched-hit selection, and whether we use randomization or not.

742  {

755  rand_select(

758  context,

761  }
762  }
763  else

765  if (n_multi > 1)
766  {
767  // batched-hit selection

771  context,
772  scoring_queues,
773  n_multi,
774  params );
775  }

780  read_batch,
