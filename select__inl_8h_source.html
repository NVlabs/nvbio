503 // Prepare for a round of seed extension by selecting a set of up
504 // to 'n_multi' next SA rows.
505 // For each read in the input queue, this kernel generates:
506 // 1. one or zero output reads, in the main output read queue,

509 // one in 'seed_queue', storing information about the seed hit,

534 //
535 // Prepare for a round of seed extension by selecting the next SA rows for each read
536 //
537 template <typename BatchType, typename ContextType>
538 void select(
539  const BatchType read_batch,
540  SeedHitDequeArrayDeviceView hits,
541  uint32* rseeds,
542  const ContextType context,
543  ScoringQueuesDeviceView scoring_queues,
544  const uint32 n_multi,
545  const ParamsPOD params)
546 {
547  //
548  // Dispatch the call to the proper version based on whether we are performing
549  // single-hit or batched-hit selection, and whether we use randomization or not.
550  //
551  if (n_multi > 1)
552  {
553  // batched-hit selection
554  select_multi(
555  read_batch,
556  hits,
557  context,
558  scoring_queues,
559  n_multi,
560  params );
561  }
562  else if (params.randomized)
563  {
564  // randomized single-hit selection
565  rand_select(
566  read_batch,
567  hits, rseeds,
568  context,
569  scoring_queues,
570  params );
571  }
572  else
573  {
574  // non-randomized single-hit selection
575  select(
576  read_batch,
577  hits,
578  context,
579  scoring_queues,
580  params );
581  }
