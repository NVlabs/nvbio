55  // gather device memory stats

57  device_memory( &free_device, &total_device );
58  log_stats(stderr, " device has %ld of %ld MB free\n", free_device/1024/1024, total_device/1024/1024);

60  const uint32 bytes_per_word = 4u;
61  const uint64 needed_bytes = (sampled_words + trusted_words)*bytes_per_word + 256*1024*1024;
62 
63  if (needed_bytes > free_device)
64  {
65  log_warning(stderr, " insufficient memory: %.2f GB required\n", float( needed_bytes ) / float(1024*1024*1024));
66  return false;
67  }
68 
69  log_info(stderr, " allocating sampled kmer filter (%.2f GB)\n", float( sampled_words*bytes_per_word ) / float(1024*1024*1024));
70  sampled_kmers_storage.resize( sampled_words, 0u );
71 
72  log_info(stderr, " allocating trusted kmer filter (%.2f GB)\n", float( trusted_words*bytes_per_word ) / float(1024*1024*1024));
73  trusted_kmers_storage.resize( trusted_words, 0u );
74 
75  stats.resize( 10 );
76  return true;
77  }

131 inline
132 void merge(BloomFilters<host_tag>* h_bloom_filters, const uint32 device_count, BloomFilters<device_tag>* d_bloom_filters, const KmersType type)
133 {
134  // merge the Bloom filters on the host
135  if (h_bloom_filters && device_count)
136  {
137  log_info(stderr," merge filters\n");
138 
139  // get the host Bloom filter
140  nvbio::vector<host_tag,uint32>& bf = h_bloom_filters->get_kmers( type );
141  nvbio::vector<host_tag,uint32> bf2;
142 
143  // merge with all the device ones
144  for (uint32 d = 0; d < device_count; ++d)
145  {
146  d_bloom_filters[d].get_kmers( type, bf2 );
147 
148  #pragma omp parallel for
149  for (int64 i = 0; i < (int64)bf.size(); ++i)
150  bf[i] |= bf2[i];
151  }
152 
153  // broadcast the merged Bloom filter to all devices
154  for (uint32 d = 0; d < device_count; ++d)
155  d_bloom_filters[d].set_kmers( type, bf );
156  }
157  else if (device_count > 1)
158  {
159  log_info(stderr," merge filters\n");
160  nvbio::vector<host_tag,uint32> bf;
161  nvbio::vector<host_tag,uint32> bf2;
162 
163  // get the first device Bloom filter
164  d_bloom_filters[0].get_kmers( type, bf );
165 
166  // merge with all the rest
167  for (uint32 d = 1; d < device_count; ++d)
168  {
169  d_bloom_filters[d].get_kmers( type, bf2 );
170 
171  #pragma omp parallel for
172  for (int64 i = 0; i < (int64)bf.size(); ++i)
173  bf[i] |= bf2[i];
174  }
175 
176  // broadcast the merged Bloom filter to all devices
177  for (uint32 d = 0; d < device_count; ++d)
178  d_bloom_filters[d].set_kmers( type, bf );
179  }
180 }

209 struct block_occupancy_functor
210 {
211  typedef uint4 argument_type;
212  typedef double result_type;
213 
214  block_occupancy_functor(const uint32 k) : K(float(k)) {}
215 
216  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
217  double operator() (const uint4 block) const
218  {
219  const uint32 used = popc( block.x ) + popc( block.y ) + popc( block.z ) + popc( block.w );
220  return powf( float(used)/128.0f, K );
221  }

226 // compute Bloom filter usage statistics
227 template <typename system_tag>
228 void compute_bloom_filter_stats(
229  const BloomFilters<system_tag>& bloom_filters,
230  const KmersType type,
231  const uint32 K,
232  float& occupancy,
233  float& approx_size,
234  float& fp)
235 {
236  typedef typename if_equal<system_tag,device_tag,thrust::device_ptr<const uint32>, const uint32*>::type uint_pointer_type;
237  typedef typename if_equal<system_tag,device_tag,thrust::device_ptr<const uint4>, const uint4*>::type uint4_pointer_type;
238 
239  bloom_filters.set_device();
240 
241  const nvbio::vector<system_tag,uint32>& bf = bloom_filters.get_kmers( type );
242 
243  const uint32 n_words = uint32( bf.size() );
244  const uint32* words = raw_pointer( bf );
245 
246  // compute the number of bits set
247  nvbio::vector<system_tag,uint8> temp_storage;
248 
249  const uint64 bits_per_word = 32;
250 
251  const uint64 bits_set = nvbio::reduce(
252  n_words,
253  thrust::make_transform_iterator(
254  thrust::make_transform_iterator(
255  uint_pointer_type( words ), popc_functor<uint32>() ),
256  cast_functor<uint32,uint64>() ),
257  thrust::plus<uint64>(),
258  temp_storage );
259 
260  occupancy = float( double( bits_set ) / double( n_words * bits_per_word ) );
261  approx_size = float( -(double( n_words * bits_per_word ) * std::log( 1.0 - occupancy )) / double(K) );
262 
263  #if 0
264  // compute the actual false positive rate
265  fp = std::pow( occupancy, K );
266  #else
267  // compute the actual false positive rate - using the block-occupancy
268  fp = float( nvbio::reduce(
269  n_words / 4,
270  thrust::make_transform_iterator(
271  uint4_pointer_type( (const uint4*)words ),
272  block_occupancy_functor( K ) ),
273  thrust::plus<double>(),
274  temp_storage ) / double(n_words / 4) );
275  #endif
276 }
