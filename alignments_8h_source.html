80 struct Alignment
81 {
82  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
83  static uint32 max_ed() { return 255u; }
84 
85  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
86  static int32 max_score() { return (1 << 17) - 1; }
87 
88  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
89  static int32 min_score() { return -((1 << 17) - 1); }
90 
91  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE Alignment() {}
92  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE Alignment(
93  const uint32 pos,
94  const uint32 ed,
95  const int32 score,
96  const uint32 rc,
97  const uint32 mate = 0,
98  const bool paired = false,
99  const bool discordant = false)
100  {
101  //NVBIO_CUDA_ASSERT( ed < 1024u );
102  //NVBIO_CUDA_ASSERT( score >= min_score() && score <= max_score() );
103  m_align = pos;
104  m_ed = ed;
105  m_score = score < 0 ? uint32( -score ) : uint32( score );
106  m_score_sgn = score < 0 ? 1u : 0u;
107  m_rc = rc;
108  m_mate = mate;
109  m_paired = paired ? 1u : 0u;
110  m_discordant = discordant ? 1u : 0u;
111  }
112 
113  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE int32 score() const { return m_score_sgn ? -int32( m_score ) : int32( m_score ); }
114  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_aligned() const { return m_align != uint32(-1); }
115  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 alignment() const { return m_align; }
116  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_rc() const { return m_rc; }
117  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 ed() const { return m_ed; }
118  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 sink() const { return m_ed; }
119  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 mate() const { return m_mate; }
120  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_paired() const { return (m_paired == true) && is_aligned(); }
121  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_unpaired() const { return (m_paired == false) && is_aligned(); }
122  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_concordant() const { return (m_paired == true) && (m_discordant == false); }
123  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_discordant() const { return (m_paired == true) && (m_discordant == true); }
124 
125  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE
126  static Alignment invalid() { return Alignment( uint32(-1), max_ed(), max_score(), 0u, 0u, false ); }
127 
128  uint32 m_score_sgn : 1, m_score:17, m_ed:10, m_rc:1, m_mate:1, m_paired:1, m_discordant:1;
129  uint32 m_align;
130 };
