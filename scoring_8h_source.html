295  NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
296  int score(const uint8 read, const uint8 ref_mask, int q) const
297  {
298  return (read > 3 || ref_mask > 15) ? // is there an N?
299  -m_np(q) : // N score
300  ((ref_mask & (1u << read)) != 0) ? // is there a match ?
301  m_match(q) : // match score
302  -m_mmp(q); // mismatch score
303  }
