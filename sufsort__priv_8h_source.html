275 uint32 extract_word_generic(
276  const string_type string,
277  const index_type string_len,
278  const index_type suffix_idx,
279  const uint32 w)
280 {
281  const uint32 SYMBOLS_PER_WORD = uint32(WORD_BITS - DOLLAR_BITS)/SYMBOL_SIZE;
282  const uint32 SYMBOL_OFFSET = uint32(WORD_BITS) - SYMBOL_SIZE;
283 
284  uint32 word = 0u;
285  for (uint32 j = 0; j < SYMBOLS_PER_WORD; ++j)
286  {
287  const index_type jj = suffix_idx + w*SYMBOLS_PER_WORD + j;
288  const uint32 c = jj < string_len ? string[jj] : 0u;
289  word |= (c << (SYMBOL_OFFSET - j*SYMBOL_SIZE));
290  }
291 
292  if (DOLLAR_BITS)
293  {
294  // encode the dollar's position in the least significant bits of the word
295  const uint32 dollar_offset =
296  string_len <= suffix_idx + w*SYMBOLS_PER_WORD + SYMBOLS_PER_WORD ? // is there a dollar sign?
297  (string_len < suffix_idx + w*SYMBOLS_PER_WORD) ? 0u :
298  uint32(string_len - suffix_idx - w*SYMBOLS_PER_WORD) :
299  (1u << DOLLAR_BITS)-1u; // no dollar sign in this word
300 
301  return word | dollar_offset;
302  }
303  else
304  return word;
305 }

319 typename std::iterator_traits<storage_type>::value_type extract_word_packed(
320  const storage_type base_words,
321  const index_type string_len,
322  const index_type string_off,
323  const sufindex_type suffix_idx,
324  const uint32 w)
325 {
326  typedef typename std::iterator_traits<storage_type>::value_type word_type;
327 
328  const uint32 STORAGE_BITS = uint32( 8u * sizeof(word_type) );
329  const uint32 STORAGE_SYMBOLS = STORAGE_BITS / SYMBOL_SIZE;
330  const uint32 SYMBOLS_PER_WORD = uint32(WORD_BITS - DOLLAR_BITS)/SYMBOL_SIZE;
331  //const uint32 SYMBOL_OFFSET = uint32(WORD_BITS) - SYMBOL_SIZE;
332 
333  const sufindex_type suffix_off = suffix_idx + w*SYMBOLS_PER_WORD; // suffix offset
334 
335  // do we have any symbols to encode?
336  if (suffix_off >= string_len)
337  return 0u;
338 
339  const index_type range_len = string_len - suffix_off; // partial suffix length
340  const index_type range_off = string_off + suffix_off; // partial suffix offset
341 
342  const uint32 n_symbols = (uint32)nvbio::min( range_len, index_type(SYMBOLS_PER_WORD) ); // symbols to pack
343 
344  //
345  // As SYMBOLS_PER_WORD is less than 32, we know that the symbols we are looking for
346  // will span at most 2 32-bit words.
347  // Of the n_symbols we want to read, there might be n1 in the first word, and n2 in the
348  // second word.
349  // As the beginning of our symbol stream (range_off) might stride the 32-bit word boundary,
350  // the highest m1 = range_off % 16 symbols of the first word might have to be discarded,
351  // and we'll find our n1 symbols in the following position:
352  //
353  // |-------------------------------------------|
354  // |* * * * * *| x x x x x x x x x | * * * * * |
355  // |-------------------------------------------|
356  // | m1 | n1 | r1 |
357  //
358  // What we do is shifting the n1 symbols to the top of the 32-bit word (i.e. m1 to the left).
359  // Clearing the remaining symbols is only needed if n1 == n_symbols; if n1 < n_symbols, r1 will
360  // be necessarily zero.
361  //
362  // At this point, we might have n2 more symbols to read in the highest bits of the second word:
363  //
364  // |-------------------------------------------|
365  // | y y y y y y y y | * * * * * * * * * * * * |
366  // |-------------------------------------------|
367  // | n2 | r2 |
368  //
369  // which we need to shift right by (n1*SYMBOL_SIZE) bits.
370  // At the very end, we'll shift everything right by (32 - WORD_BITS) bits in order to have
371  // our output tightly packed in the lowest WORD_BITS:
372  //
373  // 32 WORD_BITS DOLLAR_BITS 0
374  // |-----------|-----------------------|-------|
375  // | * * * * * | x x x x | y y y | 0 0 | $ $ $ | // notice the possible presence of 0's before
376  // |-------------------------------------------| // the $ sign: these are bits that need to be
377  // | | n1 | n2 | | | // cleared if the suffix is short
378  //
379 
380  const uint32 k1 = uint32( range_off/STORAGE_SYMBOLS ); // index of the first word
381 
382  const uint32 m1 = range_off & (STORAGE_SYMBOLS-1); // offset in the word
383  const uint32 r1 = STORAGE_SYMBOLS - m1; // symbols to read
384  const word_type word1 = (base_words[ k1 ] << (m1*SYMBOL_SIZE)); // fetch the first word, shifted left
385 
386  word_type word = word1;
387 
388  if (n_symbols > r1) // do we need to read another word?
389  {
390  const word_type word2 = base_words[ k1+1u ]; // fetch the second word
391  word |= word2 >> (r1*SYMBOL_SIZE); // shift by n1 symbols to the right
392  }
393 
394  word >>= (STORAGE_BITS - WORD_BITS); // align the top to WORD_BITS
395 
396  // clear every symbol we don't need among the word's LSD
397  word &= clearmask<word_type>( WORD_BITS - n_symbols*SYMBOL_SIZE );
398 
399  if (DOLLAR_BITS)
400  {
401  // encode the dollar's position in the least significant bits of the word
402  const word_type dollar_offset =
403  range_len <= SYMBOLS_PER_WORD ? // is there a dollar sign?
404  range_len :
405  (1u << DOLLAR_BITS)-1u; // no dollar sign in this word
406 
407  return word | dollar_offset;
408  }
409  else
410  return word;
411 }
