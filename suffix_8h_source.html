425 template <typename string_set_type>
426 int32 compare_suffixes(
427  const string_set_type string_set,
428  const uint2 suffix1,
429  const uint2 suffix2)
430 {
431  typedef typename string_set_type::string_type string_type;
432  typedef Suffix<string_type,uint32> suffix_type;
433 
434  // check whether both inputs represent the same suffix
435  if (suffix1.y == suffix2.y &&
436  suffix1.x == suffix2.x)
437  return 0;
438 
439  const suffix_type string1 = make_suffix( string_set[suffix1.y], suffix1.x );
440  const suffix_type string2 = make_suffix( string_set[suffix2.y], suffix2.x );
441 
442  const uint32 len1 = nvbio::length( string1 );
443  const uint32 len2 = nvbio::length( string2 );
444 
445  const uint32 min_len = nvbio::min( len1, len2 );
446 
447  // compare character by character
448  int32 cmp = 0;
449  for (uint32 j = 0; j < min_len; ++j)
450  {
451  const uint8 c_i = string1[j];
452  const uint8 c_n = string2[j];
453  if (c_i < c_n)
454  {
455  cmp = -1;
456  break;
457  }
458  if (c_i > c_n)
459  {
460  cmp = 1;
461  break;
462  }
463  }
464 
465  // break ties...
466  if (cmp == 0)
467  {
468  if (len1 < len2) // $ is smaller than any other character => s1 < s2
469  cmp = -1;
470  else if (len2 < len1) // $ is smaller than any other character => s2 < s1
471  cmp = 1;
472  else // $_1 < $_2 ? -1 : 1
473  cmp = suffix1.y < suffix2.y ? -1 : 1;
474  }
475  return cmp;
476 }
