21 //
22 // Build the occurrence table for a given string, packing a set of counters
23 // every K elements.
24 // The table must contain ((n+K-1)/K)*4 entries.
25 //
26 // Optionally save the table of the global counters as well.
27 //
28 // \param begin symbol sequence begin
29 // \param end symbol sequence end
30 // \param occ output occurrence map
31 // \param cnt optional table of the global counters
32 //
33 template <uint32 K, typename SymbolIterator, typename IndexType>
34 void build_occurrence_table(
35  SymbolIterator begin,
36  SymbolIterator end,
37  IndexType* occ,
38  IndexType* cnt)
39 {
40  IndexType counters[4] = { 0u, 0u, 0u, 0u };
41 
42  const IndexType n = end - begin;
43 
44  for (IndexType i = 0; i < n; ++i)
45  {
46  const IndexType i_mod_K = i & (K-1);
47 
48  if (i_mod_K == 0)
49  {
50  // save the counters
51  const uint32 k = i / K;
52  for (uint32 c = 0; c < 4; ++c)
53  occ[ k*4 + c ] = counters[c];
54  }
55 
56  // update counters
57  ++counters[ begin[i] ];
58  }
59 
60  if (cnt)
61  {
62  // build a cumulative table of the final counters
63  for (uint32 i = 0; i < 4; ++i)
64  cnt[i] = counters[i];
65  }
66 }

219  // pop-count the occurrences of symbols in two given text blocks, switching
220  // between pop-counting a single symbol if T is a uint32, or all 4 symbols
221  // if T is the count-table.
222  template <typename T>
223  static NVBIO_FORCEINLINE NVBIO_HOST_DEVICE uint2 popc2(
224  const TextStorage text,
225  const range_type range,
226  const uint32 kl,
227  const uint32 kh,
228  const T c)
229  {
230  const uint32 ml = (range.x - kl*K) >> LOG_SYMS_PER_WORD;
231  const uint32 mh = (range.y - kh*K) >> LOG_SYMS_PER_WORD;
232 
233  const word_type l_mod = ~word_type(range.x) & (SYMS_PER_WORD-1);
234  const word_type h_mod = ~word_type(range.y) & (SYMS_PER_WORD-1);
235 
236  const uint32 offl = kl*(K >> LOG_SYMS_PER_WORD);
237 
238  // sum up all the pop-counts of the relevant masks, up to ml-1
239  uint32 xl = occ::popc_2bit( text, c, offl, offl + ml );
240 
241  // check whether we can share the base value for the entire range
242  uint32 xh = (kl == kh) ? xl : 0u;
243  uint32 startm = (kl == kh) ? ml+1 : 0u;
244 
245  const word_type mask = text[ offl + ml ];
246  xl += occ::popc_2bit( mask, c, l_mod );
247 
248  // if the range fits in a single block, add the last mask to xh
249  if (kl == kh)
250  xh += occ::popc_2bit( mask, c, (mh == ml) ? h_mod : 0u );
251 
252  // finish computing the end of the range
253  if (kl != kh || mh > ml)
254  {
255  const uint32 offh = kh*(K >> LOG_SYMS_PER_WORD);
256  xh += occ::popc_2bit( text, c, offh + startm, offh + mh, h_mod );
257  }
258  return make_uint2( xl, xh );
259  }

347  // pop-count the occurrences of symbols in a given text block, switching
348  // between pop-counting a single symbol if T is a uint32, or all 4 symbols
349  // if T is the count-table.
350  template <typename T>
351  static NVBIO_FORCEINLINE NVBIO_HOST_DEVICE uint32 popc(
352  const TextStorage text,
353  const uint32 i,
354  const uint32 k,
355  const T c)
356  {
357  const uint32 m = (i - k*K) >> 4u;
358  const uint32 i_16 = ~i&15;
359 
360  // sum up all the pop-counts of the relevant masks
361  const uint4 masks = text[ k ];
362  uint32 x = 0;
363  if (m > 0) x += occ::popc_2bit( masks.x, c );
364  if (m > 1) x += occ::popc_2bit( masks.y, c );
365  if (m > 2) x += occ::popc_2bit( masks.z, c );
366 
367  // fetch the m-th mask and pop-count its nucleotides only up to i_mod_16
368  return x + occ::popc_2bit( comp( masks, m ), c, i_16 );
369  }

371  // pop-count the occurrences of symbols in two given text blocks, switching
372  // between pop-counting a single symbol if T is a uint32, or all 4 symbols
373  // if T is the count-table.
374  template <typename T>
375  static NVBIO_FORCEINLINE NVBIO_HOST_DEVICE uint2 popc2(
376  const TextStorage text,
377  const uint2 range,
378  const uint32 kl,
379  const uint32 kh,
380  const T c)
381  {
382  const uint32 ml = (range.x - kl*K) >> 4u;
383  const uint32 mh = (range.y - kh*K) >> 4u;
384 
385  const uint32 l_16 = ~range.x & 15;
386  const uint32 h_16 = ~range.y & 15;
387 
388  // fetch the 128-bit text masks for the respective blocks
389  const uint4 masks_l = text[ kl ];
390  const uint4 masks_h = (kl == kh) ? masks_l : text[ kh ];
391 
392  // sum up all the pop-counts of the relevant masks
393  uint32 xl = 0u;
394  if (ml > 0) xl += occ::popc_2bit( masks_l.x, c );
395  if (ml > 1) xl += occ::popc_2bit( masks_l.y, c );
396  if (ml > 2) xl += occ::popc_2bit( masks_l.z, c );
397 
398  uint32 xh = (kl == kh) ? xl : 0u;
399  uint32 startm = (kl == kh) ? ml : 0u;
400 
401  // sum up all the pop-counts of the relevant masks
402  if (mh > 0 && startm == 0) xh += occ::popc_2bit( masks_h.x, c );
403  if (mh > 1 && startm <= 1) xh += occ::popc_2bit( masks_h.y, c );
404  if (mh > 2 && startm <= 2) xh += occ::popc_2bit( masks_h.z, c );
405 
406  xl += occ::popc_2bit( comp( masks_l, ml ), c, l_16 );
407  xh += occ::popc_2bit( comp( masks_h, mh ), c, h_16 );
408  return make_uint2( xl, xh );
