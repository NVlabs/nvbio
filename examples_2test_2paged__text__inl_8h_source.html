48 template <typename input_storage, typename output_storage, uint32 SYMBOL_SIZE, typename index_type>
49 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
50 void copy(
51  const uint32 len,
52  const PackedStream<input_storage,uint8,SYMBOL_SIZE,true,index_type> in,
53  PackedStream<output_storage,uint8,SYMBOL_SIZE,true,index_type> out)
54 {
55  const uint32 SYMBOLS_PER_WORD = 32u / SYMBOL_SIZE;
56 
57  input_storage in_words = in.stream();
58  output_storage out_words = out.stream();
59 
60  const index_type k_start = in.index();
61  const index_type n_start = out.index();
62 
63  index_type k = k_start;
64  index_type n = n_start;
65 
66  const index_type k_end = k + len;
67 
68  const index_type out_word_begin = util::divide_ri( k, SYMBOLS_PER_WORD );
69  const index_type out_word_end = util::divide_rz( k_end, SYMBOLS_PER_WORD );
70 
71  // check whether the whole segment is contained in one word
72  if (out_word_end <= out_word_begin)
73  {
74  while (k < k_end)
75  out[k++ - k_start] = in[n++ - n_start];
76 
77  return;
78  }
79 
80  // align k to a word boundary
81  while (k < out_word_begin*SYMBOLS_PER_WORD)
82  out[k++ - k_start] = in[n++ - n_start];
83 
84  for (index_type out_word = out_word_begin; out_word < out_word_end; ++out_word)
85  {
86  // fetch a word's worth of input, starting from n
87  const uint32 n_word = n / SYMBOLS_PER_WORD;
88  const uint32 n_syms = SYMBOLS_PER_WORD - (n & (SYMBOLS_PER_WORD-1));
89 
90  uint32 in_word = in_words[n_word] >> (32 - n_syms * SYMBOL_SIZE);
91 
92  if (n_syms < SYMBOLS_PER_WORD)
93  {
94  const uint32 r_syms = SYMBOLS_PER_WORD - n_syms;
95  in_word |= (in_words[n_word+1] >> (32 - r_syms * SYMBOL_SIZE)) << (n_syms * SYMBOL_SIZE);
96  }
97 
98  // ...and write it out
99  out_words[ out_word ] = in_word;
100 
101  // go forward
102  k += SYMBOLS_PER_WORD;
103  n += SYMBOLS_PER_WORD;
104  }
105 
106  // finish copying leftovers
107  while (k < k_end)
108  out[k++ - k_start] = in[n++ - n_start];
109 }

319 //
320 template <uint32 SYMBOL_SIZE_T, bool BIG_ENDIAN_T>
321 uint64 PagedText<SYMBOL_SIZE_T,BIG_ENDIAN_T>::rank(const uint64 i, const uint32 c) const
322 {
323  // fetch the page containing index 'i'
324  const uint32 page_idx = find_page( i );
325  const uint32* page = get_page( page_idx );
326  const uint32* occ = page + m_page_size;
327 
328  const uint32 local_i = uint32( i - m_offsets[ page_idx ] );
329  assert( local_i < m_page_size * SYMBOLS_PER_WORD );
330 
331  // comoute the index of the occurrence block containing 'i', and the offset within it
332  const uint32 block_idx = local_i / m_occ_intv;
333  const uint32 block_offset = local_i & (m_occ_intv-1);
334 
335  // fetch the base occurrence counters for the page and block
336  uint64 out =
337  m_counters[ SYMBOL_COUNT * page_idx + c ] +
338  occ[ SYMBOL_COUNT * block_idx + c ];
339 
340  // compute the index of the word containing 'i', and the corresponding modulo
341  const uint32 word_idx = block_offset / SYMBOLS_PER_WORD;
342  const uint32 word_mod = ~word_type(local_i) & (SYMBOLS_PER_WORD-1);
343 
344  const uint32 word_begin = block_idx*m_occ_intv_w;
345 
346  // sum up all the pop-counts of the relevant masks
347  for (uint32 j = word_begin; j < word_begin + word_idx; ++j)
348  out += popc_2bit( page[j], c );
349 
350  return out + popc_2bit( page[ word_begin + word_idx ], c, word_mod );
351 }

705  // compute the original leaf sizes
706  thrust::adjacent_difference(

718  }

722  // new leaves
723  //

741  const float t1 = timer.seconds();

748

777  100.0f * t0 / (t0 + t1 + t2 + t3),
778  100.0f * t1 / (t0 + t1 + t2 + t3),
779  100.0f * t2 / (t0 + t1 + t2 + t3),
780  100.0f * t3 / (t0 + t1 + t2 + t3));
781

783  m_new_offsets[ out_leaves ] = m_offsets[ n_leaves ] + n;
784 
785  // swap-in the new pages
786  m_pages.swap( m_new_pages );
787  m_offsets.swap( m_new_offsets );

790  uint64* cnts = &m_counters[0];
791  for (uint32 i = 0; i < out_leaves-1; ++i)
792  {
793  for (uint32 j = 0; j < SYMBOL_COUNT; ++j)
794  cnts[SYMBOL_COUNT + j] += cnts[j];
795 
796  cnts += SYMBOL_COUNT;
797  }
798

801  for (uint32 j = 0; j < SYMBOL_COUNT; ++j)
802  n_occ += cnts[j];
803 
804  if (n_occ != m_offsets[ out_leaves ])
805  {
806  log_error(stderr, "mismatching occurrence counters: expected %llu symbols, got %llu\n",
807  m_offsets[ out_leaves ], n_occ );
808  }
