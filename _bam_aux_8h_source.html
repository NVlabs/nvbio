101  template <typename T>
102  bool GetTag(const char* tagName, T& tagVal) const
103  {
104  if ( TagData.empty() ) { return false; }
105 
106  // localize the tag data
107  const char* pTagData = (const char*)TagData.c_str();
108  const char* pTagStorageType = NULL;
109  const unsigned int tagDataLen = (unsigned int)TagData.size() + 1;
110  unsigned int numBytesParsed = 0;
111 
112  bool foundTag = false;
113  while( numBytesParsed < tagDataLen )
114  {
115  // skip white spaces
116  while ( (*pTagData == ' ' || *pTagData == '\n') && numBytesParsed < tagDataLen)
117  {
118  ++numBytesParsed;
119  ++pTagData;
120  }
121  if (*pTagData == '\0' || numBytesParsed == tagDataLen)
122  break;
123 
124  const char* pTagType = pTagData;
125  pTagStorageType = pTagData + 2;
126  pTagData += 3;
127  numBytesParsed += 3;
128 
129  // check the current tag
130  if (strncmp(pTagType, tagName, 2) == 0) {
131  foundTag = true;
132  break;
133  }
134 
135  // get the storage class and find the next tag
136  SkipToNextTag( pTagType, *pTagStorageType, pTagData, numBytesParsed );
137  }
138  // return if the edit distance tag was not present
139  if ( !foundTag ) { return false; }
140 
141  // assign the tag value
142  return GetTagValue( tagName, pTagData, *pTagStorageType, &tagVal );
143  }

306  static void SkipToNextTag(const char* tagName, const char storageType, const char* &pTagData, unsigned int& numBytesParsed)
307  {
308  switch(storageType) {
309  case 'A':
310  case 'c':
311  case 'C':
312  ++numBytesParsed;
313  ++pTagData;
314  break;
315 
316  case 'B':
317  {
318  const char type = *pTagData++;
319  const uint8_t len0 = *pTagData++;
320  const uint8_t len1 = *pTagData++;
321  const uint8_t len2 = *pTagData++;
322  const uint8_t len3 = *pTagData++;
323  const uint32_t len = len0 | (len1 << 8) | (len2 << 16) | (len3 << 24);
324 
325  switch (type)
326  {
327  case 'c':
328  case 'C':
329  pTagData += len;
330  break;
331  case 's':
332  case 'S':
333  pTagData += len * 2;
334  break;
335  case 'i':
336  case 'I':
337  case 'f':
338  pTagData += len * 4;
339  break;
340  }
341  break;
342  }
343 
344  case 's':
345  case 'S':
346  numBytesParsed += 2;
347  pTagData += 2;
348  break;
349 
350  case 'i':
351  case 'f':
352  case 'I':
353  numBytesParsed += 4;
354  pTagData += 4;
355  break;
356 
357  case 'Z':
358  case 'H':
359  while(*pTagData != '\0') {
360  ++numBytesParsed;
361  ++pTagData;
362  }
363  if(*pTagData == '\0')
364  {
365  ++numBytesParsed;
366  ++pTagData;
367  }
