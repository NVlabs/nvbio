192  void insert(uint32 i, const Symbol s)
193  {
194  //fprintf(stderr,"insert %u (root: %u:%u)\n", i, m_root, m_nodes[ m_root ].weight());
195  uint32 parent_index = INVALID_NODE;
196  uint32 node_index = m_root;
197  uint32 node_base = 0u;
198 
199  while (1u)
200  {
201  if (m_nodes[ node_index ].size() == FAN_OUT)
202  {
203  // create a new root if necessary
204  if (parent_index == INVALID_NODE)
205  {
206  m_root = uint32( m_nodes.size() );
207  m_nodes.resize( m_root + 1u + FAN_OUT );
208  m_nodes[m_root].set_size( 1u );
209  m_nodes[m_root].set_child( m_root + 1u );
210  m_nodes[m_root].set_weight( m_nodes[ node_index ].weight() + 1u );
211 
212  // copy the old root into the first child of the new one
213  memcpy( &m_nodes[ m_root + 1u ], &m_nodes[ node_index ], sizeof(node_type) );
214 
215  m_nodes[ m_root + 1u ].set_parent( m_root );
216 
217  parent_index = m_root;
218  node_index = m_root + 1u;
219  node_base = m_root + 1u;
220  }
221 
222  // split the node
223  split_node( parent_index, node_index );
224 
225  // i might now be in another sibling of this node, find it
226  const uint32 last_node = node_base + m_nodes[ parent_index ].size() - 1u;
227  while (node_index < last_node)
228  {
229  // check if we reached the proper sibing
230  if (i < m_nodes[ node_index ].weight())
231  break;
232 
233  // continue the search
234  i -= m_nodes[ node_index++ ].weight();
235  }
236  }
237 
238  if (m_nodes[ node_index ].is_leaf())
239  {
240  const uint32 leaf_index = m_nodes[ node_index ].leaf();
241 
242  // check whether this leaf is already full
243  if (m_leaves[ leaf_index ].size() == SYMBOLS_PER_LEAF)
244  {
245  const uint2 new_leaf = split_leaf( parent_index, node_index, leaf_index, SYMBOLS_PER_LEAF/2u );
246 
247  if (i < SYMBOLS_PER_LEAF/2)
248  {
249  m_leaves[ leaf_index ].insert( i, s );
250  m_nodes[ node_index ].add_weight( 1u );
251  return;
252  }
253  else
254  {
255  m_leaves[ new_leaf.x ].insert( i - SYMBOLS_PER_LEAF/2, s );
256  m_nodes[ new_leaf.y ].add_weight( 1u );
257  return;
258  }
259  }
260  else
261  {
262  // we can safely insert in the current leaf
263  m_leaves[ leaf_index ].insert( i, s );
264  m_nodes[ node_index ].add_weight( 1u );
265  return;
266  }
267  }
268  else
269  {
270  // increase the weight of each node we touch
271  m_nodes[ node_index ].add_weight( 1u );
272 
273  const uint32 n = m_nodes[ node_index ].size();
274 
275  // fetch the address of the first child
276  node_base = m_nodes[ node_index ].child();
277 
278  // find the proper child
279  for (uint32 c = 0; c < n; ++c)
280  {
281  const uint64 child_w = m_nodes[ node_base + c ].weight();
282 
283  // check whether i falls within this child
284  if (i < child_w || c == n-1)
285  {
286  // descend into this child
287  parent_index = node_index;
288  node_index = node_base + c;
289  break;
290  }
291 
292  i -= child_w;
293  }
294  }
295  }
296  }

307  uint32 split_node(const uint32 parent_idx, const uint32 node_idx)
308  {
309  // fetch the parent
310  node_type& parent = m_nodes[ parent_idx ];
311 
312  // fetch the new node from the parent
313  const uint32 new_node_idx = parent.child() + parent.size();
314 
315  // increase the parent size
316  parent.set_size( parent.size() + 1u );
317 
318  // reset the old node's size
319  m_nodes[ node_idx ].set_size( FAN_OUT / 2u );
320 
321  const uint32 old_children = m_nodes[ node_idx ].child();
322 
323  // recompute the old node's weight
324  {
325  uint64 w = 0u;
326  for (uint32 c = 0; c < FAN_OUT / 2u; ++c)
327  w += m_nodes[ old_children + c ].weight();
328 
329  m_nodes[ node_idx ].set_weight( w );
330  }
331 
332  // allocate space for the new node's children
333  const uint32 new_children = uint32( m_nodes.size() );
334  m_nodes.resize( new_children + FAN_OUT );
335 
336  memcpy( &m_nodes[ new_children ], &m_nodes[ old_children + FAN_OUT/2 ], sizeof(node_type)*FAN_OUT/2 );
337 
338  // setup the new node
339  m_nodes[ new_node_idx ].set_size( FAN_OUT / 2u );
340  m_nodes[ new_node_idx ].set_parent( parent_idx );
341  m_nodes[ new_node_idx ].set_child( new_children );
342 
343  // recompute the new node's weight
344  {
345  uint64 w = 0u;
346  for (uint32 c = 0; c < FAN_OUT / 2u; ++c)
347  w += m_nodes[ new_children + c ].weight();
348 
349  m_nodes[ new_node_idx ].set_weight( w );
350  }
351 
352  // return the index of the new node
353  return new_node_idx;
354  }

358  uint2 split_leaf(const uint32 parent_idx, const uint32 node_idx, const uint32 leaf_idx, const uint32 offset_idx)
359  {
360  // create a new leaf
361  const uint32 new_leaf_idx = uint32( m_leaves.size() );
362  //fprintf(file, " new leaf: %u\n", new_leaf_idx); fflush(file);
363  m_leaves.resize( new_leaf_idx + 1u );
364 
365  leaf_type& leaf = m_leaves[ leaf_idx ];
366  leaf_type& new_leaf = m_leaves[ new_leaf_idx ];
367 
368  const uint32 old_size = leaf.size();
369 
370  nvbio::assign( old_size - offset_idx, leaf.begin() + offset_idx, new_leaf.begin() );
371 
372  new_leaf.set_size( old_size - offset_idx );
373  leaf.set_size( offset_idx );
374 
375  // fetch the parent
376  node_type& parent = m_nodes[ parent_idx ];
377 
378  // fetch the new node parenting the new leaf
379  const uint32 new_node_idx = parent.child() + parent.size();
380 
381  // increase the parent size
382  parent.set_size( parent.size() + 1u );
383 
384  m_nodes[ new_node_idx ].set_size( 0u );
385  m_nodes[ new_node_idx ].set_parent( parent_idx );
386  m_nodes[ new_node_idx ].set_leaf( new_leaf_idx );
387  m_nodes[ new_node_idx ].set_weight( old_size - offset_idx );
388 
389  // reset the weight of the old leaf's node
390  m_nodes[ node_idx ].set_weight( offset_idx );
391 
392  // return the index of the new leaf and that of its parent
