203  void locked_push(const WorkItem work)
204  {

206  m_queue.push( work ); m_size++;
207  }
208 
210  bool pop(WorkItem& work)
211  {
212  ScopedLock block( &m_lock );
213  if (m_queue.empty())
214  return false;
215 
216  work = m_queue.front();
217  m_queue.pop();
218 
219  m_callback( m_size -( uint32)m_queue.size() - 1u, m_size );
220  return true;
221  }
222

225

234 inline uint32 balance_batch_size(uint32 batch_size, uint32 total_count, uint32 thread_count)
235 {
236  // How many batches we'd get with the proposed batch_size
237  const uint32 batch_count = util::divide_ri(total_count, batch_size);
238  // How many rounds we'd need for those batches
239  const uint32 rounds = util::divide_ri(batch_count, thread_count);
240  // Might as well assume all threads should work, and see how many batches
241  // they would consume
242  const uint32 bal_batches = rounds * thread_count;
243  // So that the batch size that will attain it, is computed as follows
244  return util::divide_ri(total_count, bal_batches);
245 }
