96  bool process(PipelineContext& context)
97  {
98  m_data->m_mutex.lock();
99 
100  Timer timer;
101  timer.start();
102 
103  // fetch the output
104  io::SequenceDataHost* h_read_data = context.output<io::SequenceDataHost>();
105 
106  const int ret = io::next( DNA_N, h_read_data, m_data->m_file, m_data->m_max_strings, m_data->m_max_bps );
107 
108  timer.stop();
109  m_data->m_time += timer.seconds();
110  m_data->m_reads += h_read_data->size();
111  m_data->m_bps += h_read_data->bps();
112 
113  if (h_read_data->max_sequence_len() > MAX_READ_LENGTH)
114  {
115  log_error(stderr, " maximum read length exceeded: %u > %u\n", h_read_data->max_sequence_len(), MAX_READ_LENGTH);
116  return false;
117  }
118 
119  log_verbose(stderr, "\r loaded reads [%llu, %llu] (%.1fM / %.2fG bps, %.1fK reads/s, %.1fM bps/s) ",
120  m_data->m_reads,
121  m_data->m_reads + h_read_data->size(),
122  1.0e-6f * (h_read_data->bps()),
123  1.0e-9f * (m_data->m_bps + h_read_data->bps()),
124  m_data->m_time ? (1.0e-3f * (m_data->m_reads + h_read_data->size())) / m_data->m_time : 0.0f,
125  m_data->m_time ? (1.0e-6f * (m_data->m_bps + h_read_data->bps() )) / m_data->m_time : 0.0f );
126  log_debug_cont(stderr, "\n");
127 
128  m_data->m_mutex.unlock();
129 
130  return ret;
131  }
