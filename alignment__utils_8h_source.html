61 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
62 void frame_opposite_mate(
63  const int policy,
64  const uint32 anchor,
65  const bool anchor_fw,
66  bool& left,
67  bool& fw)
68 {
69  const bool anchor_1 = (anchor == 0);
70 
71  switch(policy)
72  {
73  case io::PE_POLICY_FF:
74  {
75  left = (anchor_1 != anchor_fw);
76  fw = anchor_fw;
77  break;
78  }
79  case io::PE_POLICY_RR:
80  {
81  left = (anchor_1 == anchor_fw);
82  fw = anchor_fw;
83  break;
84  }
85  case io::PE_POLICY_FR:
86  {
87  left = !anchor_fw;
88  fw = !anchor_fw;
89  break;
90  }
91  case io::PE_POLICY_RF:
92  {
93  left = anchor_fw;
94  fw = !anchor_fw;
95  break;
96  }
97  }
98 }

103 NVBIO_FORCEINLINE NVBIO_HOST_DEVICE
104 int32 compute_target_score(const io::BestPairedAlignments& best, const int32 a_worst_score, const int32 o_worst_score)
105 {
106  if (best.has_second_paired() == false)
107  return a_worst_score + o_worst_score;
108 
109  // reproduce Bowtie2's score bounding behavior (under its default setting 'tighten = 3')
110  const int32 delta = best.best_score() - best.second_score();
111  return best.second_score() + delta * 3/4;
112 }

124 template <typename vector>
125 struct Backtracker
126 {
127  // constructor
128  NVBIO_FORCEINLINE NVBIO_DEVICE
129  Backtracker(vector vec, const uint32 _capacity) : out(vec), size(0), prev(255), capacity(_capacity) {}
130 
131  // encode a soft clipping operation
132  NVBIO_FORCEINLINE NVBIO_DEVICE
133  void clip(const uint32 l)
134  {
135  if (l)
136  {
137  NVBIO_CUDA_DEBUG_ASSERT( size < capacity, "Backtracker: exceeded maximum CIGAR size!\n" );
138  out[size++] = io::Cigar( io::Cigar::SOFT_CLIPPING, l );
139  }
140  }
141 
142  // encode a new operation
143  NVBIO_FORCEINLINE NVBIO_DEVICE
144  void push(uint8 type)
145  {
146  NVBIO_CUDA_DEBUG_ASSERT( type == aln::SUBSTITUTION ||
147  type == aln::DELETION ||
148  type == aln::INSERTION,
149  "Backtracker: unknown op type %u\n", uint32(type) );
150 
151  // perform run-length encoding of the symbol sequence
152  if (prev == type)
153  out[size-1u].m_len++;
154  else
155  {
156  NVBIO_CUDA_DEBUG_ASSERT( size < capacity, "Backtracker: exceeded maximum CIGAR size!\n" );
157  // encode a new symbol
158  out[size++] = io::Cigar( type, 1u );
159  prev = type;
160  }
161  }
162 
163  vector out;
164  uint32 size;
165  uint8 prev;
166  uint32 capacity;
167 };

195  template <typename context_type>
196  NVBIO_HOST_DEVICE
197  void load(const PipelineType& pipeline, const context_type* context)
198  {
199  // select the read batch based on context->mate
200  read_batch_type reads = context->mate ?
201  pipeline.reads_o :
202  pipeline.reads;
203 
204  const DirType read_dir = context->read_rc ? FORWARD : REVERSE;
205  const ReadType read_type = context->read_rc ? COMPLEMENT : STANDARD;
206 
207  pattern = pattern_loader.load(
208  reads,
209  context->read_range,
210  read_dir, // the reads are loaded in REVERSE fashion, so we invert this flag here
211  read_type );
212 
213  quals = pattern.qualities();
214  text = text_string(
215  context->genome_end - context->genome_begin,
216  text_loader.load( pipeline.genome + context->genome_begin, context->genome_end - context->genome_begin ) );
217  }
218

220  text_loader_type text_loader;
221  pattern_string pattern;
222  qual_string quals;
223  text_string text;
224 };
