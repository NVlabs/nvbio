556  // reserve enough storage for a given number of reads and bps
557  //
558  void reserve(const uint32 n_seqs, const uint32 n_bps)
559  {
560  // a default read id length used to reserve enough space upfront and avoid frequent allocations
561  const uint32 AVG_NAME_LENGTH = 250;
562 
563  const uint32 bps_per_word = 32u / bits_per_symbol( SequenceDataInfo::m_alphabet );
564 
565  m_sequence_index_vec.reserve( n_seqs+1 );
566  m_sequence_vec.reserve( n_bps / bps_per_word );
567  m_qual_vec.reserve( n_bps );
568  m_name_index_vec.reserve( AVG_NAME_LENGTH * n_seqs );
569  m_name_index_vec.reserve( n_seqs+1 );
570  }
