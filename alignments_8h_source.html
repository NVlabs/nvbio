82  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE Alignment() {}
83  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE Alignment(
84  const uint32 pos,
85  const uint32 ed,
86  const int32 score,
87  const uint32 rc,
88  const uint32 mate = 0,
89  const bool paired = false)
90  {
91  //NVBIO_CUDA_ASSERT( ed < 1024u );
92  //NVBIO_CUDA_ASSERT( score >= min_score() && score <= max_score() );
93  m_align = pos;
94  m_ed = ed;
95  m_score = score < 0 ? uint32( -score ) : uint32( score );
96  m_score_sgn = score < 0 ? 1u : 0u;
97  m_rc = rc;
98  m_mate = mate;
99  m_paired = paired ? 1u : 0u;
100  }

102  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE int32 score() const { return m_score_sgn ? -int32( m_score ) : int32( m_score ); }
103  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_aligned() const { return m_align != uint32(-1); }
104  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 alignment() const { return m_align; }
105  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_rc() const { return m_rc; }
106  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 ed() const { return m_ed; }
107  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 sink() const { return m_ed; }
108  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE uint32 mate() const { return m_mate; }
109  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_paired() const { return (m_paired == true) && is_aligned(); }
110  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE bool is_unpaired() const { return (m_paired == false) && is_aligned(); }
111 
112  NVBIO_HOST_DEVICE NVBIO_FORCEINLINE
113  static Alignment invalid() { return Alignment( uint32(-1), max_ed(), max_score(), 0u, 0u, false ); }
