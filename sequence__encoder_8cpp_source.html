123 template <SequenceDataEncoder::StrandOp FLAGS>
124 struct sequence_string
125 {
126  // constructor
127  sequence_string(const uint32 len, const uint8* sequence, const uint8* qual) : m_len(len), m_sequence(sequence), m_qual(qual) {}
128 
129  // string length
130  uint32 length() const { return m_len; }
131 
132  // indexing operator
133  uint8 operator[] (const uint32 i) const
134  {
135  const uint32 index = (FLAGS & SequenceDataEncoder::REVERSE_OP) ? m_len - i - 1u : i;
136 
137  // fetch the bp
138  const uint8 bp = nst_nt4_encode( m_sequence[index] );
139 
140  if (FLAGS & SequenceDataEncoder::COMPLEMENT_OP)
141  return bp < 4u ? 3u - bp : 4u;
142 
143  return bp;
144  }
145 
146  // quality operator
147  uint8 quality(const uint32 i) const
148  {
149  const uint32 index = (FLAGS & REVERSE) ? m_len - i - 1u : i;
150 
151  return m_qual[index];
152  }

155  const uint8* m_sequence;
156  const uint8* m_qual;

215 // encode a sequence according to some given run-time flags and quality-encoding
216 //
217 template <Alphabet ALPHABET>
218 void encode(
219  const SequenceDataEncoder::StrandOp conversion_flags,
220  const QualityEncoding quality_encoding,
221  const uint32 sequence_len,
222  const uint8* sequence,
223  const uint8* quality,
224  typename SequenceDataEdit<ALPHABET,SequenceDataView>::sequence_stream_type stream,
225  char* qual_stream)
226 {
227 
228  const sequence_string<SequenceDataEncoder::REVERSE_OP> r_sequence( sequence_len, sequence, quality );
229  const sequence_string<SequenceDataEncoder::REVERSE_COMPLEMENT_OP> rc_sequence( sequence_len, sequence, quality );
230  const sequence_string<SequenceDataEncoder::COMPLEMENT_OP> fc_sequence( sequence_len, sequence, quality );
231  const sequence_string<SequenceDataEncoder::NO_OP> f_sequence( sequence_len, sequence, quality );
232 
233  if (conversion_flags & SequenceDataEncoder::REVERSE_OP)
234  {
235  if (conversion_flags & SequenceDataEncoder::COMPLEMENT_OP)
236  encode<ALPHABET>( quality_encoding, rc_sequence, stream, qual_stream );
237  else
238  encode<ALPHABET>( quality_encoding, r_sequence, stream, qual_stream );
239  }
240  else
241  {
242  if (conversion_flags & SequenceDataEncoder::COMPLEMENT_OP)
243  encode<ALPHABET>( quality_encoding, fc_sequence, stream, qual_stream );
244  else
245  encode<ALPHABET>( quality_encoding, f_sequence, stream, qual_stream );
246  }

314  void push_back(
315  const uint32 in_sequence_len,
316  const char* name,
317  const uint8* base_pairs,
318  const uint8* quality,
319  const QualityEncoding quality_encoding,
320  const uint32 max_sequence_len,
321  const uint32 trim3,
322  const uint32 trim5,

326  in_sequence_len - trim3 - trim5 : 0u;

334  assert(sequence_len);

336  // resize the sequences & quality buffers
337  {
338  static const uint32 bps_per_word = 32u / SEQUENCE_BITS;
339  const uint32 stream_len = m_data->m_sequence_stream_len + sequence_len;
340  const uint32 words = util::divide_ri( stream_len, bps_per_word );
341

354  quality_encoding,
355  sequence_len,
356  base_pairs,

385  m_data->m_name_index_vec[ m_data->m_n_seqs ] = m_data->m_name_stream_len;
