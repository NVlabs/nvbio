25 // \return # of allocated bytes
26 inline
27 uint64 SeedHitDequeArray::resize(const uint32 n_reads, const uint32 max_hits, const bool do_alloc)
28 {
29  uint64 bytes = 0u;
30  uint32 max_nodes = SumTree<float*>::node_count( max_hits );
31  if (do_alloc) m_counts.resize( n_reads ); bytes += n_reads * sizeof(uint32);
32  if (do_alloc) m_index.resize( n_reads ); bytes += n_reads * sizeof(uint32);
33  if (do_alloc) m_hits.resize( n_reads * max_hits ); bytes += n_reads * max_hits * sizeof(SeedHit);
34  if (do_alloc) m_probs.resize( n_reads * max_nodes ); bytes += n_reads * max_nodes * sizeof(float);
35  if (do_alloc) m_pool.resize( 1, 0u ); bytes += sizeof(uint32);
36 
37  if (do_alloc) thrust::fill( m_counts.begin(), m_counts.end(), uint32(0) );
38  if (do_alloc) thrust::fill( m_index.begin(), m_index.end(), uint32(0) );
39  return bytes;
40 }
